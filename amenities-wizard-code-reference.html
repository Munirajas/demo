Amenities creation/edit wizard — component reference

Drop-in components for the Add/Edit Amenity drawer. Existing server-amenities.ts, amenity.mutations.ts, amenity.queries.ts, and the three hooks (useInsertAmenity, useUpdateAmenity, useRemoveAmenity) are untouched — only the drawer's internals changed, from a tab layout to a dropdown-first stepper. Each amenity type has its own form component so it can evolve independently; General Info is shared across both.

Open item for BE: per requirement #4, the FE generates and sends all expanded timestamps on save. Confirm whether BE also persists the original recurrence rule (start time, days-of-week, interval, end condition) — not just the flat timestamp list. Without that, an existing Scheduled Admission can't be reconstructed into an editable form; it can only be recreated from scratch. The drawer below has a clear TODO marking this gap.
types/wizard-types.ts

Shared types for the wizard: general info shape, schedule rule shape, the resolveWizardKind heuristic that decides General vs Scheduled from the selected amenity type, and the step labels per type.

import type { AmenityType } from '@/types/amenities';

export const GENERAL_ADMISSION_KEY = 'GENERAL_ADMISSION' as const;
export const SCHEDULED_ADMISSION_KEY = 'SCHEDULED_ADMISSION' as const;

export type AmenityTypeKey =
  | typeof GENERAL_ADMISSION_KEY
  | typeof SCHEDULED_ADMISSION_KEY;

export interface GeneralInfoValues {
  amenityTypeId: number | '';
  amenityName: string;
  amenityDescription: string;
}

export const EMPTY_GENERAL_INFO: GeneralInfoValues = {
  amenityTypeId: '',
  amenityName: '',
  amenityDescription: '',
};

export type RecurrenceFrequency = 'NONE' | 'DAILY' | 'WEEKLY';
export type WeekDay = 'SUN' | 'MON' | 'TUE' | 'WED' | 'THU' | 'FRI' | 'SAT';

export interface ScheduleRuleValues {
  startTime: string; // "HH:mm"
  endTime: string; // "HH:mm"
  frequency: RecurrenceFrequency;
  interval: number; // recur every N days/weeks
  daysOfWeek: WeekDay[]; // used when frequency === 'WEEKLY'
  startDate: string; // "YYYY-MM-DD"
  endMode: 'ON_DATE' | 'AFTER_OCCURRENCES';
  endDate?: string; // "YYYY-MM-DD"
  occurrences?: number;
}

export const EMPTY_SCHEDULE_RULE: ScheduleRuleValues = {
  startTime: '15:00',
  endTime: '15:30',
  frequency: 'WEEKLY',
  interval: 1,
  daysOfWeek: ['SUN', 'MON', 'TUE', 'WED', 'THU', 'FRI', 'SAT'],
  startDate: '',
  endMode: 'ON_DATE',
  endDate: '',
  occurrences: undefined,
};

export interface GeneratedTimestamp {
  startDateTime: string; // ISO 8601
  endDateTime: string; // ISO 8601
}

// Final shape handed back up to AmenityList's handleSave.
// NOTE: field names below (schedule, timestamps) are placeholders --
// confirm exact keys against InsertAmenityInput / UpdateAmenityInput
// once BE shares the schema.
export interface AmenitySavePayload {
  amenityTypeId: number;
  amenityName: string;
  amenityDescription: string;
  schedule?: {
    rule: ScheduleRuleValues;
    timestamps: GeneratedTimestamp[];
  };
}

// TEMPORARY heuristic. Replace with a stable discriminant from BE
// (e.g. amenityType.category === 'SCHEDULED') as soon as it's available --
// matching on display name breaks the moment someone renames a type in admin.
export function resolveWizardKind(
  amenityTypeId: number | '',
  amenityTypes: AmenityType[],
): AmenityTypeKey {
  const match = amenityTypes.find((t) => t.amenityTypeId === amenityTypeId);
  const name = match?.amenityTypeName?.toLowerCase() ?? '';
  if (name.includes('schedul') || name.includes('slot')) {
    return SCHEDULED_ADMISSION_KEY;
  }
  return GENERAL_ADMISSION_KEY;
}

export const WIZARD_STEP_LABELS: Record<AmenityTypeKey, string[]> = {
  [GENERAL_ADMISSION_KEY]: ['General info'],
  [SCHEDULED_ADMISSION_KEY]: ['General info', 'Scheduling'],
};
lib/schedule-timestamp-generator.ts

Expands a recurrence rule into concrete ISO start/end timestamps using moment.js, per requirement #4. Includes a hard safety cap so a bad end-date input can't generate an unbounded list.

import moment from 'moment';
import type { ScheduleRuleValues, GeneratedTimestamp, WeekDay } from '../types/wizard-types';

const DAY_INDEX: Record<WeekDay, number> = {
  SUN: 0, MON: 1, TUE: 2, WED: 3, THU: 4, FRI: 5, SAT: 6,
};

// Hard safety cap so a mistyped end date (e.g. year 2099) can't hang the tab
// generating tens of thousands of rows client-side.
const MAX_OCCURRENCES_SAFEGUARD = 366;

function dayKeyFromIndex(index: number): WeekDay {
  return (Object.keys(DAY_INDEX) as WeekDay[]).find((k) => DAY_INDEX[k] === index) ?? 'SUN';
}

/**
 * Expands a recurrence rule into concrete start/end ISO timestamps.
 * This is a first-pass implementation -- the "every N weeks" interval jump
 * logic especially needs unit tests against real calendar edge cases
 * (month/year boundaries, DST) before shipping.
 */
export function generateScheduleTimestamps(rule: ScheduleRuleValues): GeneratedTimestamp[] {
  if (!rule.startDate) return [];

  const [startHour, startMinute] = rule.startTime.split(':').map(Number);
  const [endHour, endMinute] = rule.endTime.split(':').map(Number);

  const results: GeneratedTimestamp[] = [];
  const rangeStart = moment(rule.startDate, 'YYYY-MM-DD');
  const rangeEnd = rule.endMode === 'ON_DATE' && rule.endDate
    ? moment(rule.endDate, 'YYYY-MM-DD')
    : null;

  function pushOccurrence(day: moment.Moment) {
    const start = day.clone().hour(startHour).minute(startMinute).second(0).millisecond(0);
    const end = day.clone().hour(endHour).minute(endMinute).second(0).millisecond(0);
    results.push({ startDateTime: start.toISOString(), endDateTime: end.toISOString() });
  }

  if (rule.frequency === 'NONE') {
    pushOccurrence(rangeStart);
    return results;
  }

  let cursor = rangeStart.clone();
  let count = 0;
  const anchorWeekday = rangeStart.day();

  while (count < MAX_OCCURRENCES_SAFEGUARD) {
    if (rangeEnd && cursor.isAfter(rangeEnd, 'day')) break;
    if (rule.endMode === 'AFTER_OCCURRENCES' && rule.occurrences && count >= rule.occurrences) break;

    if (rule.frequency === 'DAILY') {
      pushOccurrence(cursor);
      count += 1;
      cursor = cursor.clone().add(rule.interval, 'days');
      continue;
    }

    // WEEKLY
    if (rule.daysOfWeek.includes(dayKeyFromIndex(cursor.day()))) {
      pushOccurrence(cursor);
      count += 1;
      if (rule.endMode === 'AFTER_OCCURRENCES' && rule.occurrences && count >= rule.occurrences) break;
    }

    cursor = cursor.clone().add(1, 'day');

    // Once we've scanned a full week, skip ahead (interval - 1) more weeks
    // to honor "recur every N weeks".
    if (cursor.day() === anchorWeekday && rule.interval > 1) {
      cursor = cursor.clone().add((rule.interval - 1) * 7, 'days');
    }
  }

  return results;
}

components/ScheduleBuilder.tsx

The recurrence UI: start/end date, start/end time, daily/weekly frequency, interval, day-of-week chips, and end condition (on date vs after N occurrences). Used only by Scheduled Admission.

'use client';

import type { ScheduleRuleValues, WeekDay } from '../types/wizard-types';

const ALL_DAYS: { key: WeekDay; label: string }[] = [
  { key: 'SUN', label: 'Sun' }, { key: 'MON', label: 'Mon' },
  { key: 'TUE', label: 'Tue' }, { key: 'WED', label: 'Wed' },
  { key: 'THU', label: 'Thu' }, { key: 'FRI', label: 'Fri' },
  { key: 'SAT', label: 'Sat' },
];

interface Props {
  values: ScheduleRuleValues;
  onChange: (values: ScheduleRuleValues) => void;
  error?: string | null;
}

export function ScheduleBuilder({ values, onChange, error }: Props) {
  function toggleDay(day: WeekDay) {
    const has = values.daysOfWeek.includes(day);
    onChange({
      ...values,
      daysOfWeek: has ? values.daysOfWeek.filter((d) => d !== day) : [...values.daysOfWeek, day],
    });
  }

  return (
    <div className="flex flex-col gap-4">
      {error && <p className="text-sm text-red-600">{error}</p>}

      <div className="grid grid-cols-2 gap-3">
        <div className="flex flex-col gap-1">
          <label className="text-sm font-medium text-gray-700">Start date</label>
          <input
            type="date"
            className="w-full rounded-md border border-gray-300 px-3 py-2 text-sm"
            value={values.startDate}
            onChange={(e) => onChange({ ...values, startDate: e.target.value })}
          />
        </div>
        <div className="flex flex-col gap-1">
          <label className="text-sm font-medium text-gray-700">Recurrence</label>
          <select
            className="w-full rounded-md border border-gray-300 px-3 py-2 text-sm"
            value={values.frequency}
            onChange={(e) => onChange({ ...values, frequency: e.target.value as ScheduleRuleValues['frequency'] })}
          >
            <option value="NONE">Does not repeat</option>
            <option value="DAILY">Daily</option>
            <option value="WEEKLY">Weekly</option>
          </select>
        </div>
      </div>

      <div className="grid grid-cols-2 gap-3">
        <div className="flex flex-col gap-1">
          <label className="text-sm font-medium text-gray-700">Start time</label>
          <input
            type="time"
            className="w-full rounded-md border border-gray-300 px-3 py-2 text-sm"
            value={values.startTime}
            onChange={(e) => onChange({ ...values, startTime: e.target.value })}
          />
        </div>
        <div className="flex flex-col gap-1">
          <label className="text-sm font-medium text-gray-700">End time</label>
          <input
            type="time"
            className="w-full rounded-md border border-gray-300 px-3 py-2 text-sm"
            value={values.endTime}
            onChange={(e) => onChange({ ...values, endTime: e.target.value })}
          />
        </div>
      </div>

      {values.frequency !== 'NONE' && (
        <div className="flex flex-col gap-1">
          <label className="text-sm font-medium text-gray-700">Recur every</label>
          <div className="flex items-center gap-2">
            <input
              type="number"
              min={1}
              className="w-20 rounded-md border border-gray-300 px-3 py-2 text-sm"
              value={values.interval}
              onChange={(e) => onChange({ ...values, interval: Number(e.target.value) || 1 })}
            />
            <span className="text-sm text-gray-600">{values.frequency === 'DAILY' ? 'day(s)' : 'week(s)'}</span>
          </div>
        </div>
      )}

      {values.frequency === 'WEEKLY' && (
        <div className="flex flex-col gap-1">
          <label className="text-sm font-medium text-gray-700">Repeat on</label>
          <div className="flex flex-wrap gap-2">
            {ALL_DAYS.map((day) => {
              const active = values.daysOfWeek.includes(day.key);
              return (
                <button
                  key={day.key}
                  type="button"
                  onClick={() => toggleDay(day.key)}
                  className={`text-xs px-3 py-1.5 rounded-full border ${
                    active ? 'bg-blue-600 text-white border-blue-600' : 'bg-white text-gray-600 border-gray-300'
                  }`}
                >
                  {day.label}
                </button>
              );
            })}
          </div>
        </div>
      )}

      {values.frequency !== 'NONE' && (
        <div className="flex flex-col gap-2">
          <label className="text-sm font-medium text-gray-700">Ends</label>
          <div className="flex items-center gap-2">
            <input
              type="radio"
              id="endOnDate"
              checked={values.endMode === 'ON_DATE'}
              onChange={() => onChange({ ...values, endMode: 'ON_DATE' })}
            />
            <label htmlFor="endOnDate" className="text-sm text-gray-700 w-16">On date</label>
            <input
              type="date"
              disabled={values.endMode !== 'ON_DATE'}
              className="flex-1 rounded-md border border-gray-300 px-3 py-2 text-sm disabled:bg-gray-50"
              value={values.endDate ?? ''}
              onChange={(e) => onChange({ ...values, endDate: e.target.value })}
            />
          </div>
          <div className="flex items-center gap-2">
            <input
              type="radio"
              id="endAfter"
              checked={values.endMode === 'AFTER_OCCURRENCES'}
              onChange={() => onChange({ ...values, endMode: 'AFTER_OCCURRENCES' })}
            />
            <label htmlFor="endAfter" className="text-sm text-gray-700 w-16">After</label>
            <input
              type="number"
              min={1}
              disabled={values.endMode !== 'AFTER_OCCURRENCES'}
              className="w-24 rounded-md border border-gray-300 px-3 py-2 text-sm disabled:bg-gray-50"
              value={values.occurrences ?? ''}
              onChange={(e) => onChange({ ...values, occurrences: Number(e.target.value) || undefined })}
            />
            <span className="text-sm text-gray-600">occurrences</span>
          </div>
        </div>
      )}
    </div>
  );

components/StepperHeader.tsx

Generic top-of-drawer stepper: "Step X of Y" label, progress bar, and clickable step chips. Steps marked pending aren't clickable, so a user can't jump into scheduling before general info is valid. Reusable for any future amenity type with more steps.

'use client';

export interface StepConfig {
  id: string;
  label: string;
  status: 'done' | 'active' | 'pending';
}

interface Props {
  steps: StepConfig[];
  currentIndex: number;
  onStepClick: (index: number) => void;
}

export function StepperHeader({ steps, currentIndex, onStepClick }: Props) {
  if (steps.length <= 1) return null;

  return (
    <div className="mb-4">
      <div className="flex items-center justify-between mb-1">
        <span className="text-sm font-medium text-gray-900">
          Step {currentIndex + 1} of {steps.length}
        </span>
        <span className="text-sm text-gray-500">{steps[currentIndex]?.label}</span>
      </div>
      <div className="h-1 w-full rounded bg-gray-100 overflow-hidden">
        <div
          className="h-full bg-blue-600 transition-all"
          style={{ width: `${((currentIndex + 1) / steps.length) * 100}%` }}
        />
      </div>
      <div className="flex items-center gap-2 mt-2">
        {steps.map((step, index) => {
          const clickable = step.status !== 'pending';
          return (
            <button
              key={step.id}
              type="button"
              disabled={!clickable}
              onClick={() => clickable && onStepClick(index)}
              className={`text-xs px-2 py-1 rounded ${
                index === currentIndex
                  ? 'bg-blue-50 text-blue-700 font-medium'
                  : clickable
                  ? 'text-gray-600 hover:bg-gray-50'
                  : 'text-gray-300 cursor-not-allowed'
              }`}
            >
              {step.status === 'done' ? '\u2713 ' : ''}
              {step.label}
            </button>
          );
        })}
      </div>
    </div>
  );
}

components/AmenityFormDrawer.tsx

The single component to import for both create and edit. Resolves which wizard kind applies from the selected amenity type, drives step state and validation, resets/prefills on open, and builds the final save payload (including generated timestamps for scheduled types).

Assumes Drawer accepts plain children when no tabs prop is passed. If your @hlt/react-component-library version strictly requires tabs, this needs a small adjustment to a single-tab wrapper instead. 
'use client';

import { useEffect, useState } from 'react';
import { Drawer, ChevronRightIcon } from '@hlt/react-component-library';
import { Amenity, AmenityType } from '@/types/amenities';
import { StepperHeader, type StepConfig } from './StepperHeader';
import { GeneralAdmissionForm } from './forms/GeneralAdmissionForm';
import { ScheduledAdmissionForm } from './forms/ScheduledAdmissionForm';
import { generateScheduleTimestamps } from '../lib/schedule-timestamp-generator';
import {
  EMPTY_GENERAL_INFO,
  EMPTY_SCHEDULE_RULE,
  GENERAL_ADMISSION_KEY,
  SCHEDULED_ADMISSION_KEY,
  WIZARD_STEP_LABELS,
  resolveWizardKind,
  type GeneralInfoValues,
  type ScheduleRuleValues,
  type AmenitySavePayload,
} from '../types/wizard-types';

interface Props {
  isOpen: boolean;
  onClose: () => void;
  onSave: (payload: AmenitySavePayload) => void;
  isSaving: boolean;
  amenityTypes: AmenityType[];
  amenity?: Amenity | null;
}

export function AmenityFormDrawer({
  isOpen,
  onClose,
  onSave,
  isSaving,
  amenityTypes,
  amenity = null,
}: Props) {
  const isEditing = amenity !== null;

  const [generalInfo, setGeneralInfo] = useState<GeneralInfoValues>(EMPTY_GENERAL_INFO);
  const [scheduleRule, setScheduleRule] = useState<ScheduleRuleValues>(EMPTY_SCHEDULE_RULE);
  const [step, setStep] = useState(0);
  const [error, setError] = useState<string | null>(null);

  const wizardKind = resolveWizardKind(generalInfo.amenityTypeId, amenityTypes);
  const stepLabels = WIZARD_STEP_LABELS[wizardKind];

  // Reset / prefill whenever the drawer opens or the target amenity changes.
  useEffect(() => {
    if (!isOpen) return;
    setStep(0);
    setError(null);

    if (amenity) {
      setGeneralInfo({
        amenityTypeId: amenity.amenityTypeId,
        amenityName: amenity.amenityName,
        amenityDescription: amenity.amenityDescription ?? '',
      });
      // TODO: once BE persists the original recurrence rule (not just the
      // expanded timestamps), map it into ScheduleRuleValues here. Until
      // then, editing a schedule starts from a blank rule.
      setScheduleRule(EMPTY_SCHEDULE_RULE);
    } else {
      setGeneralInfo(EMPTY_GENERAL_INFO);
      setScheduleRule(EMPTY_SCHEDULE_RULE);
    }
  }, [isOpen, amenity]);

  // If the type changes mid-flow (upgrade/downgrade) and we're now past the
  // step count for the new kind, snap back to a valid step.
  useEffect(() => {
    if (step >= stepLabels.length) setStep(stepLabels.length - 1);
  }, [stepLabels.length, step]);

  const steps: StepConfig[] = stepLabels.map((label, index) => ({
    id: label,
    label,
    status: index < step ? 'done' : index === step ? 'active' : 'pending',
  }));

  function handleClose() {
    setError(null);
    onClose();
  }

  function validateGeneralInfo(): boolean {
    if (!generalInfo.amenityTypeId) {
      setError('Please select an amenity type.');
      return false;
    }
    if (!generalInfo.amenityName.trim()) {
      setError('Name is required.');
      return false;
    }
    setError(null);
    return true;
  }

  function validateSchedule(): boolean {
    if (!scheduleRule.startDate) {
      setError('Start date is required.');
      return false;
    }
    if (scheduleRule.frequency === 'WEEKLY' && scheduleRule.daysOfWeek.length === 0) {
      setError('Select at least one day.');
      return false;
    }
    setError(null);
    return true;
  }

  function buildPayload(): AmenitySavePayload {
    const base = {
      amenityTypeId: Number(generalInfo.amenityTypeId),
      amenityName: generalInfo.amenityName.trim(),
      amenityDescription: generalInfo.amenityDescription.trim(),
    };

    if (wizardKind === SCHEDULED_ADMISSION_KEY) {
      return {
        ...base,
        schedule: {
          rule: scheduleRule,
          timestamps: generateScheduleTimestamps(scheduleRule),
        },
      };
    }
    return base;
  }

  function handlePrimaryAction() {
    const isLastStep = step === stepLabels.length - 1;

    if (!isLastStep) {
      if (!validateGeneralInfo()) return;
      setStep(step + 1);
      return;
    }

    if (wizardKind === SCHEDULED_ADMISSION_KEY) {
      if (!validateSchedule()) return;
    } else if (!validateGeneralInfo()) {
      return;
    }

    onSave(buildPayload());
  }

  const isLastStep = step === stepLabels.length - 1;

  return (
    <Drawer
      title={isEditing ? 'Edit amenity' : 'Add amenity'}
      width="sm"
      isOpen={isOpen}
      onClose={handleClose}
      footerActions={{
        secondary:
          step > 0
            ? { variant: 'outline', label: 'Back', onClick: () => setStep(step - 1) }
            : { variant: 'outline', label: 'Cancel', onClick: handleClose },
        primary: {
          variant: 'primary',
          label: isSaving ? 'Saving...' : isLastStep ? 'Save' : 'Next',
          onClick: handlePrimaryAction,
          icon: <ChevronRightIcon className="w-6 h-6" />,
        },
      }}
    >
      <StepperHeader steps={steps} currentIndex={step} onStepClick={setStep} />

      {wizardKind === GENERAL_ADMISSION_KEY ? (
        <GeneralAdmissionForm
          values={generalInfo}
          onChange={setGeneralInfo}
          amenityTypes={amenityTypes}
          error={error}
        />
      ) : (
        <ScheduledAdmissionForm
          step={step}
          generalInfo={generalInfo}
          onGeneralInfoChange={setGeneralInfo}
          scheduleRule={scheduleRule}
          onScheduleRuleChange={setScheduleRule}
          amenityTypes={amenityTypes}
          error={error}
        />
      )}
    </Drawer>
  );
}
