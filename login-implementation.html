Token storage flow
User types email + password ↓ LoginForm (CLIENT) react-hook-form validates immediately startTransition → loginAction(values) ↓ Server Action — no fetch(), no API route loginAction (SERVER) server-side validation runs again GraphQLClient → BE GraphQL endpoint ↓ BE returns AuthToken { token, idToken, refreshToken } ↓ Validate — all 3 tokens present? ↓ setAuthCookies(data.login) decodes idToken.sub → userId access_token → httpOnly cookie (24h) id_token → httpOnly cookie (24h) refresh_token → httpOnly cookie (30d) user_id → httpOnly cookie (24h) ↓ redirect('/dashboard') — tokens never visible in browser — network tab shows zero token data
Why 4 cookies not 3
Your refreshAuth mutation requires userId as an argument (confirmed from your schema screenshot).

userId = idToken.sub (Cognito sub claim). We decode it at login time and store it separately so refresh can use it without an extra BE call.

};

};

import 'server-only';

// ─────────────────────────────────────────────────────────────
// COOKIES — SERVER ONLY
//
// 'server-only' import causes a BUILD ERROR if this file is
// accidentally imported in a Client Component. Hard protection.
//
// Stores 4 httpOnly cookies after login:
//   access_token   → token from BE (accessToken JWT)
//   id_token       → idToken from BE
//   refresh_token  → refreshToken from BE
//   user_id        → decoded from idToken.sub
//                    (needed for refreshAuth mutation)
// ─────────────────────────────────────────────────────────────

import { cookies } from 'next/headers';
import {
  COOKIE_KEYS,
  COOKIE_OPTIONS,
  REFRESH_COOKIE_OPTIONS,
} from '@/lib/constants';
import type { AuthToken, DecodedIdToken } from '@/types/auth.types';

// ── Decode JWT payload (no signature verification — BE's job) ─
function decodeJwtPayload<T>(token: string): T | null {
  try {
    // JWT is base64url encoded: header.payload.signature
    const base64 = token.split('.')[1];
    // Replace base64url chars to standard base64
    const standard = base64.replace(/-/g, '+').replace(/_/g, '/');
    const json = Buffer.from(standard, 'base64').toString('utf-8');
    return JSON.parse(json) as T;
  } catch {
    return null;
  }
}

// ── WRITE — called after successful login or token refresh ────
export async function setAuthCookies(tokens: AuthToken): Promise<void> {
  const cookieStore = await cookies();

  // Decode userId from idToken — needed for refreshAuth mutation
  const payload = decodeJwtPayload<DecodedIdToken>(tokens.idToken);
  const userId = payload?.sub ?? '';

  // Store all 4 cookies atomically
  cookieStore.set(COOKIE_KEYS.ACCESS_TOKEN,  tokens.token,         COOKIE_OPTIONS);
  cookieStore.set(COOKIE_KEYS.ID_TOKEN,      tokens.idToken,       COOKIE_OPTIONS);
  cookieStore.set(COOKIE_KEYS.REFRESH_TOKEN, tokens.refreshToken,  REFRESH_COOKIE_OPTIONS);
  cookieStore.set(COOKIE_KEYS.USER_ID,       userId,               COOKIE_OPTIONS);
}

// ── READ — individual getters ─────────────────────────────────
export async function getAccessToken(): Promise<string | undefined> {
  const cookieStore = await cookies();
  return cookieStore.get(COOKIE_KEYS.ACCESS_TOKEN)?.value;
}

export async function getRefreshToken(): Promise<string | undefined> {
  const cookieStore = await cookies();
  return cookieStore.get(COOKIE_KEYS.REFRESH_TOKEN)?.value;
}

export async function getIdToken(): Promise<string | undefined> {
  const cookieStore = await cookies();
  return cookieStore.get(COOKIE_KEYS.ID_TOKEN)?.value;
}

export async function getUserId(): Promise<string | undefined> {
  const cookieStore = await cookies();
  return cookieStore.get(COOKIE_KEYS.USER_ID)?.value;
}

// ── CLEAR — called on logout ──────────────────────────────────
export async function clearAuthCookies(): Promise<void> {
  const cookieStore = await cookies();
  Object.values(COOKIE_KEYS).forEach((key) => cookieStore.delete(key));
}

loginAction.ts
src/modules/auth/actions/loginAction.ts
SERVER ACTION
Complete login flow. Server-side validation runs even if client validation was bypassed. Parses BE error messages into user-friendly text. redirect() is outside try/catch — critical.
'use server';

// ─────────────────────────────────────────────────────────────
// LOGIN ACTION — Server Action
//
// WHY Server Action instead of API Route:
//   - Runs directly on the server — zero HTTP round trip
//   - Has direct access to cookies() — no bridge needed
//   - Tokens NEVER appear in the browser network tab
//   - Form submits → this runs → cookies set → redirect
//
// Flow:
//   LoginForm calls this → GraphQLClient → BE returns 3 tokens
//   → setAuthCookies stores all 3 + userId as httpOnly cookies
//   → redirect('/dashboard')
// ─────────────────────────────────────────────────────────────

import { GraphQLClient, ClientError } from 'graphql-request';
import { redirect } from 'next/navigation';
import { GQL_ENDPOINT, ROUTES } from '@/lib/constants';
import { setAuthCookies } from '@/modules/auth/lib/cookies';
import { LOGIN_MUTATION } from '@/modules/auth/graphql/auth.mutations';
import type { LoginMutationResponse, ActionResult } from '@/types/auth.types';

export async function loginAction(
  input: { email: string; password: string }
): Promise<ActionResult> {
  const { email, password } = input;

  // ── 1. Server-side validation ──────────────────────────────
  // Never trust client-only validation
  if (!email?.trim()) {
    return { success: false, error: 'Email is required.' };
  }

  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  if (!emailRegex.test(email.trim())) {
    return { success: false, error: 'Enter a valid email address.' };
  }

  if (!password) {
    return { success: false, error: 'Password is required.' };
  }

  if (password.length < 6) {
    return { success: false, error: 'Password must be at least 6 characters.' };
  }

  // ── 2. Call BE GraphQL endpoint ────────────────────────────
  const client = new GraphQLClient(GQL_ENDPOINT, {
    // No Authorization header — login is a public mutation
    headers: {
      'Content-Type': 'application/json',
    },
  });

  try {
    const data = await client.request<LoginMutationResponse>(
      LOGIN_MUTATION,
      {
        email:    email.trim().toLowerCase(),
        password: password,
      }
    );

    // ── 3. Validate BE response ──────────────────────────────
    if (!data?.login?.token) {
      return {
        success: false,
        error: 'Login failed: no token received from server.',
      };
    }

    if (!data.login.idToken) {
      return {
        success: false,
        error: 'Login failed: no idToken received from server.',
      };
    }

    if (!data.login.refreshToken) {
      return {
        success: false,
        error: 'Login failed: no refreshToken received from server.',
      };
    }

    // ── 4. Store all 3 tokens as httpOnly cookies ───────────
    // setAuthCookies also decodes idToken.sub → stores userId
    // userId is needed later for the refreshAuth mutation
    await setAuthCookies(data.login);

  } catch (error) {
    // ── 5. Parse BE error for user-friendly message ──────────
    if (error instanceof ClientError) {
      const beMessage =
        error.response?.errors?.[0]?.message ?? '';

      // Map common BE error messages to user-friendly text
      if (
        beMessage.toLowerCase().includes('invalid') ||
        beMessage.toLowerCase().includes('incorrect') ||
        beMessage.toLowerCase().includes('wrong password') ||
        beMessage.toLowerCase().includes('bad credentials')
      ) {
        return {
          success: false,
          error: 'Invalid email or password. Please try again.',
        };
      }

      if (
        beMessage.toLowerCase().includes('not found') ||
        beMessage.toLowerCase().includes('no user') ||
        beMessage.toLowerCase().includes('does not exist')
      ) {
        return {
          success: false,
          error: 'No account found with this email.',
        };
      }

      if (
        beMessage.toLowerCase().includes('disabled') ||
        beMessage.toLowerCase().includes('blocked') ||
        beMessage.toLowerCase().includes('suspended')
      ) {
        return {
          success: false,
          error: 'Your account has been disabled. Contact support.',
        };
      }

      // Return actual BE message if we don't recognise it
      return {
        success: false,
        error: beMessage || 'Login failed. Please try again.',
      };
    }

    // Network error or unexpected failure
    console.error('[loginAction] Unexpected error:', error);
    return {
      success: false,
      error: 'Unable to connect. Please check your connection and try again.',
    };
  }

  // ── 6. Redirect MUST be outside try/catch ─────────────────
  // redirect() throws a special Next.js error internally.
  // If it's inside catch, the catch block will swallow it.
  redirect(ROUTES.DASHBOARD);
}

CLIENT
Uses react-hook-form + zod (already in your package.json). useTransition gives isPending loading state. Clears password on error — security best practice.
'use client';

// ─────────────────────────────────────────────────────────────
// LOGIN FORM — Client Component
//
// Responsibilities (UI only):
//   ✓ Controlled form with react-hook-form
//   ✓ Client-side validation with zod (instant feedback)
//   ✓ Loading state via useTransition
//   ✓ Server error display
//   ✓ Calls loginAction (Server Action) — no fetch needed
//
// Does NOT:
//   ✗ Store tokens
//   ✗ Know about cookies
//   ✗ Handle redirects
//   All of that is loginAction's responsibility
// ─────────────────────────────────────────────────────────────

import { useState, useTransition } from 'react';
import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { z } from 'zod';
import { loginAction } from '@/modules/auth/actions/loginAction';

// ── Validation schema ────────────────────────────────────────
const loginSchema = z.object({
  email: z
    .string()
    .min(1, 'Email is required')
    .email('Enter a valid email address'),
  password: z
    .string()
    .min(1, 'Password is required')
    .min(6, 'Password must be at least 6 characters'),
});

type LoginFormValues = z.infer<typeof loginSchema>;

// ── Component ────────────────────────────────────────────────
export function LoginForm() {
  // Server-returned errors (wrong credentials etc)
  const [serverError, setServerError] = useState<string | null>(null);

  // useTransition gives isPending loading state for free
  // No need for a separate loading useState
  const [isPending, startTransition] = useTransition();

  const {
    register,
    handleSubmit,
    formState: { errors },
    reset,
  } = useForm<LoginFormValues>({
    resolver: zodResolver(loginSchema),
    defaultValues: { email: '', password: '' },
  });

  function onSubmit(values: LoginFormValues) {
    setServerError(null); // clear previous server error

    startTransition(async () => {
      const result = await loginAction(values);

      // If loginAction redirected, this code never runs.
      // It only runs when loginAction returned an error object.
      if (!result.success) {
        setServerError(result.error);
        // Clear password field on error — security best practice
        reset({ email: values.email, password: '' });
      }
    });
  }

  return (
    <div className="min-h-screen flex items-center justify-center bg-gray-950">
      <div className="w-full max-w-md px-6">

        {/* Card */}
        <div className="bg-gray-900 border border-gray-800 rounded-2xl p-8 shadow-2xl">

          {/* Header */}
          <div className="text-center mb-8">
            <div className="inline-flex items-center justify-center w-12 h-12 bg-blue-500/10 border border-blue-500/20 rounded-xl mb-4">
              <svg width="24" height="24" viewBox="0 0 24 24" fill="none">
                <path d="M12 2L2 7l10 5 10-5-10-5zM2 17l10 5 10-5M2 12l10 5 10-5"
                  stroke="#38BDF8" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"/>
              </svg>
            </div>
            <h1 className="text-xl font-semibold text-white tracking-tight">
              Admin Portal
            </h1>
            <p className="text-sm text-gray-400 mt-1">
              Sign in to your account
            </p>
          </div>

          {/* Server Error Banner */}
          {serverError && (
            <div
              role="alert"
              className="flex items-start gap-3 bg-red-500/10 border border-red-500/20 rounded-lg px-4 py-3 mb-6"
            >
              <svg width="16" height="16" viewBox="0 0 16 16" fill="none" className="mt-0.5 shrink-0">
                <circle cx="8" cy="8" r="7" stroke="#F87171" strokeWidth="1.5"/>
                <path d="M8 5v3.5M8 11v.5" stroke="#F87171" strokeWidth="1.5" strokeLinecap="round"/>
              </svg>
              <span className="text-sm text-red-300">{serverError}</span>
            </div>
          )}

          {/* Form */}
          <div className="space-y-5">

            {/* Email */}
            <div>
              <label
                htmlFor="email"
                className="block text-sm font-medium text-gray-300 mb-1.5"
              >
                Email address
              </label>
              <input
                id="email"
                type="email"
                autoComplete="email"
                placeholder="you@company.com"
                disabled={isPending}
                className={`
                  w-full h-10 px-3 rounded-lg text-sm
                  bg-gray-800 text-white placeholder-gray-500
                  border transition-colors outline-none
                  disabled:opacity-50 disabled:cursor-not-allowed
                  focus:border-blue-400 focus:ring-2 focus:ring-blue-400/20
                  ${errors.email
                    ? 'border-red-500'
                    : 'border-gray-700'
                  }
                `}
                {...register('email')}
              />
              {errors.email && (
                <p className="text-xs text-red-400 mt-1">
                  {errors.email.message}
                </p>
              )}
            </div>

            {/* Password */}
            <div>
              <label
                htmlFor="password"
                className="block text-sm font-medium text-gray-300 mb-1.5"
              >
                Password
              </label>
              <input
                id="password"
                type="password"
                autoComplete="current-password"
                placeholder="••••••••"
                disabled={isPending}
                className={`
                  w-full h-10 px-3 rounded-lg text-sm
                  bg-gray-800 text-white placeholder-gray-500
                  border transition-colors outline-none
                  disabled:opacity-50 disabled:cursor-not-allowed
                  focus:border-blue-400 focus:ring-2 focus:ring-blue-400/20
                  ${errors.password
                    ? 'border-red-500'
                    : 'border-gray-700'
                  }
                `}
                {...register('password')}
              />
              {errors.password && (
                <p className="text-xs text-red-400 mt-1">
                  {errors.password.message}
                </p>
              )}
            </div>

            {/* Submit */}
            <button
              type="button"
              onClick={handleSubmit(onSubmit)}
              disabled={isPending}
              className="
                w-full h-10 rounded-lg text-sm font-medium
                bg-blue-500 hover:bg-blue-400
                text-white
                transition-colors
                disabled:opacity-60 disabled:cursor-not-allowed
                focus:outline-none focus:ring-2 focus:ring-blue-400/50
                mt-2
              "
            >
              {isPending ? (
                <span className="flex items-center justify-center gap-2">
                  <svg
                    className="animate-spin w-4 h-4"
                    viewBox="0 0 24 24"
                    fill="none"
                  >
                    <circle
                      className="opacity-25"
                      cx="12" cy="12" r="10"
                      stroke="currentColor" strokeWidth="4"
                    />
                    <path
                      className="opacity-75"
                      fill="currentColor"
                      d="M4 12a8 8 0 018-8V0C5.4 0 0 5.4 0 12h4z"
                    />
                  </svg>
                  Signing in...
                </span>
              ) : (
                'Sign in'
              )}
            </button>

          </div>
        </div>
      </div>
    </div>
  );
}

// ─────────────────────────────────────────────────────────────
// LOGIN PAGE — Server Component (no 'use client')
//
// Runs on the server. Checks if user already has a valid token.
// If yes → redirect to dashboard (no flicker, no client code).
// If no  → render LoginForm client component.
// ─────────────────────────────────────────────────────────────

import { redirect } from 'next/navigation';
import { getAccessToken } from '@/modules/auth/lib/cookies';
import { LoginForm } from '@/modules/auth/components/LoginForm';
import { ROUTES } from '@/lib/constants';
import type { Metadata } from 'next';

export const metadata: Metadata = {
  title: 'Sign In | Admin Portal',
  description: 'Sign in to your admin account',
};

export default async function LoginPage() {
  // Already logged in? Skip login page entirely.
  // Server-side redirect — no client-side flicker.
  const token = await getAccessToken();
  if (token) {
    redirect(ROUTES.DASHBOARD);
  }

  return (
    <LoginForm />
  );
}
