SERVER src/modules/auth/lib/refresh.ts
FIXED: Added authorization header on refresh call. Your current code was missing it — causing BE to reject the refresh mutation.
import 'server-only';

// ─────────────────────────────────────────────────────────────
// WHAT CHANGED vs your current code (Image 3):
//
//   1. Added 'authorization' header on refresh call
//      Your current code sends no auth header → BE rejects it
//
//   2. Using getRefreshToken() as the token value for the mutation
//      (confirmed from your Postman screenshots earlier)
//
//   3. setAuthCookies instead of setAccessTokenCookies
//      (updates ALL 4 cookies, not just access token)
// ─────────────────────────────────────────────────────────────

import { GraphQLClient, ClientError } from 'graphql-request';
import { GQL_ENDPOINT }               from '@/lib/constants';
import {
  getRefreshToken,
  getUserId,
  setAuthCookies,
  clearAuthCookies,
} from '@/modules/auth/lib/cookies';
import { REFRESH_AUTH_MUTATION }        from '../graphql/auth.mutations';
import type { RefreshAuthMutationResponse } from '@/types/auth';

export async function refreshAccessToken(): Promise<string | null> {
  const [refreshToken, userId] = await Promise.all([
    getRefreshToken(),
    getUserId(),
  ]);

  if (!refreshToken || !userId) {
    console.warn('[refresh] Missing refreshToken or userId — cannot refresh');
    return null;
  }

  try {
    const client = new GraphQLClient(GQL_ENDPOINT, {
      headers: {
        // ✅ FIXED: must send the refreshToken as authorization header
        // Your current code was missing this — causing BE to reject refresh
        'authorization': refreshToken,
        'Content-Type':  'application/json',
        'x-tenant-id':   process.env.TENANT_ID ?? '',
      },
    });

    const data = await client.request<RefreshAuthMutationResponse>(
      REFRESH_AUTH_MUTATION,
      { userId, token: refreshToken }
    );

    if (!data?.refreshAuth?.token) {
      console.warn('[refresh] BE returned no token');
      return null;
    }

    // ✅ FIXED: update ALL 4 cookies (access, id, refresh, userId)
    await setAuthCookies(data.refreshAuth);
    return data.refreshAuth.token;

  } catch (error) {
    if (error instanceof ClientError) {
      const msg = (error.response?.errors?.[0]?.message ?? '').toLowerCase();
      console.error('[refresh] BE error:', msg);
      if (
        msg.includes('expired')       ||
        msg.includes('invalid')       ||
        msg.includes('unauthenticated')
      ) {
        await clearAuthCookies(); // force re-login
      }
    } else {
      console.error('[refresh] Unexpected error:', error);
    }
    return null;
  }
}
SERVER src/app/api/graphql/route.ts
NEW PROXY: All client GQL calls go through here. Fixes timeout. Handles token refresh automatically. Returns SESSION_EXPIRED if refresh fails.
// src/app/api/graphql/route.ts
// ─────────────────────────────────────────────────────────────
// WHY THIS EXISTS:
//   Browser cannot reach the GQL endpoint directly (network restriction)
//   → ERR_CONNECTION_TIMED_OUT
//
//   This proxy receives requests from the browser, adds auth headers
//   server-side (from httpOnly cookies), and forwards to GQL endpoint.
//
// REFRESH FLOW:
//   No token     → refresh → retry
//   Request 401  → refresh → retry once
//   Refresh fail → 401 SESSION_EXPIRED → client redirects to login
// ─────────────────────────────────────────────────────────────

import { NextRequest, NextResponse } from 'next/server';
import { getAccessToken }            from '@/modules/auth/lib/cookies';
import { refreshAccessToken }        from '@/modules/auth/lib/refresh';

const GQL_ENDPOINT = process.env.NEXT_PUBLIC_GRAPHQL_URL!;

async function forwardToGql(token: string, body: string) {
  return fetch(GQL_ENDPOINT, {
    method:  'POST',
    headers: {
      'Content-Type':  'application/json',
      'authorization': token,
      'x-tenant-id':   process.env.TENANT_ID ?? '',
    },
    body,
  });
}

export async function POST(req: NextRequest) {
  const body = await req.text();

  let token          = await getAccessToken();
  let alreadyRefreshed = false;

  // No access token — try refresh before first request
  if (!token) {
    token = await refreshAccessToken();
    alreadyRefreshed = true;

    if (!token) {
      return NextResponse.json(
        { errors: [{ message: 'SESSION_EXPIRED' }] },
        { status: 401 }
      );
    }
  }

  // Forward request to GQL endpoint
  let res = await forwardToGql(token, body);

  // Token rejected by BE — try refresh once and retry
  if (res.status === 401 && !alreadyRefreshed) {
    const newToken = await refreshAccessToken();

    if (!newToken) {
      return NextResponse.json(
        { errors: [{ message: 'SESSION_EXPIRED' }] },
        { status: 401 }
      );
    }

    res = await forwardToGql(newToken, body);
  }

  // Return whatever GQL sent back (success or error)
  const data = await res.json();
  return NextResponse.json(data, { status: res.ok ? 200 : res.status });
}

export async function POST() {
  const newToken = await refreshAccessToken();

  if (!newToken) {

SERVER src/app/api/auth/refresh-and-redirect/route.ts
Server-side refresh. Called when serverGqlFetch detects expired access token on page reload.
// src/app/api/auth/refresh-and-redirect/route.ts
// Used by serverGqlFetch when access token is missing/expired.
// Route handlers CAN write cookies — Server Components cannot.

import { NextRequest, NextResponse } from 'next/server';
import { refreshAccessToken }        from '@/modules/auth/lib/refresh';

export async function GET(req: NextRequest) {
  const callbackUrl = req.nextUrl.searchParams.get('callbackUrl') ?? '/dashboard';

  // Safety: prevent redirect loops
  if (callbackUrl.includes('refresh-and-redirect')) {
    return NextResponse.redirect(new URL('/login?reason=expired', req.url));
  }

  const newToken = await refreshAccessToken();

  if (!newToken) {
    // Refresh failed → force re-login
    return NextResponse.redirect(
      new URL('/login?reason=expired', req.url)
    );
  }

  // Cookies already updated by refreshAccessToken()
  // Redirect back to original page
  return NextResponse.redirect(new URL(callbackUrl, req.url));
}

import {
  useQuery,
  useMutation,
  useQueryClient,
  type UseQueryOptions,
} from '@tanstack/react-query';
import { useRouter } from 'next/navigation';

// ── Core fetcher via proxy ────────────────────────────────────
// Calls /api/graphql which runs server-side and:
//   1. Reads auth token from httpOnly cookie
//   2. Adds headers (authorization, x-tenant-id)
//   3. Forwards to actual GQL endpoint
//   4. Handles token refresh if 401
async function gqlFetch<TData>(
  query:      string,
  variables?: object,
  onExpired?: () => void,
): Promise<TData> {

  const res = await fetch('/api/graphql', {
    method:      'POST',
    headers:     { 'Content-Type': 'application/json' },
    credentials: 'include',  // send cookies so proxy can read them
    body:        JSON.stringify({ query, variables }),
    cache:       'no-store',
  });

  const json = await res.json() as {
    data?:   TData;
    errors?: { message: string }[];
  };

  // Session expired — proxy tried refresh but it also failed
  if (
    res.status === 401 ||
    json.errors?.[0]?.message === 'SESSION_EXPIRED'
  ) {
    onExpired?.();
    throw new Error('SESSION_EXPIRED');
  }

  // GraphQL-level errors (not auth)
  if (json.errors?.length) {
    throw new Error(json.errors[0].message);
  }

  if (json.data === undefined || json.data === null) {
    throw new Error('No data returned from server');
  }

  return json.data;
}

// ── useAuthQuery ──────────────────────────────────────────────
// Unchanged signature — all existing usage works as-is
export function useAuthQuery<
  TData,
  TSelected = TData,
>(
  queryKey:   readonly unknown[],
  query:      string,
  variables?: object,
  options?:   Omit<
    UseQueryOptions<TData, Error, TSelected>,
    'queryKey' | 'queryFn'
  >,
) {
  const router = useRouter();

  return useQuery<TData, Error, TSelected>({
    queryKey,
    queryFn: () =>
      gqlFetch<TData>(
        query,
        variables,
        () => router.push('/login?reason=expired') // ← redirect on session expire
      ),
    staleTime: 1000 * 60 * 5,
    retry:     1,
    ...options,
  });
}

// ── useAuthMutation ───────────────────────────────────────────
// Unchanged signature — same options shape as your current code
export function useAuthMutation<
  TData,
  TVars extends object = object,
>(
  mutation: string,
  options?: {
    onSuccess?:      (data: TData) => void;
    onError?:        (error: Error) => void;
    invalidateKeys?: unknown[][];
  },
) {
  const router      = useRouter();
  const queryClient = useQueryClient();

  return useMutation<TData, Error, TVars>({
    mutationFn: (variables: TVars) =>
      gqlFetch<TData>(
        mutation,
        variables,
        () => router.push('/login?reason=expired') // ← redirect on session expire
      ),

    onSuccess: (data) => {
      // Invalidate provided query keys
      options?.invalidateKeys?.forEach(key =>
        queryClient.invalidateQueries({ queryKey: key })
      );
      options?.onSuccess?.(data);
    },

    onError: options?.onError,
  });
}
