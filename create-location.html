CLIENT src/modules/locations/hooks/useCreateLocation.ts
Mutation hook. Handles success/error toasts + invalidates tree cache.
'use client';

// ─────────────────────────────────────────────────────────────
// useCreateLocation
// Handles the full create flow:
//   mutation → loading → success toast → invalidate tree cache
// ─────────────────────────────────────────────────────────────

import { useQueryClient }  from '@tanstack/react-query';
import { useAuthMutation } from '@/modules/auth/hooks/useAuthQuery';
import { locationKeys }    from '@/modules/locations/hooks/useLocations';
import { useToasts }       from '@/store/ui.store';
import {
  CREATE_LOCATION_MUTATION,
  type CreateLocationInput,
  type CreateLocationResponse,
} from '../graphql/create-location.mutation';

export function useCreateLocation(onSuccess?: () => void) {
  const queryClient       = useQueryClient();
  const { add: addToast } = useToasts();

  return useAuthMutation<CreateLocationResponse, CreateLocationInput>(
    CREATE_LOCATION_MUTATION,
    {
      onSuccess: (data) => {
        if (!data.createLocation.success) {
          addToast({
            type:    'error',
            message: data.createLocation.message ?? 'Failed to create location.',
          });
          return;
        }

        // Refresh the sidebar tree
        queryClient.invalidateQueries({
          queryKey: locationKeys.tree(),
        });

        addToast({
          type:    'success',
          message: 'Location created successfully.',
        });

        onSuccess?.();
      },
      onError: (error) => {
        addToast({
          type:    'error',
          message: error.message ?? 'Something went wrong.',
        });
      },
    }
  );
}
CLIENT src/modules/locations/components/createLocation/CreateLocationModal.tsx
Form with parent dropdown + name field. Zod validation. Loading spinner.
'use client';

// ─────────────────────────────────────────────────────────────
// CREATE LOCATION MODAL
// - Parent location dropdown (from existing tree cache)
// - Location name text field
// - Zod validation
// - Loading indicator during submission
// ─────────────────────────────────────────────────────────────

import { useForm }          from 'react-hook-form';
import { zodResolver }      from '@hookform/resolvers/zod';
import { z }                from 'zod';
import { useLocationsTree } from '@/modules/locations/hooks/useLocations';
import { useCreateLocation } from '../hooks/useCreateLocation';
import { flattenTreeForDropdown } from '@/modules/locations/lib/utils';

// ── Validation schema ─────────────────────────────────────────
const schema = z.object({
  parentLocationId: z
    .string()
    .min(1, 'Please select a parent location'),
  locationName: z
    .string()
    .min(1, 'Location name is required')
    .min(2, 'Name must be at least 2 characters')
    .max(100, 'Name must be under 100 characters'),
});

type FormValues = z.infer<typeof schema>;

interface Props {
  onClose: () => void;
}

export function CreateLocationModal({ onClose }: Props) {
  // Get locations for dropdown from existing cache — no extra API call
  const { data: tree = [], isLoading: treeLoading } = useLocationsTree();

  // Flatten tree into flat list for dropdown (skip root client node)
  const locationOptions = flattenTreeForDropdown(tree);

  const createLocation = useCreateLocation(onClose);

  const {
    register,
    handleSubmit,
    formState: { errors },
  } = useForm<FormValues>({
    resolver: zodResolver(schema),
  });

  function onSubmit(values: FormValues) {
    createLocation.mutate({
      locationName:     values.locationName.trim(),
      parentLocationId: Number(values.parentLocationId),
    });
  }

  const isSubmitting = createLocation.isPending;

  return (
    // Backdrop
    <div
      className="fixed inset-0 z-50 flex items-center justify-center bg-black/40"
      onClick={(e) => e.target === e.currentTarget && onClose()}
    >
      <div className="w-full max-w-md rounded-2xl bg-white shadow-2xl">

        {/* Header */}
        <div className="flex items-center justify-between border-b border-gray-100 px-6 py-4">
          <h2 className="text-base font-semibold text-gray-900">
            Create Location
          </h2>
          <button
            onClick={onClose}
            disabled={isSubmitting}
            className="text-gray-400 hover:text-gray-600 transition-colors
              disabled:opacity-50"
          >
            <svg width="16" height="16" viewBox="0 0 24 24" fill="none">
              <path d="M18 6L6 18M6 6l12 12"
                stroke="currentColor" strokeWidth="2" strokeLinecap="round"/>
            </svg>
          </button>
        </div>

        {/* Form */}
        <div className="px-6 py-5 space-y-4">

          {/* Parent Location Dropdown */}
          <div>
            <label className="block text-xs font-medium text-gray-700 mb-1">
              Parent Location <span className="text-red-500">*</span>
            </label>
            <select
              disabled={treeLoading || isSubmitting}
              className={`w-full h-10 rounded-lg border px-3 text-sm outline-none
                transition-colors bg-white
                focus:border-blue-400 focus:ring-1 focus:ring-blue-400/30
                disabled:bg-gray-50 disabled:cursor-not-allowed
                ${errors.parentLocationId
                  ? 'border-red-300'
                  : 'border-gray-200'
                }`}
              {...register('parentLocationId')}
            >
              <option value="">
                {treeLoading ? 'Loading locations...' : 'Select parent location'}
              </option>
              {locationOptions.map((loc) => (
                <option key={loc.id} value={loc.id}>
                  {loc.indent}{loc.label}
                </option>
              ))}
            </select>
            {errors.parentLocationId && (
              <p className="mt-1 text-xs text-red-500">
                {errors.parentLocationId.message}
              </p>
            )}
          </div>

          {/* Location Name */}
          <div>
            <label className="block text-xs font-medium text-gray-700 mb-1">
              Location Name <span className="text-red-500">*</span>
            </label>
            <input
              type="text"
              placeholder="Enter location name"
              disabled={isSubmitting}
              className={`w-full h-10 rounded-lg border px-3 text-sm outline-none
                transition-colors
                focus:border-blue-400 focus:ring-1 focus:ring-blue-400/30
                disabled:bg-gray-50 disabled:cursor-not-allowed
                ${errors.locationName
                  ? 'border-red-300'
                  : 'border-gray-200'
                }`}
              {...register('locationName')}
            />
            {errors.locationName && (
              <p className="mt-1 text-xs text-red-500">
                {errors.locationName.message}
              </p>
            )}
          </div>

        </div>

        {/* Footer */}
        <div className="flex gap-3 border-t border-gray-100 px-6 py-4">
          <button
            onClick={onClose}
            disabled={isSubmitting}
            className="flex-1 h-10 rounded-lg border border-gray-200
              text-sm text-gray-600 hover:bg-gray-50
              transition-colors disabled:opacity-50"
          >
            Cancel
          </button>
          <button
            onClick={handleSubmit(onSubmit)}
            disabled={isSubmitting || treeLoading}
            className="flex-1 h-10 rounded-lg bg-blue-600 text-white
              text-sm font-medium hover:bg-blue-700
              transition-colors disabled:opacity-60
              flex items-center justify-center gap-2"
          >
            {isSubmitting ? (
              <>
                {/* Loading spinner */}
                <svg className="animate-spin w-4 h-4" viewBox="0 0 24 24" fill="none">
                  <circle className="opacity-25" cx="12" cy="12" r="10"
                    stroke="currentColor" strokeWidth="4"/>
                  <path className="opacity-75" fill="currentColor"
                    d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4z"/>
                </svg>
                Creating...
              </>
            ) : (
              'Create Location'
            )}
          </button>
        </div>

      </div>
    </div>
  );
}

// ─────────────────────────────────────────────────────────────
// ADD TO: src/modules/locations/lib/utils.ts
//
// Flattens TreeNode[] into a flat list for dropdown use.
// Adds visual indentation to show hierarchy.
// Skips root client nodes (parentId: null).
// ─────────────────────────────────────────────────────────────

import type { TreeNode } from '@/types/locations';

export interface DropdownOption {
  id:     string;
  label:  string;
  indent: string;  // e.g. "  " "    " for visual depth
}

export function flattenTreeForDropdown(
  nodes:  TreeNode[],
  depth:  number = 0,
  result: DropdownOption[] = []
): DropdownOption[] {
  for (const node of nodes) {
    // Skip root client node (depth 0, parentId null)
    // Include all location nodes as valid parent options
    if (node.parentId !== null) {
      result.push({
        id:     node.id,
        label:  node.label,
        indent: '  '.repeat(depth - 1), // visual indent for hierarchy
      });
    }

    // Always recurse into children
    if (node.children.length) {
      flattenTreeForDropdown(node.children, depth + 1, result);
    }
  }

  return result;
}
