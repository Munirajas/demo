Key insight — API vs page requests
API calls (fetch to /api/*) must get a JSON 401 response — NOT a redirect.
Redirecting an API call would return HTML to your fetch() code, causing a parse error.

useAuthQuery already handles 401 responses — catches it, calls /api/auth/refresh, retries.
So API requests during token expiry are handled client-side, not by middleware redirect.

import { NextRequest, NextResponse } from 'next/server';
import {
  COOKIE_KEYS,
  PUBLIC_ROUTES,
  BYPASS_PREFIXES,
  ROUTES,
} from '@/lib/constants';

// ─────────────────────────────────────────────────────────────
// MIDDLEWARE — Edge Runtime
//
// Handles EVERY incoming request type correctly:
//
//   Static assets     → always bypass (no auth needed)
//   Auth API routes   → always bypass (these SET the tokens)
//   Public pages      → bypass if logged out, redirect if logged in
//   Protected pages   → full auth flow below
//   Data API routes   → return 401 JSON (not redirect)
//   Image/file routes → return 401 JSON (not redirect)
//
// TOKEN STATES:
//   access_token ✓               → authenticated, let through
//   access_token ✗ + refresh ✓  → send to silent-refresh
//   access_token ✗ + refresh ✗  → send to login
//
// RACE CONDITION PREVENTION:
//   When access_token is missing, silent-refresh replaces
//   the entire page navigation — only ONE refresh fires.
//   API calls from client components don't reach middleware
//   as page navigations, so they are handled separately
//   by returning 401 JSON, which useAuthQuery catches
//   and retries after getting a new token.
// ─────────────────────────────────────────────────────────────

// Prefixes that are NEVER intercepted regardless of auth state
const ALWAYS_BYPASS = [
  ...BYPASS_PREFIXES,
  '/auth/silent-refresh',  // must bypass or infinite loop
  '/api/auth',             // these routes manage tokens themselves
  '/_next',
  '/favicon.ico',
  '/public',
];

// Prefixes that are API/data routes (return JSON 401, not redirect)
const API_PREFIXES = [
  '/api/',
];

// File extensions that should return 401, not redirect
const ASSET_EXTENSIONS = [
  '.json', '.xml', '.csv',
  '.png', '.jpg', '.jpeg', '.gif', '.svg', '.webp', '.ico',
  '.woff', '.woff2', '.ttf', '.eot',
  '.mp4', '.mp3', '.pdf',
];

function isApiRequest(pathname: string): boolean {
  return API_PREFIXES.some(p => pathname.startsWith(p));
}

function isAssetRequest(pathname: string): boolean {
  return ASSET_EXTENSIONS.some(ext => pathname.endsWith(ext));
}

export default function middleware(req: NextRequest) {
  const { pathname } = req.nextUrl;

  // ── Step 1: Always bypass these ────────────────────────────
  if (ALWAYS_BYPASS.some(p => pathname.startsWith(p))) {
    return NextResponse.next();
  }

  const accessToken  = req.cookies.get(COOKIE_KEYS.ACCESS_TOKEN)?.value;
  const refreshToken = req.cookies.get(COOKIE_KEYS.REFRESH_TOKEN)?.value;
  const isPublicRoute = PUBLIC_ROUTES.some(
    r => pathname === r || pathname.startsWith(r)
  );

  // ── Step 2: Has valid access token ─────────────────────────
  if (accessToken) {
    // Logged in user hitting public route → dashboard
    if (isPublicRoute) {
      return NextResponse.redirect(new URL(ROUTES.DASHBOARD, req.url));
    }
    // All other routes → let through normally
    return NextResponse.next();
  }

  // ── Step 3: No access token — handle each request type ─────

  // Public routes (login, register) — let through without token
  if (isPublicRoute) {
    return NextResponse.next();
  }

  // API requests — return 401 JSON, NOT a redirect
  // useAuthQuery catches this, triggers refresh, retries
  // Redirecting an API call would break the client
  if (isApiRequest(pathname)) {
    return NextResponse.json(
      {
        error: 'UNAUTHENTICATED',
        message: 'Access token expired. Please retry after refresh.',
      },
      { status: 401 }
    );
  }

  // Asset requests — return 401, not redirect
  // Images/files don't need to redirect to refresh flow
  if (isAssetRequest(pathname)) {
    return new NextResponse(null, { status: 401 });
  }

  // ── Step 4: Page navigation without access token ───────────

  // Has refresh token → can recover silently
  // Redirect to silent-refresh which will:
  //   1. Call refreshAccessToken() server-side
  //   2. Write new cookies
  //   3. Redirect back to this page (callbackUrl)
  // This is the only place refresh is triggered from middleware
  if (refreshToken) {
    const url = new URL('/auth/silent-refresh', req.url);
    url.searchParams.set('callbackUrl', pathname);
    return NextResponse.redirect(url);
  }

  // No refresh token either → truly logged out
  const loginUrl = new URL(ROUTES.LOGIN, req.url);
  loginUrl.searchParams.set('callbackUrl', pathname);
  return NextResponse.redirect(loginUrl);
}

export const config = {
  matcher: [
    '/((?!_next/static|_next/image|favicon.ico).*)',
  ],
};

// src/app/auth/silent-refresh/page.tsx
// SERVER COMPONENT — no 'use client'
//
// ─────────────────────────────────────────────────────────────
// SILENT REFRESH PAGE
//
// RACE CONDITION PREVENTION:
//   This page replaces the entire browser navigation.
//   While it runs, no other page components are mounted.
//   No useAuthQuery hooks are running.
//   No parallel API calls are being made.
//   Only ONE refresh request fires — this one.
//
// FLOW:
//   1. User's page navigation hits middleware
//   2. Middleware sees: no access_token + has refresh_token
//   3. Middleware redirects here with callbackUrl
//   4. This page runs server-side refreshAccessToken()
//   5a. Success → redirect to callbackUrl (original destination)
//   5b. Failure → redirect to /login?reason=expired
// ─────────────────────────────────────────────────────────────

import { redirect }           from 'next/navigation';
import { refreshAccessToken } from '@/modules/auth/lib/refresh';

interface Props {
  searchParams: Promise<{ callbackUrl?: string }>;
}

export default async function SilentRefreshPage({ searchParams }: Props) {
  const { callbackUrl } = await searchParams;

  // Validate callbackUrl — must be a relative path
  // Prevents open redirect attacks
  const destination =
    callbackUrl &&
    callbackUrl.startsWith('/') &&
    !callbackUrl.startsWith('//')
      ? callbackUrl
      : '/dashboard';

  // Attempt token refresh — server-side, secure
  const newToken = await refreshAccessToken();

  if (newToken) {
    redirect(destination);
  }

  // Refresh failed — both tokens expired
  redirect('/login?reason=expired');
}
