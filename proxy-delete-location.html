SERVER src/app/api/graphql/route.ts
PROXY — Fixes ERR_CONNECTION_TIMED_OUT. Browser calls this, server forwards to GQL endpoint.
// src/app/api/graphql/route.ts
//
// ─────────────────────────────────────────────────────────────
// GRAPHQL PROXY ROUTE
//
// WHY THIS EXISTS:
//   The GQL endpoint is only reachable from the server
//   (network/firewall restriction). Browsers can't reach it
//   directly → ERR_CONNECTION_TIMED_OUT.
//
// HOW IT WORKS:
//   Browser → POST /api/graphql (Next.js server)
//             → reads auth token from httpOnly cookie
//             → forwards to actual GQL endpoint
//             → returns response to browser
//
// BENEFITS:
//   ✓ Fixes timeout/CORS (request comes from server)
//   ✓ Token never exposed to JS (stays in httpOnly cookie)
//   ✓ One place to handle auth refresh
//   ✓ Works for all client-side mutations
// ─────────────────────────────────────────────────────────────

import { NextRequest, NextResponse } from 'next/server';
import {
  getAccessToken,
  getRefreshToken,
  getUserId,
  setAuthCookies,
} from '@/modules/auth/lib/cookies';
import { refreshAccessToken } from '@/modules/auth/lib/refresh';

const GQL_ENDPOINT = process.env.NEXT_PUBLIC_GRAPHQL_URL!;

async function executeGql(
  token:     string,
  body:      string
): Promise<Response> {
  return fetch(GQL_ENDPOINT, {
    method:  'POST',
    headers: {
      'Content-Type':  'application/json',
      'authorization': token,
      'x-tenant-id':   process.env.TENANT_ID ?? '',
    },
    body,
  });
}

export async function POST(req: NextRequest) {
  try {
    const body = await req.text(); // keep as string for forwarding

    let token = await getAccessToken();

    // No token — try refresh
    if (!token) {
      token = await refreshAccessToken();
      if (!token) {
        return NextResponse.json(
          { errors: [{ message: 'SESSION_EXPIRED' }] },
          { status: 401 }
        );
      }
    }

    let res = await executeGql(token, body);

    // Token expired during request — refresh and retry once
    if (res.status === 401) {
      const newToken = await refreshAccessToken();
      if (!newToken) {
        return NextResponse.json(
          { errors: [{ message: 'SESSION_EXPIRED' }] },
          { status: 401 }
        );
      }
      res = await executeGql(newToken, body);
    }

    const data = await res.json();
    return NextResponse.json(data);

  } catch (error) {
    console.error('[/api/graphql]', error);
    return NextResponse.json(
      { errors: [{ message: 'Proxy error' }] },
      { status: 500 }
    );
  }
}

// ─────────────────────────────────────────────────────────────
// useAuthQuery — UPDATED to use /api/graphql proxy
//
// KEY CHANGE: gqlFetch now calls /api/graphql (Next.js proxy)
// instead of the GQL endpoint directly.
//
// Browser → /api/graphql → actual GQL endpoint (server-side)
//
// This fixes ERR_CONNECTION_TIMED_OUT (network restriction)
// and also avoids CORS issues.
// ─────────────────────────────────────────────────────────────

import {
  useQuery,
  useMutation,
  useQueryClient,
  type UseQueryOptions,
} from '@tanstack/react-query';
import { useRouter } from 'next/navigation';

// ── Core proxy fetcher ────────────────────────────────────────
// Calls /api/graphql which proxies to actual GQL endpoint
// No token management here — proxy handles it server-side
async function gqlFetch<TData>(
  query:      string,
  variables?: object,
  onExpired?: () => void,
): Promise<TData> {
  const res = await fetch('/api/graphql', {
    method:      'POST',
    headers:     { 'Content-Type': 'application/json' },
    credentials: 'include',  // send cookies to proxy
    body:        JSON.stringify({ query, variables }),
    cache:       'no-store',
  });

  const json = await res.json() as {
    data?:   TData;
    errors?: { message: string }[];
  };

  // Session expired
  if (res.status === 401 ||
      json.errors?.[0]?.message === 'SESSION_EXPIRED') {
    onExpired?.();
    throw new Error('SESSION_EXPIRED');
  }

  // GraphQL errors
  if (json.errors?.length) {
    throw new Error(json.errors[0].message);
  }

  // No data returned
  if (json.data === undefined || json.data === null) {
    throw new Error('No data returned');
  }

  return json.data;
}

// ── useAuthQuery ──────────────────────────────────────────────
export function useAuthQuery<
  TData,
  TSelected = TData,
>(
  queryKey:   readonly unknown[],
  query:      string,
  variables?: object,
  options?:   Omit<
    UseQueryOptions<TData, Error, TSelected>,
    'queryKey' | 'queryFn'
  >,
) {
  const router = useRouter();
  return useQuery<TData, Error, TSelected>({
    queryKey,
    queryFn: () =>
      gqlFetch<TData>(
        query,
        variables,
        () => router.push('/login?reason=expired')
      ),
    staleTime: 1000 * 60 * 2,
    retry:     1,
    ...options,
  });
}

// ── useAuthMutation ───────────────────────────────────────────
export function useAuthMutation<
  TData,
  TVars extends object = object,
>(
  mutation: string,
  options?: {
    onSuccess?: (data: TData, vars: TVars) => void;
    onError?:   (error: Error) => void;
  },
) {
  const router = useRouter();

  return useMutation<TData, Error, TVars>({
    mutationFn: (vars: TVars) =>
      gqlFetch<TData>(
        mutation,
        vars,
        () => router.push('/login?reason=expired')
      ),
    onSuccess: (data, vars) => options?.onSuccess?.(data, vars),
    onError:   (error)      => options?.onError?.(error),
  });
}

export const REMOVE_LOCATION_MUTATION = `
  mutation RemoveLocation($type: String!, $locationId: Int!) {
    removeLocation(request: {
      type:       $type
      locationId: $locationId
    }) {
      success
      message
      data
    }
  }
`;

export interface RemoveLocationInput {
  type:       string;  // always "client"
  locationId: number;
}

export interface RemoveLocationResponse {
  removeLocation: {
    success: boolean;
    message: string;
    data:    unknown;
  };
}
CLIENT src/modules/locations/hooks/useRemoveLocation.ts
Delete hook. Invalidates tree cache + toast + navigates to dashboard.
'use client';

import { useQueryClient }  from '@tanstack/react-query';
import { useRouter }       from 'next/navigation';
import { toast }           from 'sonner';
import { useAuthMutation } from '@/modules/auth/hooks/useAuthQuery';
import { locationKeys }    from '@/modules/locations/hooks/useLocations';
import {
  REMOVE_LOCATION_MUTATION,
  type RemoveLocationInput,
  type RemoveLocationResponse,
} from '../graphql/remove-location.mutation';

export function useRemoveLocation(onSuccess?: () => void) {
  const queryClient = useQueryClient();
  const router      = useRouter();

  return useAuthMutation<RemoveLocationResponse, RemoveLocationInput>(
    REMOVE_LOCATION_MUTATION,
    {
      onSuccess: (data) => {
        const result = data.removeLocation;

        if (!result.success) {
          // BE returned success:false (e.g. location has children)
          toast.error(result.message || 'Failed to delete location.');
          return;
        }

        // Refresh sidebar tree
        queryClient.invalidateQueries({
          queryKey: locationKeys.tree(),
        });

        toast.success('Location deleted successfully.');
        onSuccess?.();

        // Navigate to dashboard since this location no longer exists
        router.push('/dashboard');
      },

      onError: (error) => {
        toast.error(error.message || 'Something went wrong.');
      },
    }
  );
}

'use client';

import { useRemoveLocation } from '@/modules/locations/hooks/useRemoveLocation';

interface Props {
  locationId:   string;
  locationName: string;
  onClose:      () => void;
}

export function DeleteLocationModal({
  locationId,
  locationName,
  onClose,
}: Props) {
  const removeLocation = useRemoveLocation(onClose);
  const isLoading      = removeLocation.isPending;

    removeLocation.mutate({
      type:       'client',
      locationId: Number(locationId),
    });
  }

      onClick={(e) => e.target === e.currentTarget && !isLoading && onClose()}
    >
      <div className="w-full max-w-sm rounded-2xl bg-white shadow-2xl p-6">

        {/* Icon */}
        <div className="w-10 h-10 rounded-full bg-red-100 flex items-center
          justify-center mx-auto mb-4">
          <svg width="18" height="18" viewBox="0 0 24 24" fill="none">
            <path d="M3 6h18M8 6V4h8v2M19 6l-1 14H6L5 6"
              stroke="#EF4444" strokeWidth="2" strokeLinecap="round"/>
          </svg>
        </div>

        {/* Text */}
        <h2 className="text-base font-semibold text-gray-900 text-center mb-1">
          Delete Location
        </h2>
        <p className="text-sm text-gray-500 text-center mb-6">
          Are you sure you want to delete{' '}

            {locationName}

'use client';

// ─────────────────────────────────────────────────────────────
// LOCATION TREE NODE WITH ACTIONS
// Shows "..." action button on hover → Delete option
// Add this to your existing LocationTreeView component

import { useState }            from 'react';
import { DeleteLocationModal } from './DeleteLocationModal';
import type { TreeNode }       from '@/types/locations';

interface NodeActionsProps {
  node: TreeNode;
}

export function LocationNodeActions({ node }: NodeActionsProps) {
  const [menuOpen,   setMenuOpen]   = useState(false);
  const [showDelete, setShowDelete] = useState(false);

  // Don't show actions on root client node
  if (node.parentId === null) return null;

  return (
    <>
      {/* "..." button — shown on hover via parent group-hover */}
      <div className="relative">
        <button
          onClick={(e) => {
            e.stopPropagation();
            setMenuOpen(o => !o);
          }}
          className="p-1 rounded hover:bg-gray-200 text-gray-400
            hover:text-gray-600 transition-colors opacity-0
            group-hover:opacity-100"
          title="Location actions"
        >
          <svg width="14" height="14" viewBox="0 0 24 24" fill="currentColor">
            <circle cx="5"  cy="12" r="2"/>
            <circle cx="12" cy="12" r="2"/>
            <circle cx="19" cy="12" r="2"/>
          </svg>
        </button>

        {/* Dropdown menu */}
        {menuOpen && (
          <>
            {/* Backdrop to close menu */}
            <div
              className="fixed inset-0 z-40"
              onClick={() => setMenuOpen(false)}
            />
            <div className="absolute right-0 top-full mt-1 w-32 bg-white
              rounded-lg shadow-lg border border-gray-100 z-50 overflow-hidden">
              <button
                onClick={(e) => {
                  e.stopPropagation();
                  setMenuOpen(false);
                  setShowDelete(true);
                }}
                className="w-full flex items-center gap-2 px-3 py-2
                  text-sm text-red-600 hover:bg-red-50 transition-colors"
              >
                <svg width="13" height="13" viewBox="0 0 24 24" fill="none">
                  <path d="M3 6h18M8 6V4h8v2M19 6l-1 14H6L5 6"
                    stroke="currentColor" strokeWidth="2"
                    strokeLinecap="round"/>
                </svg>
                Delete
              </button>
            </div>
          </>
        )}
      </div>

      {/* Delete confirmation modal */}
      {showDelete && (
        <DeleteLocationModal
          locationId={node.id}
          locationName={node.label}
          onClose={() => setShowDelete(false)}
        />
      )}
    </>
  );
}
