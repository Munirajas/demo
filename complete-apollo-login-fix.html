'use client';

import {
  ApolloClient,
  InMemoryCache,
  from,
  HttpLink,
} from '@apollo/client';
import { createAuthLink, createErrorLink }
  from './apollo-links';
import { GQL_ENDPOINT }
  from '@/lib/constants';

let instance: ApolloClient<object> | null = null;

function createApolloClient(): ApolloClient<object> {
  return new ApolloClient({
    link: from([
      createErrorLink(), // 1. catch UNAUTHENTICATED
      createAuthLink(),  // 2. attach Bearer token
      new HttpLink({ uri: GQL_ENDPOINT }),
    ]),
    cache: new InMemoryCache(),
    defaultOptions: {
      watchQuery: {
        fetchPolicy: 'cache-and-network',
        errorPolicy: 'all',
      },
      query: { errorPolicy: 'all' },
    },
  });
}

export function getApolloClient(): ApolloClient<object> {
  if (!instance) instance = createApolloClient();
  return instance;
}

// Must call on logout — clears cache between users
export function resetApolloClient(): void {
  instance?.clearStore();
  instance = null;
}

'use client';

import {
  ApolloLink,
  Observable,
  from,
  fromPromise,
} from '@apollo/client';
// type-only import — required in Apollo 4.x strict mode
import type { FetchResult } from '@apollo/client';
import { onError }
  from '@apollo/client/link/error';
import { GQL_ENDPOINT }
  from '@/lib/constants';

// ── Auth Link ──────────────────────────────────
// Reads token from /api/auth/token
// httpOnly cookies cannot be read by JS directly
export function createAuthLink(): ApolloLink {
  return new ApolloLink((operation, forward) => {
    return new Observable<FetchResult>((observer) => {
      fetch('/api/auth/token')
        .then((r) => r.json())
        .then(({ accessToken }: {
          accessToken: string | null;
        }) => {
          if (accessToken) {
            operation.setContext(
              ({ headers = {} }: {
                headers?: Record<string, string>;
              }) => ({
                headers: {
                  ...headers,
                  Authorization: `Bearer ${accessToken}`,
                },
              })
            );
          }
        })
        .catch(() => {})
        .finally(() => {
          forward(operation).subscribe({
            next:     (v) => observer.next(v),
            error:    (e) => observer.error(e),
            complete: ()  => observer.complete(),
          });
        });
    });
  });
}

// ── Error Link ─────────────────────────────────
// Queues concurrent 401s — only one refresh fires
let isRefreshing = false;
let pendingQueue: Array<() => void> = [];

function drainQueue(): void {
  pendingQueue.forEach((r) => r());
  pendingQueue = [];
}

export function createErrorLink(): ApolloLink {
  return onError(
    ({ graphQLErrors, operation, forward }) => {
      const isAuth = graphQLErrors?.some(
        (e) => e.extensions?.code === 'UNAUTHENTICATED'
      );
      if (!isAuth) return;

      if (isRefreshing) {
        return fromPromise(
          new Promise<void>((res) => pendingQueue.push(res))
        ).flatMap(() => forward(operation));
      }

      isRefreshing = true;

      return fromPromise(
        fetch('/api/auth/refresh', { method: 'POST' })
          .then((res) => {
            if (!res.ok) throw new Error('refresh_failed');
            return res.json() as Promise<{
              accessToken: string;
            }>;
          })
          .then(({ accessToken }) => {
            operation.setContext(
              ({ headers = {} }: {
                headers?: Record<string, string>;
              }) => ({
                headers: {
                  ...headers,
                  Authorization: `Bearer ${accessToken}`,
                },
              })
            );
            drainQueue();
          })
          .catch(() => {
            pendingQueue = [];
            window.location.href = '/login';
          })
          .finally(() => {
            isRefreshing = false;
          })
      ).flatMap(() => forward(operation));
    }
  );
}

Stores all 3 tokens in httpOnly cookies.
No API route. No extra fetch. Tokens never visible in browser network tab.
Uses GraphQLClient from graphql-request. Correct field names from your schema. 
'use server';

import { GraphQLClient, ClientError }
  from 'graphql-request';
import { redirect }
  from 'next/navigation';
import { GQL_ENDPOINT, ROUTES }
  from '@/lib/constants';
import { setAuthCookies }
  from '@/modules/auth/lib/cookies';
import type {
  LoginMutationResponse,
  ActionResult,
} from '@/types/auth.types';

// Matches your exact BE schema
const LOGIN_MUTATION = `
  mutation Login(
    $email: String!
    $password: String!
  ) {
    login(email: $email, password: $password) {
      token
      idToken
      refreshToken
    }
  }
`;

export async function loginAction(input: {
  email: string;
  password: string;
}): Promise<ActionResult> {
  const { email, password } = input;

  // Server-side validation
  if (!email?.trim() || !password) {
    return {
      success: false,
      error: 'Email and password are required.',
    };
  }

  try {
    const client = new GraphQLClient(GQL_ENDPOINT);

    const data = await client
      .request<LoginMutationResponse>(
        LOGIN_MUTATION,
        {
          email: email.trim().toLowerCase(),
          password,
        }
      );

    // Store all 3 tokens as httpOnly cookies
    // Also decodes idToken.sub → stores userId
    // (needed for refreshAuth mutation)
    await setAuthCookies(data.login);

  } catch (error) {
    // Parse BE error messages for user feedback
    if (error instanceof ClientError) {
      const msg =
        error.response?.errors?.[0]?.message ?? '';

      if (
        msg.toLowerCase().includes('invalid') ||
        msg.toLowerCase().includes('incorrect') ||
        msg.toLowerCase().includes('wrong')
      ) {
        return {
          success: false,
          error: 'Invalid email or password.',
        };
      }

      if (msg.toLowerCase().includes('not found')) {
        return {
          success: false,
          error: 'Account not found.',
        };
      }

      return {
        success: false,
        error: msg || 'Login failed. Please try again.',
      };
    }

    // Network or unexpected error
    console.error('[loginAction Error]', error);
    return {
      success: false,
      error: 'Unable to connect. Check your connection.',
    };
  }

  // redirect() MUST be outside try/catch
  // It throws internally — catch would swallow it
  redirect(ROUTES.DASHBOARD);
}

'use client';

import { useState, useTransition }
  from 'react';
import { useForm }
  from 'react-hook-form';
import { zodResolver }
  from '@hookform/resolvers/zod';
import { z } from 'zod';
import { loginAction }
  from '@/modules/auth/actions/loginAction';

const schema = z.object({
  email: z.string()
    .min(1, 'Email is required')
    .email('Enter a valid email'),
  password: z.string()
    .min(1, 'Password is required')
    .min(6, 'Minimum 6 characters'),
});

type FormValues = z.infer<typeof schema>;

export function LoginForm() {
  const [serverError, setServerError] =
    useState<string | null>(null);
  const [isPending, startTransition] =
    useTransition();

  const {
    register,
    handleSubmit,
    formState: { errors },
  } = useForm<FormValues>({
    resolver: zodResolver(schema),
  });

  function onSubmit(values: FormValues) {
    setServerError(null);
    startTransition(async () => {
      const result = await loginAction(values);
      // Only runs if action returned error
      // Redirect means this line never executes
      if (!result.success) {
        setServerError(result.error);
      }
    });
  }

  return (
    <div className="login-card">

      <div className="login-header">
        <h1 className="login-title">Admin Portal</h1>
        <p className="login-subtitle">
          Sign in to continue
        </p>
      </div>

      {serverError && (
        <div className="error-banner" role="alert">
          {serverError}
        </div>
      )}

      <div className="form-fields">

        <div className="field-group">
          <label htmlFor="email">Email</label>
          <input
            id="email"
            type="email"
            autoComplete="email"
            placeholder="you@company.com"
            disabled={isPending}
            {...register('email')}
          />
          {errors.email && (
            <span className="field-error">
              {errors.email.message}
            </span>
          )}
        </div>

        <div className="field-group">
          <label htmlFor="password">Password</label>
          <input
            id="password"
            type="password"
            autoComplete="current-password"
            placeholder="●●●●●●●●"
            disabled={isPending}
            {...register('password')}
          />
          {errors.password && (
            <span className="field-error">
              {errors.password.message}
            </span>
          )}
        </div>

        <button
          onClick={handleSubmit(onSubmit)}
          disabled={isPending}
          className="submit-button"
        >
          {isPending ? 'Signing in...' : 'Sign in'}
        </button>

      </div>
    </div>
  );
}
