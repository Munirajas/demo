Recurring schedule builder — RRULE-based (v3)

Rebuilt the scheduling piece on rrule.js, a JS implementation of the RFC 5545 (iCalendar) recurrence standard — the same model Outlook and Google Calendar use, which is why it naturally covers every pattern in your mockup: single occurrence, daily, weekly, "day N of every M months," "the 2nd Thursday of every month," "on a fixed date every year," and "the 2nd Thursday of July every year."

  • Structured rule is the source of truth. The exact fields the UI captures are what gets sent to BE and should be persisted as-is — this is what makes edit/prepopulate actually work: reopen the drawer, hand the stored rule object back into the form, done.
  • Standard RRULE text is also generated alongside the structured rule, for portability to anything that wants to consume a standard recurrence string later.
  • Expanded occurrences are still generated and sent (per requirement #4), but explicitly labeled as a derived cache, not the source of truth.
  • New component is fully generic — RecurrenceRuleBuilder has zero knowledge of amenities. It's a value/onChange controlled component, reusable for any entity that needs recurring scheduling later.
  • Dropped moment.js in favor of rrule.js's own date handling — one less aging dependency.

lib/recurrence-types.ts new

The structured rule shape — this is exactly what's sent to BE and exactly what should come back on edit. Field names map directly to the RRULE concepts rrule.js expects.

export type Frequency = 'NONE' | 'DAILY' | 'WEEKLY' | 'MONTHLY' | 'YEARLY';
export type WeekDay = 'SUN' | 'MON' | 'TUE' | 'WED' | 'THU' | 'FRI' | 'SAT';
export type MonthlyMode = 'DAY_OF_MONTH' | 'WEEKDAY_POSITION';
export type YearlyMode = 'ON_DATE' | 'WEEKDAY_POSITION';
export type SetPosition = 1 | 2 | 3 | 4 | -1; // 1st..4th, -1 = last
export type RecurrenceEndMode = 'ON_DATE' | 'AFTER_OCCURRENCES' | 'NEVER';

export interface RecurrenceRuleValues {
  // Defines the first occurrence AND, via (end - start), the duration
  // reused for every subsequent occurrence.
  startDate: string; // "YYYY-MM-DD"
  startTime: string; // "HH:mm"
  endDate: string; // "YYYY-MM-DD"
  endTime: string; // "HH:mm"

  frequency: Frequency;
  interval: number; // "every N days/weeks/months/years"

  // WEEKLY
  daysOfWeek: WeekDay[];

  // MONTHLY
  monthlyMode: MonthlyMode;
  monthlyDayOfMonth: number; // e.g. 9 -> "Day 9 of every N months"
  monthlySetPosition: SetPosition; // e.g. 2 -> "second ___ of every N months"
  monthlyWeekday: WeekDay;

  // YEARLY
  yearlyMode: YearlyMode;
  yearlyMonth: number; // 1-12
  yearlyDayOfMonth: number;
  yearlySetPosition: SetPosition;
  yearlyWeekday: WeekDay;

  // Recurrence end condition (irrelevant when frequency === 'NONE')
  recurrenceEndMode: RecurrenceEndMode;
  recurrenceEndDate?: string;
  occurrenceCount?: number;
}

lib/recurrence-engine.ts new

Builds an RRule instance from the structured values, expands occurrences (capped for open-ended "no end date" rules), and produces the full save payload.

Timezone/DST handling here is a first pass using native Date — worth a dedicated review with real test cases (DST transitions, multi-timezone locations) before this goes to production, same caution as any recurrence engine. 
import { RRule, type Weekday } from 'rrule';
import type {
  RecurrenceRuleValues,
  WeekDay,
  GeneratedOccurrence,
  ScheduleSavePayload,
} from './recurrence-types';

const WEEKDAY_MAP: Record<WeekDay, Weekday> = {
  SUN: RRule.SU, MON: RRule.MO, TUE: RRule.TU, WED: RRule.WE,
  THU: RRule.TH, FRI: RRule.FR, SAT: RRule.SA,
};

// Safety limits for open-ended ("no end date") rules -- without these, a
// NEVER-ending weekly rule would try to expand indefinitely.
const MAX_OCCURRENCES_SAFEGUARD = 366;
const MAX_HORIZON_YEARS = 2;

function combine(dateStr: string, timeStr: string): Date {
  const [h, m] = timeStr.split(':').map(Number);
  const d = new Date(`${dateStr}T00:00:00`);
  d.setHours(h, m, 0, 0);
  return d;
}

/** Builds an RRule instance, or null for a one-off (non-recurring) event. */
export function buildRRule(values: RecurrenceRuleValues): RRule | null {
  if (values.frequency === 'NONE') return null;

  const dtstart = combine(values.startDate, values.startTime);
  const options: Partial<RRule.Options> = { dtstart, interval: values.interval || 1 };

  if (values.frequency === 'DAILY') {
    options.freq = RRule.DAILY;
  } else if (values.frequency === 'WEEKLY') {
    options.freq = RRule.WEEKLY;
    options.byweekday = values.daysOfWeek.map((d) => WEEKDAY_MAP[d]);
  } else if (values.frequency === 'MONTHLY') {
    options.freq = RRule.MONTHLY;
    if (values.monthlyMode === 'DAY_OF_MONTH') {
      options.bymonthday = [values.monthlyDayOfMonth];
    } else {
      // e.g. "the 2nd Thursday of every month"
      options.byweekday = [WEEKDAY_MAP[values.monthlyWeekday].nth(values.monthlySetPosition)];
    }
  } else if (values.frequency === 'YEARLY') {
    options.freq = RRule.YEARLY;
    options.bymonth = [values.yearlyMonth];
    if (values.yearlyMode === 'ON_DATE') {
      options.bymonthday = [values.yearlyDayOfMonth];
    } else {
      // e.g. "the 2nd Thursday of July every year"
      options.byweekday = [WEEKDAY_MAP[values.yearlyWeekday].nth(values.yearlySetPosition)];
    }
  }

  if (values.recurrenceEndMode === 'ON_DATE' && values.recurrenceEndDate) {
    options.until = combine(values.recurrenceEndDate, '23:59');
  } else if (values.recurrenceEndMode === 'AFTER_OCCURRENCES' && values.occurrenceCount) {
    options.count = values.occurrenceCount;
  }
  // NEVER -> no until/count; expansion below applies the safety cap instead.

  return new RRule(options as RRule.Options);
}

/** Standard RRULE text, e.g. "DTSTART:...\nRRULE:FREQ=WEEKLY;...". */
export function toRRuleString(values: RecurrenceRuleValues): string | null {
  const rule = buildRRule(values);
  return rule ? rule.toString() : null;
}

/** Expands the rule into concrete start/end timestamps. */
export function generateOccurrences(values: RecurrenceRuleValues): GeneratedOccurrence[] {
  const seedStart = combine(values.startDate, values.startTime);
  const seedEnd = combine(values.endDate, values.endTime);
  const durationMs = Math.max(seedEnd.getTime() - seedStart.getTime(), 0);

  if (values.frequency === 'NONE') {
    return [{ startDateTime: seedStart.toISOString(), endDateTime: seedEnd.toISOString() }];
  }

  const rule = buildRRule(values);
  if (!rule) return [];

  const horizonCutoff = new Date(seedStart);
  horizonCutoff.setFullYear(horizonCutoff.getFullYear() + MAX_HORIZON_YEARS);

  const starts =
    values.recurrenceEndMode === 'NEVER'
      ? rule.all((date, i) => i < MAX_OCCURRENCES_SAFEGUARD && date <= horizonCutoff)
      : rule.all();

  return starts.map((start) => ({
    startDateTime: start.toISOString(),
    endDateTime: new Date(start.getTime() + durationMs).toISOString(),
  }));
}

/** Full payload handed to the save mutation. */
export function buildScheduleSavePayload(values: RecurrenceRuleValues): ScheduleSavePayload {
  return {
    rule: values,
    rruleText: toRRuleString(values),
    timezone: Intl.DateTimeFormat().resolvedOptions().timeZone,
    occurrences: generateOccurrences(values),
  };
}

The full Outlook-style recurrence UI: start/end date+time, frequency, and the conditional daily/weekly/monthly/yearly sub-patterns, plus the recurrence-end options. Purely a controlled value/onChange component with no amenity-specific logic — this is the piece to reuse for any future entity that needs scheduling.

'use client';

import type {
  RecurrenceRuleValues, WeekDay, SetPosition, Frequency,

const ALL_DAYS: { key: WeekDay; label: string }[] = [
  { key: 'SUN', label: 'Sun' }, { key: 'MON', label: 'Mon' }, { key: 'TUE', label: 'Tue' },
  { key: 'WED', label: 'Wed' }, { key: 'THU', label: 'Thu' }, { key: 'FRI', label: 'Fri' },
  { key: 'SAT', label: 'Sat' },
];

const SET_POSITIONS: { value: SetPosition; label: string }[] = [
  { value: 1, label: 'First' }, { value: 2, label: 'Second' }, { value: 3, label: 'Third' },
  { value: 4, label: 'Fourth' }, { value: -1, label: 'Last' },
];

const MONTHS = [
  'January', 'February', 'March', 'April', 'May', 'June',
  'July', 'August', 'September', 'October', 'November', 'December',
];

interface Props {
  value: RecurrenceRuleValues;
  onChange: (value: RecurrenceRuleValues) => void;
  error?: string | null;
}

export function RecurrenceRuleBuilder({ value, onChange, error }: Props) {
  function patch(partial: Partial<RecurrenceRuleValues>) {
    onChange({ ...value, ...partial });
  }

  function toggleDay(day: WeekDay) {
    const has = value.daysOfWeek.includes(day);
    patch({ daysOfWeek: has ? value.daysOfWeek.filter((d) => d !== day) : [...value.daysOfWeek, day] });
  }

  return (
    <div className="flex flex-col gap-4">
      {error && <p className="text-sm text-red-600">{error}</p>}

      {/* Start / end date-time -- also defines per-occurrence duration */}
      <div className="grid grid-cols-2 gap-3">
        <div className="flex flex-col gap-1">
          <label className="text-sm font-medium text-gray-700">Start</label>
          <div className="flex gap-2">
            <input type="date" className="w-full rounded-md border border-gray-300 px-3 py-2 text-sm"
              value={value.startDate} onChange={(e) => patch({ startDate: e.target.value })} />
            <input type="time" className="w-32 rounded-md border border-gray-300 px-3 py-2 text-sm"
              value={value.startTime} onChange={(e) => patch({ startTime: e.target.value })} />
          </div>
        </div>
        <div className="flex flex-col gap-1">
          <label className="text-sm font-medium text-gray-700">End</label>
          <div className="flex gap-2">
            <input type="date" className="w-full rounded-md border border-gray-300 px-3 py-2 text-sm"
              value={value.endDate} onChange={(e) => patch({ endDate: e.target.value })} />
            <input type="time" className="w-32 rounded-md border border-gray-300 px-3 py-2 text-sm"
              value={value.endTime} onChange={(e) => patch({ endTime: e.target.value })} />
          </div>
        </div>
      </div>
      <p className="text-xs text-gray-500 -mt-2">
        The gap between start and end is reused as the duration for every generated occurrence.
      </p>

      {/* Frequency */}
      <div className="flex flex-col gap-1">
        <label className="text-sm font-medium text-gray-700">Frequency</label>
        <select className="w-full rounded-md border border-gray-300 px-3 py-2 text-sm"
          value={value.frequency}
          onChange={(e) => patch({ frequency: e.target.value as Frequency })}>
          <option value="NONE">Does not repeat</option>
          <option value="DAILY">Daily</option>
          <option value="WEEKLY">Weekly</option>
          <option value="MONTHLY">Monthly</option>
          <option value="YEARLY">Yearly</option>
        </select>
      </div>

      {value.frequency === 'DAILY' && (
        <div className="flex items-center gap-2">
          <span className="text-sm text-gray-700">Recur every</span>
          <input type="number" min={1} className="w-20 rounded-md border border-gray-300 px-3 py-2 text-sm"
            value={value.interval} onChange={(e) => patch({ interval: Number(e.target.value) || 1 })} />
          <span className="text-sm text-gray-600">day(s)</span>
        </div>
      )}

      {value.frequency === 'WEEKLY' && (
        <>
          <div className="flex items-center gap-2">
            <span className="text-sm text-gray-700">Recur every</span>
            <input type="number" min={1} className="w-20 rounded-md border border-gray-300 px-3 py-2 text-sm"
              value={value.interval} onChange={(e) => patch({ interval: Number(e.target.value) || 1 })} />
            <span className="text-sm text-gray-600">week(s)</span>
          </div>
          <div className="flex flex-col gap-1">
            <label className="text-sm font-medium text-gray-700">Days of the week</label>
            <div className="flex flex-wrap gap-2">
              {ALL_DAYS.map((day) => {
                const active = value.daysOfWeek.includes(day.key);
                return (
                  <button key={day.key} type="button" onClick={() => toggleDay(day.key)}
                    className={`text-xs px-3 py-1.5 rounded-full border ${

          <label className="text-sm font-medium text-gray-700">Recurrence ends</label>

          <div className="flex items-center gap-2">
            <input type="radio" id="endOnDate" checked={value.recurrenceEndMode === 'ON_DATE'}
              onChange={() => patch({ recurrenceEndMode: 'ON_DATE' })} />
            <label htmlFor="endOnDate" className="text-sm text-gray-700 w-20">End by</label>
            <input type="date" disabled={value.recurrenceEndMode !== 'ON_DATE'}
              className="flex-1 rounded-md border border-gray-300 px-3 py-2 text-sm disabled:bg-gray-50"
              value={value.recurrenceEndDate ?? ''}
              onChange={(e) => patch({ recurrenceEndDate: e.target.value })} />
          </div>

          <div className="flex items-center gap-2">
            <input type="radio" id="endAfter" checked={value.recurrenceEndMode === 'AFTER_OCCURRENCES'}
              onChange={() => patch({ recurrenceEndMode: 'AFTER_OCCURRENCES' })} />
            <label htmlFor="endAfter" className="text-sm text-gray-700 w-20">End after</label>
            <input type="number" min={1} disabled={value.recurrenceEndMode !== 'AFTER_OCCURRENCES'}
              className="w-24 rounded-md border border-gray-300 px-3 py-2 text-sm disabled:bg-gray-50"
              value={value.occurrenceCount ?? ''}
              onChange={(e) => patch({ occurrenceCount: Number(e.target.value) || undefined })} />
            <span className="text-sm text-gray-600">occurrences</span>
          </div>

          <div className="flex items-center gap-2">
            <input type="radio" id="noEndDate" checked={value.recurrenceEndMode === 'NEVER'}
              onChange={() => patch({ recurrenceEndMode: 'NEVER' })} />
            <label htmlFor="noEndDate" className="text-sm text-gray-700">No end date</label>
          </div>

          {value.recurrenceEndMode === 'NEVER' && (
            <p className="text-xs text-gray-500">
              Occurrences are generated up to 2 years out; more are added automatically as that window approaches.
            </p>
          )}
        </div>
      )}
    </div>
  );
}
Amenity-specific form
components/forms/ScheduledAdmissionForm.tsx changed

Same two-step shape as v2 (owns its own Next/Back/Save/Cancel, fires its own mutation), now wired to RecurrenceRuleBuilder and buildScheduleSavePayload. Edit-mode prefill now works cleanly: the initial state reads directly from amenity.schedule.rule.

'use client';

import { useState } from 'react';
import { useInsertAmenity } from '../../hooks/useInsertAmenity';
import { useUpdateAmenity } from '../../hooks/useUpdateAmenity';
import { GeneralInfoFields } from '../GeneralInfoFields';
import { RecurrenceRuleBuilder } from '../schedule/RecurrenceRuleBuilder';
import { StepIndicator, type StepIndicatorStep } from '../StepIndicator';
import { buildScheduleSavePayload } from '../../lib/recurrence-engine';
import { EMPTY_RECURRENCE_RULE, type RecurrenceRuleValues } from '../../lib/recurrence-types';
import type { AmenityTypeFormProps } from './types';

type Step = 'general' | 'scheduler';

export function ScheduledAdmissionForm({
  generalInfo,
  onGeneralInfoChange,
  isEditing,
  amenity,
  locationFunctionId,
  locationSlug,
  onClose,
}: AmenityTypeFormProps) {
  const [step, setStep] = useState<Step>('general');

  // Prefill directly from the persisted rule -- this is what the "structured
  // rule as source of truth" design buys us: no reconstruction needed.
  const [recurrenceRule, setRecurrenceRule] = useState<RecurrenceRuleValues>(
    () => (amenity?.schedule?.rule as RecurrenceRuleValues | undefined) ?? EMPTY_RECURRENCE_RULE,
  );
  const [error, setError] = useState<string | null>(null);

  const insert = useInsertAmenity(locationFunctionId, locationSlug);
  const update = useUpdateAmenity(locationFunctionId, locationSlug);
  const isSaving = insert.isPending || update.isPending;

  const steps: StepIndicatorStep[] = [
    { id: 'general', label: 'General info', status: step === 'general' ? 'active' : 'done' },
    { id: 'scheduler', label: 'Scheduling', status: step === 'scheduler' ? 'active' : 'pending' },
  ];

  function validateGeneralInfo(): boolean {
    if (!generalInfo.amenityTypeId) { setError('Please select an amenity type.'); return false; }
    if (!generalInfo.amenityName.trim()) { setError('Name is required.'); return false; }
    setError(null);
    return true;
  }

  function validateSchedule(): boolean {
    if (!recurrenceRule.startDate || !recurrenceRule.endDate) {
      setError('Start and end date are required.');
      return false;
    }
    if (recurrenceRule.frequency === 'WEEKLY' && recurrenceRule.daysOfWeek.length === 0) {
      setError('Select at least one day.');
      return false;
    }
    setError(null);
    return true;
  }

  function handleNext() {
    if (!validateGeneralInfo()) return;
    setStep('scheduler');
  }

  function handleSave() {
    if (!validateSchedule()) return;

    const payload = {
      amenityTypeId: Number(generalInfo.amenityTypeId),
      amenityName: generalInfo.amenityName.trim(),
      amenityDescription: generalInfo.amenityDescription.trim(),
      schedule: buildScheduleSavePayload(recurrenceRule),
    };

    // NOTE: confirm field names/shape against InsertAmenityInput /
    // UpdateAmenityInput once BE finalizes the schedule contract -- see the
    // JSON payload example and suggested schema shared alongside this file.
    if (isEditing && amenity) {
      update.mutate(
        { amenityId: amenity.amenityId, locationFunctionId, ...payload },
        { onSuccess: onClose },
      );
    } else {
      insert.mutate(
        { locationFunctionId, ...payload },
        { onSuccess: onClose },
      );
    }
  }

  return (
    <div className="flex flex-col gap-4">
      <StepIndicator steps={steps} onStepClick={(i) => setStep(i === 0 ? 'general' : 'scheduler')} />

      {step === 'general' ? (
        <>
          <GeneralInfoFields values={generalInfo} onChange={onGeneralInfoChange} error={error} />
          <div className="flex justify-end gap-2 pt-3 border-t border-gray-100 mt-1">
            <button type="button" onClick={onClose}
              className="px-4 py-2 text-sm rounded-md border border-gray-300 text-gray-700">
              Cancel
            </button>
            <button type="button" onClick={handleNext}
              className="px-4 py-2 text-sm rounded-md bg-blue-600 text-white">
              Next
            </button>
          </div>
        </>
      ) : (
        <>
          <RecurrenceRuleBuilder value={recurrenceRule} onChange={setRecurrenceRule} error={error} />
          <div className="flex justify-between gap-2 pt-3 border-t border-gray-100 mt-1">
            <button type="button" onClick={() => setStep('general')}
              className="px-4 py-2 text-sm rounded-md border border-gray-300 text-gray-700">
              Back
            </button>
            <div className="flex gap-2">
              <button type="button" onClick={onClose}
                className="px-4 py-2 text-sm rounded-md border border-gray-300 text-gray-700">
                Cancel
              </button>
              <button type="button" onClick={handleSave} disabled={isSaving}
                className="px-4 py-2 text-sm rounded-md bg-blue-600 text-white disabled:opacity-60">
                {isSaving ? 'Saving...' : 'Save'}
              </button>
            </div>
          </div>
        </>
      )}
    </div>
  );
}
