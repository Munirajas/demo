Confirmed from screenshot: addLocation mutation, type + parentLocationId + locationName. data returns new Location_Id as number.
// ─────────────────────────────────────────────────────────────
// ADD LOCATION MUTATION
// Confirmed from your GraphQL playground screenshot:
//   mutation: addLocation
//   input:    type, parentLocationId, locationName
//   response: success, message, data (new location ID as number)
// ─────────────────────────────────────────────────────────────

export const ADD_LOCATION_MUTATION = `
  mutation AddLocation(
    $type:             String!
    $parentLocationId: Int!
    $locationName:     String!
  ) {
    addLocation(request: {
      type:             $type
      parentLocationId: $parentLocationId
      locationName:     $locationName
    }) {
      success
      message
      data
    }
  }
`;

// Input matches your request schema exactly
export interface AddLocationInput {
  type:             string;   // always "client"
  parentLocationId: number;   // Int! — parent location id
  locationName:     string;   // new location name
}

// data is the new location ID returned by BE
export interface AddLocationResponse {
  addLocation: {
    success: boolean;
    message: string;
    data:    number;   // new Location_Id
  };
}
CLIENT src/modules/locations/hooks/useAddLocation.ts
Mutation hook with sonner toasts. Invalidates tree cache on success. Passes new location ID to onSuccess callback.
'use client';

// ─────────────────────────────────────────────────────────────
// useAddLocation hook
// Uses sonner for toast — simple drop-in toast library.
// Invalidates locations tree cache on success.
// ─────────────────────────────────────────────────────────────

import { useQueryClient }  from '@tanstack/react-query';
import { toast }           from 'sonner';
import { useAuthMutation } from '@/modules/auth/hooks/useAuthQuery';
import { locationKeys }    from '@/modules/locations/hooks/useLocations';
import {
  ADD_LOCATION_MUTATION,
  type AddLocationInput,
  type AddLocationResponse,
} from '../graphql/add-location.mutation';

export function useAddLocation(onSuccess?: (newLocationId: number) => void) {
  const queryClient = useQueryClient();

  return useAuthMutation<AddLocationResponse, AddLocationInput>(
    ADD_LOCATION_MUTATION,
    {
      onSuccess: (data) => {
        const result = data.addLocation;

        if (!result.success) {
          toast.error(result.message || 'Failed to create location.');
          return;
        }

        // Invalidate tree so sidebar refreshes with new location
        queryClient.invalidateQueries({
          queryKey: locationKeys.tree(),
        });

        toast.success('Location created successfully!');

        // Pass the new location ID to parent (e.g. navigate to it)
        onSuccess?.(result.data);
      },

      onError: (error) => {
        toast.error(error.message || 'Something went wrong. Please try again.');
      },
    }
  );
}
CLIENT src/modules/locations/components/createLocation/CreateLocationModal.tsx
Full form: parent dropdown from cache + location name field. Zod validation. Loading spinner. Closes on success.
'use client';

// ─────────────────────────────────────────────────────────────
// CREATE LOCATION MODAL
// Uses confirmed API schema from screenshot.
// Parent dropdown + location name field.
// Zod validation + loading spinner.
// ─────────────────────────────────────────────────────────────

import { useForm }              from 'react-hook-form';
import { zodResolver }          from '@hookform/resolvers/zod';
import { z }                    from 'zod';
import { useLocationsTree }     from '@/modules/locations/hooks/useLocations';
import { useAddLocation }       from '@/modules/locations/hooks/useAddLocation';
import { flattenTreeForDropdown } from '@/modules/locations/lib/utils';

// ── Zod schema ────────────────────────────────────────────────
const schema = z.object({
  parentLocationId: z
    .string()
    .min(1, 'Please select a parent location'),
  locationName: z
    .string()
    .min(1, 'Location name is required')
    .min(2, 'Name must be at least 2 characters')
    .max(100, 'Name must be under 100 characters')
    .trim(),
});

type FormValues = z.infer<typeof schema>;

interface Props {
  onClose: () => void;
}

export function CreateLocationModal({ onClose }: Props) {
  // Uses existing TanStack cache — no extra API call for dropdown
  const { data: tree = [], isLoading: treeLoading } = useLocationsTree();
  const locationOptions = flattenTreeForDropdown(tree);

  const addLocation = useAddLocation(() => {
    // Close modal on success
    onClose();
  });

  const {
    register,
    handleSubmit,
    formState: { errors, isSubmitting },
  } = useForm<FormValues>({
    resolver: zodResolver(schema),
    defaultValues: {
      parentLocationId: '',
      locationName:     '',
    },
  });

  function onSubmit(values: FormValues) {
    addLocation.mutate({
      type:             'client',                     // always "client" per API
      parentLocationId: Number(values.parentLocationId),
      locationName:     values.locationName.trim(),
    });
  }

  const isLoading = addLocation.isPending;

  return (
    <div
      className="fixed inset-0 z-50 flex items-center justify-center bg-black/40"
      onClick={(e) => e.target === e.currentTarget && !isLoading && onClose()}
    >
      <div className="w-full max-w-md rounded-2xl bg-white shadow-2xl">

        {/* Header */}
        <div className="flex items-center justify-between border-b border-gray-100 px-6 py-4">
          <h2 className="text-base font-semibold text-gray-900">
            Create Location
          </h2>
          <button
            onClick={onClose}
            disabled={isLoading}
            className="text-gray-400 hover:text-gray-600 transition-colors
              disabled:opacity-40"
          >
            <svg width="16" height="16" viewBox="0 0 24 24" fill="none">
              <path d="M18 6L6 18M6 6l12 12"
                stroke="currentColor" strokeWidth="2" strokeLinecap="round"/>
            </svg>
          </button>
        </div>

        {/* Form fields */}
        <div className="px-6 py-5 space-y-4">

          {/* Parent Location */}
          <div>
            <label className="block text-xs font-medium text-gray-700 mb-1.5">
              Parent Location <span className="text-red-500">*</span>
            </label>
            <select
              disabled={treeLoading || isLoading}
              className={`w-full h-10 rounded-lg border px-3 text-sm
                bg-white outline-none transition-colors
                focus:border-blue-400 focus:ring-1 focus:ring-blue-400/30
                disabled:bg-gray-50 disabled:cursor-not-allowed
                ${errors.parentLocationId ? 'border-red-300' : 'border-gray-200'}`}
              {...register('parentLocationId')}
            >
              <option value="">
                {treeLoading ? 'Loading...' : 'Select parent location'}
              </option>
              {locationOptions.map((option) => (
                <option key={option.id} value={option.id}>
                  {option.indent}{option.label}
                </option>
              ))}
            </select>
            {errors.parentLocationId && (
              <p className="mt-1 text-xs text-red-500">
                {errors.parentLocationId.message}
              </p>
            )}
          </div>

          {/* Location Name */}
          <div>
            <label className="block text-xs font-medium text-gray-700 mb-1.5">
              Location Name <span className="text-red-500">*</span>
            </label>
            <input
              type="text"
              placeholder="e.g. Dallas 2"
              disabled={isLoading}
              className={`w-full h-10 rounded-lg border px-3 text-sm
                outline-none transition-colors
                focus:border-blue-400 focus:ring-1 focus:ring-blue-400/30
                disabled:bg-gray-50 disabled:cursor-not-allowed
                ${errors.locationName ? 'border-red-300' : 'border-gray-200'}`}
              {...register('locationName')}
            />
            {errors.locationName && (
              <p className="mt-1 text-xs text-red-500">
                {errors.locationName.message}
              </p>
            )}
          </div>

        </div>

        {/* Footer buttons */}
        <div className="flex gap-3 border-t border-gray-100 px-6 py-4">
          <button
            type="button"
            onClick={onClose}
            disabled={isLoading}
            className="flex-1 h-10 rounded-lg border border-gray-200
              text-sm text-gray-600 hover:bg-gray-50
              transition-colors disabled:opacity-40"
          >
            Cancel
          </button>
          <button
            type="button"
            onClick={handleSubmit(onSubmit)}
            disabled={isLoading || treeLoading}
            className="flex-1 h-10 rounded-lg bg-blue-600 hover:bg-blue-700
              text-white text-sm font-medium transition-colors
              disabled:opacity-60 flex items-center justify-center gap-2"
          >
            {isLoading ? (
              <>
                <svg className="w-4 h-4 animate-spin" viewBox="0 0 24 24" fill="none">
                  <circle className="opacity-25" cx="12" cy="12" r="10"
                    stroke="currentColor" strokeWidth="4"/>
                  <path className="opacity-75" fill="currentColor"
                    d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4z"/>
                </svg>
                Creating...
              </>
            ) : (
              'Create Location'
            )}
          </button>
        </div>

      </div>
    </div>
  );
}
