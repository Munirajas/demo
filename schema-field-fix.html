import 'server-only';

// ─────────────────────────────────────────────────────────────
// COOKIES — SERVER ONLY
//
// UPDATED: BE now returns userId directly in AuthToken.
// No more JWT decoding needed — simpler and more reliable.
//
// Stores 4 httpOnly cookies:
//   access_token   → token        (accessToken JWT)
//   id_token       → idToken
//   refresh_token  → refreshToken
//   user_id        → userId (from BE response directly)
// ─────────────────────────────────────────────────────────────

import { cookies } from 'next/headers';
import {
  COOKIE_KEYS,
  COOKIE_OPTIONS,
  REFRESH_COOKIE_OPTIONS,
} from '@/lib/constants';
import type { AuthToken } from '@/types/auth.types';

// ── WRITE — called after login or token refresh ───────────────
export async function setAuthCookies(tokens: AuthToken): Promise<void> {
  const cookieStore = await cookies();

  cookieStore.set(COOKIE_KEYS.ACCESS_TOKEN,  tokens.token,         COOKIE_OPTIONS);
  cookieStore.set(COOKIE_KEYS.ID_TOKEN,      tokens.idToken,       COOKIE_OPTIONS);
  cookieStore.set(COOKIE_KEYS.REFRESH_TOKEN, tokens.refreshToken,  REFRESH_COOKIE_OPTIONS);
  // BE returns userId directly — no JWT decode needed
  cookieStore.set(COOKIE_KEYS.USER_ID,       tokens.userId,        COOKIE_OPTIONS);
}

// ── READ ──────────────────────────────────────────────────────
export async function getAccessToken(): Promise<string | undefined> {
  return (await cookies()).get(COOKIE_KEYS.ACCESS_TOKEN)?.value;
}

export async function getRefreshToken(): Promise<string | undefined> {
  return (await cookies()).get(COOKIE_KEYS.REFRESH_TOKEN)?.value;
}

export async function getIdToken(): Promise<string | undefined> {
  return (await cookies()).get(COOKIE_KEYS.ID_TOKEN)?.value;
}

export async function getUserId(): Promise<string | undefined> {
  return (await cookies()).get(COOKIE_KEYS.USER_ID)?.value;
}

// ── CLEAR — called on logout ──────────────────────────────────
export async function clearAuthCookies(): Promise<void> {
  const cookieStore = await cookies();
  Object.values(COOKIE_KEYS).forEach((key) => cookieStore.delete(key));
}

'use server';

// ─────────────────────────────────────────────────────────────
// LOGIN ACTION — Server Action
//
// CRITICAL FIX: sending 'userName' not 'email'
// Your BE schema field is 'userName' (confirmed from playground).
//
// Flow:
//   LoginForm { email, password }
//     → loginAction maps email → userName for BE
//     → GraphQLClient sends login(userName, password)
//     → BE returns { token, idToken, refreshToken, userId }
//     → setAuthCookies stores all 4 as httpOnly cookies
//     → redirect('/dashboard')
// ─────────────────────────────────────────────────────────────

import { GraphQLClient, ClientError } from 'graphql-request';
import { redirect } from 'next/navigation';
import { GQL_ENDPOINT, ROUTES } from '@/lib/constants';
import { setAuthCookies } from '@/modules/auth/lib/cookies';
import { LOGIN_MUTATION } from '@/modules/auth/graphql/auth.mutations';
import type { LoginMutationResponse, ActionResult } from '@/types/auth.types';

export async function loginAction(
  input: { email: string; password: string }
): Promise<ActionResult> {
  const { email, password } = input;

  // ── Server-side validation ────────────────────────────────
  if (!email?.trim()) {
    return { success: false, error: 'Email is required.' };
  }
  if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email.trim())) {
    return { success: false, error: 'Enter a valid email address.' };
  }
  if (!password) {
    return { success: false, error: 'Password is required.' };
  }
  if (password.length < 6) {
    return { success: false, error: 'Password must be at least 6 characters.' };
  }

  // ── Call BE GraphQL ───────────────────────────────────────
  const client = new GraphQLClient(GQL_ENDPOINT, {
    headers: { 'Content-Type': 'application/json' },
  });

  try {
    const data = await client.request<LoginMutationResponse>(
      LOGIN_MUTATION,
      {
        // ⚠️ BE expects 'userName' — we map from the 'email' input field
        userName: email.trim().toLowerCase(),
        password: password,
      }
    );

    // ── Validate BE response ─────────────────────────────────
    if (!data?.login?.token) {
      return { success: false, error: 'Login failed: no access token received.' };
    }
    if (!data.login.idToken) {
      return { success: false, error: 'Login failed: no ID token received.' };
    }
    if (!data.login.refreshToken) {
      return { success: false, error: 'Login failed: no refresh token received.' };
    }
    if (!data.login.userId) {
      return { success: false, error: 'Login failed: no user ID received.' };
    }

    // ── Store all 4 values as httpOnly cookies ─────────────
    await setAuthCookies(data.login);

  } catch (error) {
    if (error instanceof ClientError) {
      const msg = error.response?.errors?.[0]?.message ?? '';

      if (
        msg.toLowerCase().includes('invalid') ||
        msg.toLowerCase().includes('incorrect') ||
        msg.toLowerCase().includes('wrong') ||
        msg.toLowerCase().includes('bad credentials')
      ) {
        return { success: false, error: 'Invalid email or password.' };
      }
      if (msg.toLowerCase().includes('not found') || msg.toLowerCase().includes('no user')) {
        return { success: false, error: 'No account found with this email.' };
      }
      if (msg.toLowerCase().includes('disabled') || msg.toLowerCase().includes('blocked')) {
        return { success: false, error: 'Account disabled. Contact support.' };
      }
      if (msg.toLowerCase().includes('missing tenants')) {
        return { success: false, error: 'Account not configured. Contact your administrator.' };
      }

      return { success: false, error: msg || 'Login failed. Please try again.' };
    }

    console.error('[loginAction] Unexpected error:', error);
    return { success: false, error: 'Unable to connect. Check your connection.' };
  }

  // redirect() MUST be outside try/catch
  redirect(ROUTES.DASHBOARD);
}
