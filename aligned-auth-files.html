import 'server-only';

import { cookies }    from 'next/headers';
import { COOKIE_KEYS, COOKIE_OPTIONS, REFRESH_COOKIE_OPTIONS } from '@/lib/constants';
import type { AuthToken, DecodedIdToken, SessionUser } from '@/types/auth.types';

// ── Decode JWT payload (no signature verification — BE handles that) ──
export function decodeToken<T>(token: string): T | null {
  try {
    const base64 = token.split('.')[1]
      .replace(/-/g, '+').replace(/_/g, '/');
    return JSON.parse(Buffer.from(base64, 'base64').toString('utf-8')) as T;
  } catch { return null; }
}

// ── WRITE — called after login or token refresh ───────────────
export async function setAuthCookies(tokens: AuthToken): Promise<void> {
  const store = await cookies();
  store.set(COOKIE_KEYS.ACCESS_TOKEN,  tokens.token,        COOKIE_OPTIONS);
  store.set(COOKIE_KEYS.ID_TOKEN,      tokens.idToken,      COOKIE_OPTIONS);
  store.set(COOKIE_KEYS.REFRESH_TOKEN, tokens.refreshToken, REFRESH_COOKIE_OPTIONS);
  store.set(COOKIE_KEYS.USER_ID,       tokens.userId,       COOKIE_OPTIONS);
}

// ── READ ──────────────────────────────────────────────────────
export async function getAccessToken()  { return (await cookies()).get(COOKIE_KEYS.ACCESS_TOKEN)?.value;  }
export async function getRefreshToken() { return (await cookies()).get(COOKIE_KEYS.REFRESH_TOKEN)?.value; }
export async function getIdToken()      { return (await cookies()).get(COOKIE_KEYS.ID_TOKEN)?.value;      }
export async function getUserId()       { return (await cookies()).get(COOKIE_KEYS.USER_ID)?.value;       }

// ── Get decoded session user — used by /api/auth/session ─────
export async function getSessionUser(): Promise<SessionUser | null> {
  const [idToken, userId] = await Promise.all([getIdToken(), getUserId()]);
  if (!idToken || !userId) return null;

  const p = decodeToken<DecodedIdToken>(idToken);
  if (!p) return null;

  const email = p.email ?? '';
  const name  = p.name
    ?? p['cognito:username']
    ?? `${p.given_name ?? ''} ${p.family_name ?? ''}`.trim()
    ?? email.split('@')[0];

  const parts    = name.trim().split(' ').filter(Boolean);
  const initials = parts.length >= 2
    ? `${parts[0][0]}${parts[parts.length - 1][0]}`.toUpperCase()
    : name.slice(0, 2).toUpperCase();

  return { userId, email, name, initials };
}

// ── CLEAR — called on logout ──────────────────────────────────
export async function clearAuthCookies(): Promise<void> {
  const store = await cookies();
  Object.values(COOKIE_KEYS).forEach(k => store.delete(k));
}

import 'server-only';

import { GraphQLClient, ClientError } from 'graphql-request';
import { GQL_ENDPOINT }               from '@/lib/constants';
import { getAccessToken, getUserId, setAuthCookies, clearAuthCookies } from '@/modules/auth/lib/cookies';
import type { RefreshAuthMutationResponse } from '@/types/auth.types';

const REFRESH_AUTH_MUTATION = `
  mutation RefreshAuth($userId: String!, $token: String!) {
    refreshAuth(userId: $userId, token: $token) {
      token
      idToken
      refreshToken
      userId
    }
  }
`;

export async function refreshAccessToken(): Promise<string | null> {
  const [token, userId] = await Promise.all([getAccessToken(), getUserId()]);
  if (!token || !userId) { console.warn('[refresh] missing token or userId'); return null; }

  try {
    const client = new GraphQLClient(GQL_ENDPOINT, {
      headers: { 'Content-Type': 'application/json' },
    });

    const data = await client.request<RefreshAuthMutationResponse>(
      REFRESH_AUTH_MUTATION, { userId, token }
    );
    if (!data?.refreshAuth?.token) return null;

    await setAuthCookies(data.refreshAuth);
    return data.refreshAuth.token;

  } catch (error) {
    if (error instanceof ClientError) {
      const msg = (error.response?.errors?.[0]?.message ?? '').toLowerCase();
      if (msg.includes('expired') || msg.includes('invalid') || msg.includes('unauthenticated')) {
        await clearAuthCookies();
      }
    }
    console.error('[refresh] failed:', error);
    return null;
  }
}

'use server';

import { GraphQLClient, ClientError } from 'graphql-request';
import { redirect }                   from 'next/navigation';
import { GQL_ENDPOINT, ROUTES }       from '@/lib/constants';
import { setAuthCookies }             from '@/modules/auth/lib/cookies';
import { LOGIN_MUTATION }             from '@/modules/auth/graphql/auth.mutations';
import type { LoginMutationResponse, ActionResult } from '@/types/auth.types';

export async function loginAction(
  input: { email: string; password: string }
): Promise<ActionResult> {
  const { email, password } = input;

  if (!email?.trim())
    return { success: false, error: 'Email is required.' };
  if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email.trim()))
    return { success: false, error: 'Enter a valid email.' };
  if (!password)
    return { success: false, error: 'Password is required.' };

  try {
    const client = new GraphQLClient(GQL_ENDPOINT, {
      headers: { 'Content-Type': 'application/json' },
    });

    const data = await client.request<LoginMutationResponse>(LOGIN_MUTATION, {
      userName: email.trim().toLowerCase(), // form field 'email' → BE field 'userName'
      password,
    });

    if (!data?.login?.token || !data.login.userId)
      return { success: false, error: 'Login failed: incomplete response.' };

    await setAuthCookies(data.login);

  } catch (error) {
    if (error instanceof ClientError) {
      const msg = error.response?.errors?.[0]?.message ?? '';
      if (msg.toLowerCase().includes('invalid') || msg.toLowerCase().includes('incorrect'))
        return { success: false, error: 'Invalid email or password.' };
      if (msg.toLowerCase().includes('missing tenants'))
        return { success: false, error: 'Account not configured. Contact your administrator.' };
      if (msg.toLowerCase().includes('disabled') || msg.toLowerCase().includes('blocked'))
        return { success: false, error: 'Account disabled. Contact support.' };
      return { success: false, error: msg || 'Login failed. Please try again.' };
    }
    console.error('[loginAction]', error);
    return { success: false, error: 'Unable to connect. Check your connection.' };
  }

  redirect(ROUTES.DASHBOARD); // outside try/catch — critical
}

'use server';

import { GraphQLClient, ClientError } from 'graphql-request';
import { redirect }                   from 'next/navigation';
import { GQL_ENDPOINT, ROUTES }       from '@/lib/constants';
import { getAccessToken, clearAuthCookies } from '@/modules/auth/lib/cookies';
import { END_SESSION_MUTATION }       from '@/modules/auth/graphql/auth.mutations';
import type { ActionResult }          from '@/types/auth.types';

export async function logoutAction(): Promise<ActionResult> {
  const token = await getAccessToken();

  // Step 1: Tell BE to invalidate tokens server-side
  // Do this BEFORE clearing cookies so we still have the token
  if (token) {
    try {
      await new GraphQLClient(GQL_ENDPOINT, {
        headers: { Authorization: `Bearer ${token}`, 'Content-Type': 'application/json' },
      }).request(END_SESSION_MUTATION);
    } catch (error) {
      // Log but never block logout — cookies must always be cleared
      if (error instanceof ClientError) {
        console.warn('[logoutAction] endSession failed:', error.response?.errors?.[0]?.message);
      }
    }
  }

  // Step 2: Clear all 4 httpOnly cookies
  await clearAuthCookies();

  // Step 3: Redirect to login
  redirect(ROUTES.LOGIN);
}

import 'server-only';

import { GraphQLClient, ClientError } from 'graphql-request';
import { GQL_ENDPOINT }               from '@/lib/constants';
import { getAccessToken }             from '@/modules/auth/lib/cookies';
import { refreshAccessToken }         from '@/modules/auth/lib/refresh';

function buildClient(token: string): GraphQLClient {
  return new GraphQLClient(GQL_ENDPOINT, {
    headers: {
      Authorization:   `Bearer ${token}`,
      'Content-Type': 'application/json',
    },
  });
}

function isAuthError(error: unknown): boolean {
  if (error instanceof ClientError) {
    const code = error.response?.errors?.[0]?.extensions?.code as string | undefined;
    return error.response?.status === 401 || code === 'UNAUTHENTICATED';
  }
  return false;
}

// ── Use this in ALL Server Components ────────────────────────
// Auto-refreshes expired token, retries once.
// Throws 'SESSION_EXPIRED' if refresh also fails.
export async function serverGqlFetch<
  TData,
  TVars extends object = Record<string, never>
>(query: string, variables?: TVars): Promise<TData> {
  let token = await getAccessToken();

  if (!token) {
    token = await refreshAccessToken();
    if (!token) throw new Error('SESSION_EXPIRED');
  }

  try {
    return await buildClient(token).request<TData>(query, variables);
  } catch (error) {
    if (isAuthError(error)) {
      const newToken = await refreshAccessToken();
      if (!newToken) throw new Error('SESSION_EXPIRED');
      return buildClient(newToken).request<TData>(query, variables);
    }
    throw error;
  }
}

import { NextRequest, NextResponse } from 'next/server';
import { PUBLIC_ROUTES, BYPASS_PREFIXES, ROUTES } from '@/lib/constants';

export default function middleware(req: NextRequest) {
  const { pathname } = req.nextUrl;

  // Always let internal routes through
  if (BYPASS_PREFIXES.some(p => pathname.startsWith(p)))
    return NextResponse.next();

  const token         = req.cookies.get('access_token')?.value;
  const isAuth        = Boolean(token);
  const isPublicRoute = PUBLIC_ROUTES.some(r => pathname === r || pathname.startsWith(r));

  // Not logged in → redirect to login, remember destination
  if (!isPublicRoute && !isAuth) {
    const url = new URL(ROUTES.LOGIN, req.url);
    url.searchParams.set('callbackUrl', pathname);
    return NextResponse.redirect(url);
  }

  // Already logged in → skip login/register pages
  if (isPublicRoute && isAuth)
    return NextResponse.redirect(new URL(ROUTES.DASHBOARD, req.url));

  return NextResponse.next();
}

'use client';

import { useQuery, useMutation, useQueryClient, type UseQueryOptions } from '@tanstack/react-query';
import { useRouter }         from 'next/navigation';
import { GraphQLClient, ClientError } from 'graphql-request';

const GQL_ENDPOINT = process.env.NEXT_PUBLIC_GRAPHQL_URL!;

async function getToken(): Promise<string | null> {
  try {
    const res = await fetch('/api/auth/token');
    if (!res.ok) return null;
    const { accessToken } = await res.json() as { accessToken: string | null };
    return accessToken;
  } catch { return null; }
}

async function tryRefresh(): Promise<string | null> {
  try {
    const res = await fetch('/api/auth/refresh', { method: 'POST' });
    if (!res.ok) return null;
    const { accessToken } = await res.json() as { accessToken: string | null };
    return accessToken;
  } catch { return null; }
}

function isAuthError(error: unknown): boolean {
  if (error instanceof ClientError) {
    const code = error.response?.errors?.[0]?.extensions?.code as string | undefined;
    return error.response?.status === 401 || code === 'UNAUTHENTICATED';
  }
  return false;
}

async function gqlFetch<TData>(
  query: string,
  variables?: object,
  onExpired?: () => void
): Promise<TData> {
  let token = await getToken();
  if (!token) {
    token = await tryRefresh();
    if (!token) { onExpired?.(); throw new Error('SESSION_EXPIRED'); }
  }

  const client = new GraphQLClient(GQL_ENDPOINT, {
    headers: { Authorization: `Bearer ${token}`, 'Content-Type': 'application/json' },
  });

  try {
    return await client.request<TData>(query, variables);
  } catch (error) {
    if (isAuthError(error)) {
      const newToken = await tryRefresh();
      if (!newToken) { onExpired?.(); throw new Error('SESSION_EXPIRED'); }
      return new GraphQLClient(GQL_ENDPOINT, {
        headers: { Authorization: `Bearer ${newToken}`, 'Content-Type': 'application/json' },
      }).request<TData>(query, variables);
    }
    throw error;
  }
}

// ── useAuthQuery — for fetching data ─────────────────────────
// Usage:
//   const { data, isLoading } = useAuthQuery(['users'], USERS_QUERY);
export function useAuthQuery<TData>(
  queryKey: unknown[],
  query: string,
  variables?: object,
  options?: Omit<UseQueryOptions<TData>, 'queryKey' | 'queryFn'>
) {
  const router = useRouter();
  return useQuery<TData>({
    queryKey,
    queryFn: () => gqlFetch<TData>(query, variables, () => router.push('/login?reason=expired')),
    staleTime: 1000 * 60 * 5,
    retry: 1,
    ...options,
  });
}

// ── useAuthMutation — for mutations ──────────────────────────
// Usage:
//   const mut = useAuthMutation(CREATE_USER, { invalidateKeys: [['users']] });
//   mut.mutate({ name: 'John' });
export function useAuthMutation<TData, TVars extends object = object>(
  mutation: string,
  options?: {
    onSuccess?:     (data: TData) => void;
    onError?:       (error: Error) => void;
    invalidateKeys?: unknown[][];
  }
) {
  const router      = useRouter();
  const queryClient = useQueryClient();

  return useMutation<TData, Error, TVars>({
    mutationFn: (variables: TVars) =>
      gqlFetch<TData>(mutation, variables, () => router.push('/login?reason=expired')),
    onSuccess: (data) => {
      options?.invalidateKeys?.forEach(key => queryClient.invalidateQueries({ queryKey: key }));
      options?.onSuccess?.(data);
    },
    onError: options?.onError,
  });
}

        onClick={handleLogout}
        disabled={isPending}
        className={className ?? 'flex items-center gap-2 w-full px-3 py-2 rounded-lg text-sm text-gray-400 hover:text-white hover:bg-gray-800 transition-colors disabled:opacity-50'}
      >
        {isPending ? 'Signing out...' : (children ?? 'Sign out')}
      </button>
    </div>
  );
}

CREATE CLIENT
CREATE — proactive refresh every 4 min + on tab visibility change (handles idle users).
'use client';

import { useEffect, useCallback } from 'react';
import { useQueryClient }         from '@tanstack/react-query';
import { useRouter }              from 'next/navigation';

const CHECK_INTERVAL_MS = 4 * 60 * 1000;  // every 4 minutes
const REFRESH_THRESHOLD = 5 * 60;          // refresh if < 5 min left (seconds)

export function TokenRefreshProvider({ children }: { children: React.ReactNode }) {
  const router      = useRouter();
  const queryClient = useQueryClient();

  const check = useCallback(async () => {
    try {
      const res = await fetch('/api/auth/token');
      if (!res.ok) return;
      const { accessToken } = await res.json() as { accessToken: string | null };
      if (!accessToken) return;

      // Decode exp from JWT payload
      const payload     = JSON.parse(atob(accessToken.split('.')[1])) as { exp: number };
      const secondsLeft = payload.exp - Math.floor(Date.now() / 1000);

      if (secondsLeft <= 0) {
        router.push('/login?reason=expired');
        return;
      }

      if (secondsLeft < REFRESH_THRESHOLD) {
        const refreshRes = await fetch('/api/auth/refresh', { method: 'POST' });
        if (refreshRes.ok) {
          queryClient.invalidateQueries(); // re-fetch with new token
        } else {
          router.push('/login?reason=expired');
        }
      }
    } catch { /* network error — don't redirect, might be momentary */ }
  }, [router, queryClient]);

  useEffect(() => {
    check(); // check on mount
    const interval = setInterval(check, CHECK_INTERVAL_MS);

    // When user returns to tab after being idle
    const onVisible = () => { if (document.visibilityState === 'visible') check(); };
    document.addEventListener('visibilitychange', onVisible);

    return () => {
      clearInterval(interval);
      document.removeEventListener('visibilitychange', onVisible);
    };
  }, [check]);

  return <>{children}</>;
}
