// ─────────────────────────────────────────────────────────────
// SLUG UTILITIES
//
// URLs show location NAMES (slugified), code uses IDs internally.
//
// ⚠️ IMPORTANT CAVEAT (from your tree — "World Stage Theater"
// appears TWICE under different parents):
//   findNodeBySlug() returns the FIRST match in the tree.
//   If duplicate names exist as siblings-of-different-parents,
//   both slugs collide → only the first is reachable by URL.
//
//   Fix options if this becomes a problem:
//     a) BE enforces unique location names per client, OR
//     b) Append short id suffix only when a duplicate is detected:
//        "world-stage-theater" vs "world-stage-theater-60"
//   For now: first-match wins (simplest, matches your data mostly).
// ─────────────────────────────────────────────────────────────

import type { TreeNode } from '@/types/locations';

export function slugify(label: string): string {
  return label
    .toLowerCase()
    .trim()
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/^-+|-+$/g, '');
}

// DFS — find node whose slugified label matches `slug`,
// returns the node + its full ancestor path (root → node)
export function getNodeAndPathBySlug(
  nodes: TreeNode[],
  slug:  string,
  path:  TreeNode[] = []
): { node: TreeNode; path: TreeNode[] } | null {
  for (const node of nodes) {
    const current = [...path, node];
    if (slugify(node.label) === slug) {
      return { node, path: current };
    }
    if (node.children?.length) {
      const found = getNodeAndPathBySlug(node.children, slug, current);
      if (found) return found;
    }
  }
  return null;
}

// Build a /location/{slug} URL for a given node
export function locationUrl(node: TreeNode, fn: string): string {
  return `/location/${slugify(node.label)}/${fn}`;
}

import 'server-only';
import { cache } from 'react';
import { serverGqlFetch } from '@/lib/graphql/server-gql-fetcher';
import {
  CONFIG_KEY_TO_FUNCTION,
  FUNCTION_LABELS,
  type LocationFunction,
} from './location-functions.types';

// ─────────────────────────────────────────────────────────────
// GET LOCATION CONFIG — returns enabled functions for a location
//
// ⚠️ Update this query when BE confirms getLocationConfig schema.
// Assumed shape based on your config screenshot toggles.
// ─────────────────────────────────────────────────────────────

const GET_LOCATION_CONFIG = `
  mutation GetLocationConfig($type: String!, $locationId: Int!) {
    getLocationConfig(request: { type: $type, locationId: $locationId }) {
      success
      message
      data
    }
  }
`;

interface GetLocationConfigResponse {
  getLocationConfig: {
    success: boolean;
    message: string;
    data:    Record<string, boolean>; // e.g. { physicalAddress: true, ... }
  };
}

// Config is always first, always enabled
const CONFIG_TAB: LocationFunction = {
  slug:    'config',
  label:   FUNCTION_LABELS['config'],
  enabled: true,
};

export const getLocationFunctions = cache(
  async (locationId: string): Promise<LocationFunction[]> => {
    try {
      const response = await serverGqlFetch<GetLocationConfigResponse>(
        GET_LOCATION_CONFIG,
        { type: 'client', locationId: Number(locationId) }
      );

      if (!response.getLocationConfig?.success) {
        return [CONFIG_TAB];
      }

      const configData = response.getLocationConfig.data;

      const enabledFunctions: LocationFunction[] = Object.entries(CONFIG_KEY_TO_FUNCTION)
        .filter(([configKey]) => configData?.[configKey])
        .map(([, slug]) => ({
          slug,
          label:   FUNCTION_LABELS[slug],
          enabled: true,
        }));

      return [CONFIG_TAB, ...enabledFunctions];

    } catch (error) {
      console.error('[getLocationFunctions]', error);
      // Fallback — at least Config is always available
      return [CONFIG_TAB];
    }
  }
);

'use client';

// ─────────────────────────────────────────────────────────────
// BREADCRUMBS — max 3 location crumbs + current function
//
// Short path (≤3 locations):
//   Museum of the Bible / MOTB DC / Aviation Gallery / Config
//
// Long path (>3 locations):
//   Museum of the Bible / ... / Aviation Gallery / Config
//                          ↑
//              click → dropdown lists hidden locations,
//              each clickable (navigates to that location, same fn)
// ─────────────────────────────────────────────────────────────

import { useState } from 'react';
import Link         from 'next/link';
import type { BreadcrumbItem } from '@/types/locations';

const MAX_LOCATION_CRUMBS = 3;

export function Breadcrumbs({ items }: { items: BreadcrumbItem[] }) {
  const [open, setOpen] = useState(false);

  if (!items.length) return null;

  // Last item = current function (never collapsed)
  // All preceding items = location path
  const fnItem        = items[items.length - 1];
  const locationItems = items.slice(0, -1);

  const needsCollapse = locationItems.length > MAX_LOCATION_CRUMBS;

  // Show: first location + last (MAX-1) locations + ellipsis for the rest
  const visibleLocations = needsCollapse
    ? [
        locationItems[0],
        ...locationItems.slice(-(MAX_LOCATION_CRUMBS - 1)),
      ]
    : locationItems;

  const hiddenLocations = needsCollapse
    ? locationItems.slice(1, -(MAX_LOCATION_CRUMBS - 1))
    : [];

  return (
    <nav className="flex items-center gap-1 flex-wrap text-sm min-w-0">
      {visibleLocations.map((item, i) => (
        <span key={item.href ?? item.label} className="flex items-center gap-1 min-w-0">
          {i > 0 && <Sep />}
          <Crumb item={item} />

          {/* Insert ellipsis dropdown right after the first item */}
          {needsCollapse && i === 0 && (
            <>
              <Sep />
              <div className="relative">
                <button
                  onClick={() => setOpen(o => !o)}
                  className="px-1.5 py-0.5 rounded text-gray-400
                    hover:text-gray-600 hover:bg-gray-100
                    text-sm font-medium transition-colors"
                  title="Show hidden locations"
                >
                  ...
                </button>

                {open && (
                  <>
                    <div
                      className="fixed inset-0 z-40"
                      onClick={() => setOpen(false)}
                    />
                    <div className="absolute left-0 top-full mt-1 min-w-[180px]
                      bg-white rounded-lg shadow-lg border border-gray-100
                      z-50 overflow-hidden py-1">
                      {hiddenLocations.map((h) => (
                        <Link
                          key={h.href}
                          href={h.href ?? '#'}
                          onClick={() => setOpen(false)}
                          className="block px-3 py-2 text-sm text-gray-600
                            hover:bg-gray-50 hover:text-gray-900
                            transition-colors truncate"
                        >
                          {h.label}
                        </Link>
                      ))}
                    </div>
                  </>
                )}
              </div>
            </>
          )}
        </span>
      ))}

      {/* Current function — always last, never a link */}
      <Sep />
      <span className="text-gray-900 font-medium truncate max-w-[160px]"
        title={fnItem.label}>
        {fnItem.label}
      </span>
    </nav>
  );
}

function Crumb({ item }: { item: BreadcrumbItem }) {
  if (item.href) {
    return (
      <Link
        href={item.href}
        className="text-gray-500 hover:text-gray-800 transition-colors
          truncate max-w-[160px]"
        title={item.label}
      >
        {item.label}
      </Link>
    );
  }
  return (
    <span className="text-gray-500 truncate max-w-[160px]" title={item.label}>
      {item.label}
    </span>
  );
}

function Sep() {
  return (
    <svg width="12" height="12" viewBox="0 0 24 24" fill="none"
      className="text-gray-300 shrink-0">
      <path d="M9 18l6-6-6-6" stroke="currentColor"
        strokeWidth="2" strokeLinecap="round"/>
    </svg>
  );
}

// src/app/(dashboard)/location/[slug]/[fn]/layout.tsx
// SERVER COMPONENT
//
// ─────────────────────────────────────────────────────────────
// URL: /location/{location-name-slug}/{function-slug}
//   e.g. /location/aviation-gallery/config
//        /location/aviation-gallery/physical-address
//
// This renders the RIGHT COLUMN content of (dashboard)/layout.tsx
// (which already has Header + Locations tree on the left).
//
// THIS layout further splits that right column into:
//   - Breadcrumb bar (top)
//   - Inner-left:  FunctionsNav (config, physical-address, ...)
//   - Inner-right: {children} = the function's page content
//
// DATA FETCHED HERE:
//   1. getLocationsTree()        — React cache(), likely already
//                                   fetched by (dashboard)/layout.tsx
//   2. getLocationFunctions(id)  — per-location enabled functions
//
// SLUG → ID RESOLUTION:
//   getNodeAndPathBySlug(tree, slug) walks the tree once,
//   returns both the matched node AND its full ancestor path
//   (used for breadcrumbs).
// ─────────────────────────────────────────────────────────────

import { notFound }             from 'next/navigation';
import { getLocationsTree }     from '@/modules/locations/graphql/server-locations';
import { getLocationFunctions } from '@/modules/locations/graphql/server-location-functions';
import { getNodeAndPathBySlug, slugify, locationUrl } from '@/modules/locations/lib/slug';
import { FUNCTION_LABELS, type FunctionSlug } from '@/modules/locations/graphql/location-functions.types';
import { Breadcrumbs }          from '@/components/nav/Breadcrumbs';
import { FunctionsNav }         from '@/components/layout/FunctionsNav';
import type { BreadcrumbItem }  from '@/types/locations';

interface Props {
  children: React.ReactNode;
  params:   Promise<{ slug: string; fn: string }>;
}

export default async function LocationFnLayout({ children, params }: Props) {
  const { slug, fn } = await params;

  // Tree fetch — React cache() dedupes if (dashboard)/layout.tsx
  // already called this in the same request
  const tree = await getLocationsTree();

  // Resolve slug → node + ancestor path
  const found = getNodeAndPathBySlug(tree, slug);
  if (!found) notFound();
  const { node, path } = found;

  // Per-location enabled functions (Config always first)
  const functions = await getLocationFunctions(node.id);

  // Validate fn slug — fallback to 'config' if unknown/disabled
  const fnLabel = FUNCTION_LABELS[fn as FunctionSlug] ?? fn;
  const isValidFn = functions.some(f => f.slug === fn);
  if (!isValidFn) notFound();

  // ── Breadcrumbs ──────────────────────────────────────────────
  // Each location in the path links to the SAME function (fn)
  // for that location — e.g. switching location keeps you on
  // "physical-address" if that's where you currently are.
  const breadcrumbs: BreadcrumbItem[] = [
    ...path.map(ancestor => ({
      label: ancestor.label,
      href:  ancestor.type === 'client'
        ? undefined                       // root client node — not clickable
        : locationUrl(ancestor, fn),
    })),
    { label: fnLabel }, // current function — always last, never a link
  ];

  return (
    <div className="flex flex-col h-full">

      {/* Breadcrumb bar */}
      <div className="px-6 py-3 border-b border-gray-100 bg-white shrink-0">
        <Breadcrumbs items={breadcrumbs} />
      </div>

      {/* Inner two-column shell */}
      <div className="flex flex-1 overflow-hidden">

        {/* Inner-left: Functions nav */}
        <aside className="w-48 border-r border-gray-200 flex flex-col shrink-0">
          <div className="p-3 border-b border-gray-100">
            <p className="text-xs font-semibold text-gray-500 uppercase tracking-wide truncate">
              {node.label}
            </p>
          </div>
          <div className="flex-1 overflow-y-auto">
            <FunctionsNav
              locationSlug={slug}
              activeFn={fn}
              functions={functions}
            />
          </div>
        </aside>

        {/* Inner-right: function page content */}
        <main className="flex-1 overflow-y-auto">
          {children}
        </main>
      </div>
    </div>
  );
}

          ))}
        </div>
        <div className="space-y-4">
          {right.map(key => (

              key={key}
              label={FUNCTION_LABELS[CONFIG_KEY_TO_FUNCTION[key]]}
              checked={enabled[key]}

            />
          ))}
        </div>
      </div>

      <p className="text-xs text-gray-400 mt-8">
        Location ID: {locationId}
      </p>
    </div>
  );
}

function ToggleRow({
  label, checked, onChange,
}: {
  label: string; checked: boolean; onChange: () => void;
}) {
  return (
    <div className="flex items-center gap-3">
      <button
        onClick={onChange}
        role="switch"
        aria-checked={checked}
        className={`relative w-9 h-5 rounded-full transition-colors shrink-0
          ${checked ? 'bg-blue-600' : 'bg-gray-200'}`}
      >
        <span
          className={`absolute top-0.5 left-0.5 w-4 h-4 rounded-full
            bg-white shadow-sm transition-transform
            ${checked ? 'translate-x-4' : 'translate-x-0'}`}
        />
      </button>
      <span className="text-sm text-gray-700">{label}</span>
    </div>
  );
}
