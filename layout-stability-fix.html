FunctionsNav.tsx — Replace Link with useTransition Root Fix

This is the main fix. useTransition navigates without triggering layout re-render. Spinner shows on clicked item only.

⚠️
Why not <Link>? Next.js <Link> re-evaluates the nearest layout boundary on navigation. Since [slug]/layout.tsx wraps Col 2, it re-renders and Col 2 disappears. router.push() inside useTransition only re-renders the page segment, leaving the layout intact.
src/components/nav/FunctionsNav.tsx 
'use client';

import { useTransition } from 'react';
import { useRouter, usePathname } from 'next/navigation';
import type { LocationFunction } from '@/types/locations';

interface Props {
  locationSlug: string;
  activeFn: string | null;
  functions: LocationFunction[];
}

/**
 * FunctionsNav — uses useTransition NOT Link.
 *
 * WHY: Link triggers [slug]/layout re-render → Col2 disappears.
 * useTransition + router.push() only re-renders [fn]/page.tsx
 * leaving Col2 completely intact during navigation.
 *
 * The isPending state from useTransition is used to:
 *   1. Show spinner on the clicked nav item
 *   2. Signal ContentArea to show grey overlay on Col3
 */
export function FunctionsNav({
  locationSlug,
  activeFn,
  functions,
}: Props) {
  const router = useRouter();
  const pathname = usePathname();
  const [isPending, startTransition] = useTransition();

  // Which fn slug is currently navigating (for per-item spinner)
  const [pendingSlug, setPendingSlug] =
    useState<string | null>(null);

  function handleNavigate(fnSlug: string) {
    const href = `/location/${locationSlug}/${fnSlug}`;
    if (pathname === href) return;

    setPendingSlug(fnSlug);
    startTransition(() => {
      router.push(href);
    });
  }

  // Clear pending slug when transition completes
  if (!isPending && pendingSlug) {
    setPendingSlug(null);
  }

  if (functions.length === 0) {
    return (
      <p className="px-4 py-3 text-xs text-gray-400">
        No functions available
      </p>
    );
  }

  return (
    <nav className="py-2" aria-label="Location functions">
      {functions.map((fn) => {
        const href = `/location/${locationSlug}/${fn.slug}`;
        const isActive = pathname === href;
        const isThisPending =
          isPending && pendingSlug === fn.slug;

        return (
          <button
            key={fn.slug}
            type="button"
            onClick={() => handleNavigate(fn.slug)}
            disabled={isPending}
            aria-current={isActive ? 'page' : undefined}
            className={[
              // Base styles — full width block button
              'flex w-full items-center justify-between',
              'px-4 py-2.5 text-sm transition-colors text-left',
              // Active state
              isActive
                ? 'border-l-2 border-blue-600 bg-blue-50'
                  + ' font-medium text-blue-700'
                : 'border-l-2 border-transparent text-gray-600'
                  + ' hover:bg-gray-50 hover:text-gray-900',
              // Dim all items while any transition is pending
              isPending ? 'opacity-50 cursor-not-allowed' : '',
            ].join(' ')}
          >
            <span>{fn.label}</span>

            {/* Spinner on clicked item only */}
            {isThisPending && (
              <svg
                className="h-3.5 w-3.5 animate-spin text-blue-500"
                viewBox="0 0 24 24"
                fill="none"
              >
                <circle
                  cx="12" cy="12" r="10"
                  stroke="currentColor"
                  strokeWidth="4"
                  className="opacity-25"
                />
                <path
                  fill="currentColor"
                  className="opacity-75"
                  d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4z"
                />
              </svg>
            )}
          </button>
        );
      })}
    </nav>
  );
}

[slug]/layout.tsx — Stable Col 2 TSXServer

Col 2 must be rendered here permanently. Never conditionally render it. Children = Col 3 only.

src/app/(protected)/location/[slug]/layout.tsx 
import { notFound } from 'next/navigation';
import { getLocationsTree } from
  '@/modules/locations/actions/server-locations';
import { getLocationFunctions } from
  '@/modules/locations/actions/server-location-functions';
import { getNodeAndPathBySlug } from
  '@/modules/locations/lib/utils';
import { FunctionsNav } from
  '@/components/nav/FunctionsNav';

interface Props {
  children: React.ReactNode;
  params: Promise<{ slug: string; fn?: string }>;
}

export default async function LocationLayout({
  children,
  params,
}: Props) {
  const { slug, fn } = await params;

  const tree = await getLocationsTree();
  const found = getNodeAndPathBySlug(tree, slug);
  if (!found) notFound();

  const locationId = found.node.id;
  const locationName = found.node.name;
  const functions = await getLocationFunctions(locationId);

  return (
    <div className="flex h-full w-full overflow-hidden">

      {/* ── Col 2: Function nav — NEVER re-renders ─────── */}
      <aside className="flex w-52 flex-shrink-0 flex-col
        border-r border-gray-200 bg-white">

        {/* Location breadcrumb header */}
        <div className="border-b border-gray-100 px-4 py-3">
          <p className="text-[10px] font-medium uppercase
            tracking-wider text-gray-400">
            Location
          </p>
          <p className="mt-0.5 text-sm font-semibold
            text-gray-800 truncate">
            {locationName}
          </p>
        </div>

        {/* Function nav — client component with useTransition */}
        <div className="flex-1 overflow-y-auto">
          <FunctionsNav
            locationSlug={slug}
            activeFn={fn ?? null}
            functions={functions}
          />
        </div>
      </aside>

      {/* ── Col 3: Page content — only this changes ─────── */}
      <main className="relative flex-1 overflow-y-auto
        bg-gray-50">
        {children}
      </main>
    </div>
  );
}

Pattern: Wrap your shell content in <ContentArea isLoading={isChangingPlan}> to grey out Col 3 during mutations (e.g. applying a plan). Different from navigation loading — this is for client-side async actions within the page.
src/components/layout/ContentArea.tsx 
'use client';

interface ContentAreaProps {
  children: React.ReactNode;
  isLoading?: boolean;
  loadingMessage?: string;
}

/**
 * ContentArea — wraps Col 3 content with an optional grey overlay.
 *
 * Use when:
 *   - A mutation is in progress (applying plan, saving config)
 *   - You want to block interaction but keep content visible
 *
 * NOT for navigation loading — that's handled by [fn]/loading.tsx
 *
 * Usage in PlanShell:
 *   const { isLoading } = useChangePlan(locationId);
 *   return (
 *     
 *       
 *       
 *     
 *   );
 */
export function ContentArea({
  children,
  isLoading = false,
  loadingMessage,
}: ContentAreaProps) {
  return (
    <div className="relative h-full w-full">
      {children}

      {/* Grey overlay — only when isLoading is true */}
      {isLoading && (
        <div
          className="absolute inset-0 z-10 flex
            flex-col items-center justify-center
            bg-white/60 backdrop-blur-[1px]"
          aria-busy="true"
          aria-label={loadingMessage ?? 'Loading...'}
        >
          {/* Spinner */}
          <svg
            className="mb-3 h-8 w-8 animate-spin text-blue-500"
            viewBox="0 0 24 24"
            fill="none"
          >
            <circle
              cx="12" cy="12" r="10"
              stroke="currentColor"
              strokeWidth="4"
              className="opacity-25"
            />
            <path
              fill="currentColor"
              className="opacity-75"
              d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4z"
            />
          </svg>

          {loadingMessage && (
            <p className="text-xs font-medium text-gray-500">
              {loadingMessage}
            </p>
          )}
        </div>
      )}
    </div>
  );
}
