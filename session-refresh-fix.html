KEY FIX: "BE returned no token" diagnosed — logs RAW response shape, extracts token across possible field names (token/accessToken/AccessToken/...). Run once, check terminal, then trim fallbacks to the real field name.
import 'server-only';

// ─────────────────────────────────────────────────────────────
// "[refresh] BE returned no token" — DIAGNOSIS
//
// This message means:
//   ✓ refreshToken + userId cookies WERE present (passed the early check)
//   ✓ The GQL request to refreshAuth WENT THROUGH without throwing
//   ✗ data.refreshAuth.token was falsy
//
// Two likely causes:
//   A) Response field is named differently
//      (e.g. accessToken/AccessToken instead of token)
//   B) refreshAuth itself returned success:false / an error object
//      that we're not checking
//
// FIX: log the RAW response once, then extract token defensively
// across multiple possible field names. Once you see the real
// shape in your terminal, delete the defensive fallbacks and
// keep only the correct field name.
// ─────────────────────────────────────────────────────────────

import { GraphQLClient, ClientError } from 'graphql-request';
import { GQL_ENDPOINT }               from '@/lib/constants';
import {
  getRefreshToken,
  getUserId,
  setAuthCookies,
  clearAuthCookies,
} from '@/modules/auth/lib/cookies';
import { REFRESH_AUTH_MUTATION } from '../graphql/auth.mutations';

export async function refreshAccessToken(): Promise<string | null> {
  const [refreshToken, userId] = await Promise.all([
    getRefreshToken(),
    getUserId(),
  ]);

  if (!refreshToken || !userId) {
    console.warn('[refresh] Missing refreshToken or userId — cannot refresh', {
      hasRefreshToken: Boolean(refreshToken),
      hasUserId:       Boolean(userId),
    });
    return null;
  }

  try {
    const client = new GraphQLClient(GQL_ENDPOINT, {
      headers: {
        'authorization': refreshToken,
        'Content-Type':  'application/json',
        'x-tenant-id':   process.env.TENANT_ID ?? '',
      },
    });

    // ── DIAGNOSTIC: log raw response shape ───────────────────
    // Run this once, check terminal, then read the field name
    // BE actually returns and update the extraction below.
    const raw = await client.rawRequest<any>(REFRESH_AUTH_MUTATION, {
      userId,
      token: refreshToken,
    });

    console.log('[refresh] RAW response:', JSON.stringify(raw.data, null, 2));
    if (raw.errors?.length) {
      console.log('[refresh] GQL errors:', JSON.stringify(raw.errors, null, 2));
    }
    // ── END DIAGNOSTIC ────────────────────────────────────────

    const result = raw.data?.refreshAuth;

    if (!result) {
      console.warn('[refresh] BE returned no refreshAuth field at all');
      return null;
    }

    // ── Defensive extraction across possible field names ─────
    // Remove fallbacks once you confirm the real shape from the
    // [refresh] RAW response log above.
    const token =
      result.token ??
      result.accessToken ??
      result.AccessToken ??
      result.access_token ??
      null;

    const idToken =
      result.idToken ??
      result.IdToken ??
      result.id_token ??
      null;

    const newRefreshToken =
      result.refreshToken ??
      result.RefreshToken ??
      result.refresh_token ??
      refreshToken; // BE may not rotate refresh token — reuse existing
    // ───────────────────────────────────────────────────────────

    if (!token) {
      console.warn(
        '[refresh] BE returned no token — fields present:',
        Object.keys(result)
      );
      return null;
    }

    await setAuthCookies({
      token,
      idToken:      idToken ?? '',
      refreshToken: newRefreshToken,
      userId,
    } as any);

    console.log('[refresh] SUCCESS — new access token set');
    return token;

  } catch (error) {
    if (error instanceof ClientError) {
      const msg = (error.response?.errors?.[0]?.message ?? '').toLowerCase();
      console.error('[refresh] BE ClientError:', msg, {
        status: error.response?.status,
        errors: error.response?.errors,
      });

      // Refresh token itself is invalid/expired → force re-login
      if (
        msg.includes('expired') ||
        msg.includes('invalid')  ||
        msg.includes('unauthenticated') ||
        error.response?.status === 401
      ) {
        await clearAuthCookies();
      }
    } else {
      console.error('[refresh] Unexpected error:', error);
    }
    return null;
  }
}
SERVER src/app/api/auth/session/route.ts
FIX for Image 1: was returning raw 401 without trying refresh. Now attempts refreshAccessToken() first — only 401s if refresh token itself is invalid.
// src/app/api/auth/session/route.ts
//
// ─────────────────────────────────────────────────────────────
// FIX for Image 1: GET /api/auth/session → 401 immediately
//
// BEFORE: checked access token cookie only — expired → 401
//         (useSession hook saw 401, never tried refresh)
//
// AFTER:  no/expired access token → attempt refresh first
//         success → 200 (session still valid)
//         fail    → 401 with reason: "SESSION_EXPIRED"
//                   (refresh token also expired/invalid)
// ─────────────────────────────────────────────────────────────

import { NextResponse } from 'next/server';
import {
  getAccessToken,
  getSessionUser,
} from '@/modules/auth/lib/cookies';
import { refreshAccessToken } from '@/modules/auth/lib/refresh';

export async function GET() {
  let token = await getAccessToken();

  if (!token) {
    token = await refreshAccessToken();

    if (!token) {
      // Refresh token also expired/invalid — real session end
      return NextResponse.json(
        { authenticated: false, reason: 'SESSION_EXPIRED' },
        { status: 401 }
      );
    }
  }

  const user = await getSessionUser();
  return NextResponse.json({ authenticated: true, user });
}

'use client';

// ─────────────────────────────────────────────────────────────
// SESSION PROVIDER
//
// Wraps the app (inside QueryClientProvider). Provides:
//
//   1. useSessionStatus() — 'checking' | 'valid' | 'expired'
//
//   2. Global full-screen loader while session check/refresh
//      is in flight on initial mount (Image 2 scenario —
//      page reload with expired access token)
//
//   3. "Session expired" modal when refresh token itself is
//      invalid — explains what happened + button to re-login
//      (instead of silently redirecting, which feels broken)
//
//   4. Listens for a global 'session-expired' event — dispatched
//      by useAuthQuery.ts gqlFetch (Image 1 scenario — mid-action
//      401 that even the proxy's refresh couldn't recover from)
// ─────────────────────────────────────────────────────────────

import {
  createContext, useContext, useEffect, useState, useCallback,
} from 'react';
import { useRouter } from 'next/navigation';

type SessionStatus = 'checking' | 'valid' | 'expired';

const SessionContext = createContext<{ status: SessionStatus }>({
  status: 'checking',
});

export function useSessionStatus() {
  return useContext(SessionContext).status;
}

// Dispatch this from anywhere (e.g. useAuthQuery gqlFetch) when
// the proxy reports SESSION_EXPIRED — shows the modal immediately
// instead of an abrupt redirect.
export function notifySessionExpired() {
  window.dispatchEvent(new Event('session-expired'));
}

export function SessionProvider({ children }: { children: React.ReactNode }) {
  const router = useRouter();
  const [status, setStatus] = useState<SessionStatus>('checking');

  const checkSession = useCallback(async () => {
    try {
      const res = await fetch('/api/auth/session', {
        credentials: 'include',
        cache:       'no-store',
      });
      setStatus(res.ok ? 'valid' : 'expired');
    } catch {
      setStatus('expired');
    }
  }, []);

  // Initial check on mount — covers Image 2 (reload with expired token)
  useEffect(() => {
    checkSession();
  }, [checkSession]);

  // Listen for mid-session expiry (Image 1 — action-triggered)
  useEffect(() => {
    function handleExpired() {
      setStatus('expired');
    }
    window.addEventListener('session-expired', handleExpired);
    return () => window.removeEventListener('session-expired', handleExpired);
  }, []);

  function goToLogin() {
    router.push('/login?reason=expired');
  }

  return (
    <SessionContext.Provider value={{ status }}>
      {children}

      {/* ── Initial-load checking overlay ─────────────────────── */}
      {status === 'checking' && <SessionCheckingOverlay />}

      {/* ── Session expired modal ─────────────────────────────── */}
      {status === 'expired' && (
        <SessionExpiredModal onLogin={goToLogin} />
      )}
    </SessionContext.Provider>
  );
}

'use client';

// ─────────────────────────────────────────────────────────────
// CHANGE vs previous version:
//
//   onExpired callback now calls notifySessionExpired()
//   (dispatches 'session-expired' window event) INSTEAD OF
//   router.push('/login?reason=expired') directly.
//
//   SessionProvider listens for this event and shows
//   SessionExpiredModal — explains WHY before navigating,
//   instead of an abrupt silent redirect (Image 1 scenario).
//
// Everything else — gqlFetch via /api/graphql proxy,
// useAuthQuery/useAuthMutation signatures — UNCHANGED.
// ─────────────────────────────────────────────────────────────

import {
  useQuery,
  useMutation,
  useQueryClient,
  type UseQueryOptions,
} from '@tanstack/react-query';
import { notifySessionExpired } from '@/components/providers/SessionProvider';

async function gqlFetch<TData>(
  query:      string,
  variables?: object,
): Promise<TData> {

  const res = await fetch('/api/graphql', {
    method:      'POST',
    headers:     { 'Content-Type': 'application/json' },
    credentials: 'include',
    body:        JSON.stringify({ query, variables }),
    cache:       'no-store',
  });

  const json = await res.json() as {
    data?:   TData;
    errors?: { message: string }[];
  };

  if (
    res.status === 401 ||
    json.errors?.[0]?.message === 'SESSION_EXPIRED'
  ) {
    // Show "Session Expired" modal instead of silent redirect —
    // proxy already tried refreshing and it failed (refresh
    // token itself expired/invalid).
    notifySessionExpired();
    throw new Error('SESSION_EXPIRED');
  }

  if (json.errors?.length) {
    throw new Error(json.errors[0].message);
  }

  if (json.data === undefined || json.data === null) {
    throw new Error('No data returned from server');
  }

  return json.data;
}

export function useAuthQuery<
  TData,
  TSelected = TData,
>(
  queryKey:   readonly unknown[],
  query:      string,
  variables?: object,
  options?:   Omit<
    UseQueryOptions<TData, Error, TSelected>,
    'queryKey' | 'queryFn'
  >,
) {
  return useQuery<TData, Error, TSelected>({
    queryKey,
    queryFn: () => gqlFetch<TData>(query, variables),
    staleTime: 1000 * 60 * 5,
    retry:     1,
    ...options,
  });
}

export function useAuthMutation<
  TData,
  TVars extends object = object,
>(
  mutation: string,
  options?: {
    onSuccess?:      (data: TData) => void;
    onError?:        (error: Error) => void;
    invalidateKeys?: unknown[][];
  },
) {
  const queryClient = useQueryClient();

  return useMutation<TData, Error, TVars>({
    mutationFn: (variables: TVars) => gqlFetch<TData>(mutation, variables),

    onSuccess: (data) => {
      options?.invalidateKeys?.forEach(key =>
        queryClient.invalidateQueries({ queryKey: key })
      );
      options?.onSuccess?.(data);
    },

    onError: options?.onError,
  });
}
