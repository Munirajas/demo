The core question answered
QUESTION: Should getLocationDescendents be server or client side?

ANSWER: BOTH — for different jobs.

SERVER — initial load (fast, no spinner):
  getLocationsTree() in server-locations.ts
  Called in [fn]/layout.tsx and page.tsx
  React cache() = only ONE network request per render cycle
  Result: sidebar shows instantly, breadcrumbs built server-side

CLIENT — after mutations (reactive):
  useLocationsTree(initialData) in useLocations.ts
  Called in DashboardShell.tsx
  Receives server data as initialData → zero loading state
  After create/delete → invalidate → auto-refetch → sidebar updates

THE HANDOFF PATTERN:
  Server fetches → passes as prop (tree={tree})
  Client receives as initialData → useLocationsTree(tree)
  First render: server data (instant)
  After mutations: client refetch (reactive)

When to use server vs client
USE SERVER (getLocationsTree in server-locations.ts):
  ✓ Initial page load (zero loading spinner)
  ✓ Building breadcrumbs (needs ancestor path)
  ✓ Redirecting to first location
  ✓ Validating locationId exists (notFound() if not)

USE CLIENT (useLocationsTree in useLocations.ts):
  ✓ After user creates a location → sidebar updates
  ✓ After user deletes a location → sidebar updates
  ✓ Keeping sidebar in sync without page reload
  ✓ Any reactive UI that changes based on tree mutations

NEVER:
  ✗ useEffect(() => fetch('/api/locations'), [])
  ✗ Store locations in Zustand
  ✗ Call getLocationsTree() in a client component

// src/app/(dashboard)/locations/[locationId]/[fn]/layout.tsx
// SERVER COMPONENT
//
// ─────────────────────────────────────────────────────────────
// LOCATION + FUNCTION LAYOUT — the main app shell
//
// This is the key layout that:
//   1. Fetches locations tree (server-side)
//      React cache() → if already called in page.tsx,
//      this is FREE — zero extra network call
//   2. Builds breadcrumbs from tree + URL params
//   3. Builds function nav for this location
//   4. Passes initialData to client DashboardShell
//      (client shell stays reactive after mutations)
//
// WHY server-side (not client useAuthQuery):
//   • Data needed for FIRST RENDER — no loading spinner
//   • Breadcrumbs built server-side from tree path
//   • Avoids client waterfall (no fetch-then-render)
//   • Server can access cookies directly
//
// REACT cache() DEDUPLICATION:
//   (dashboard)/page.tsx calls getLocationsTree()  → fires
//   This layout also calls getLocationsTree()      → CACHED
//   Result: exactly ONE network request total
// ─────────────────────────────────────────────────────────────

import { notFound }            from 'next/navigation';
import { DashboardShell }      from '@/components/layout/DashboardShell';
import { getLocationsTree }    from '@/modules/locations/graphql/server-locations';
import {
  findById,
  getAncestorPath,
  FUNCTION_LABELS,
  FUNCTION_SLUGS,
} from '@/modules/locations/types/location.types';
import type { BreadcrumbItem } from '@/modules/locations/types/location.types';

interface Props {
  children: React.ReactNode;
  params:   Promise<{ locationId: string; fn: string }>;
}

export default async function LocationFnLayout({ children, params }: Props) {
  const { locationId, fn } = await params;

  // ── Fetch tree (server-side, React cache deduplicates) ──────
  // This is the ONLY data fetch needed for the entire shell.
  // No waterfall. No multiple requests.
  const tree = await getLocationsTree();

  // Validate the locationId exists in the tree
  const location = findById(tree, locationId);
  if (!location) notFound();

  // ── Build breadcrumbs from ancestor path ────────────────────
  // Works for any depth — no extra params needed in URL
  const ancestorPath = getAncestorPath(tree, locationId) ?? [location];
  const fnLabel      = FUNCTION_LABELS[fn as keyof typeof FUNCTION_LABELS] ?? fn;

  const breadcrumbs: BreadcrumbItem[] = [
    ...ancestorPath.map(node => ({
      label: node.label,
      // Client node: not navigable (no href)
      // Location node: navigate keeping current function slug
      href: node.type === 'client'
        ? undefined
        : `/locations/${node.id}/${fn}`,
    })),
    // Current function = last item, no href (current page)
    { label: fnLabel },
  ];

  // ── Function nav items ──────────────────────────────────────
  // All functions enabled by default.
  // TODO: Filter by location config toggles from BE
  const fnItems = FUNCTION_SLUGS.map(slug => ({
    label: FUNCTION_LABELS[slug],
    path:  `/locations/${locationId}/${slug}`,
  }));

  return (
    <DashboardShell
      // Server-fetched tree → passed as initialData to client
      // Client DashboardShell hydrates TanStack Query cache with this
      // Result: sidebar renders instantly, stays reactive after mutations
      tree={tree}
      locationId={locationId}
      fnItems={fnItems}
      breadcrumbs={breadcrumbs}
    >
      {children}
    </DashboardShell>
  );
}

SERVER src/modules/locations/graphql/server-locations.ts
React cache() wraps getLocationsTree() — deduplicates across layouts
import 'server-only';
import { cache }             from 'react';
import { serverGqlFetch }    from '@/lib/graphql/server-gql-fetcher';
import {
  GET_LOCATION_DESCENDENTS,
} from './locations.graphql';
import {
  buildTreeFromFlat,
  getFirstLocation,
} from '../types/location.types';
import type {
  LocationNode,
  GetLocationDescendentsResponse,
} from '../types/location.types';

// ─────────────────────────────────────────────────────────────
// SERVER LOCATION FETCHERS
//
// React cache() deduplicates — if layout AND page both call
// getLocationsTree(), only ONE request fires per render cycle.
//
// Uses getLocationDescendents mutation with type: "client"
// Input locationId: 1 means "get the client node and all descendants"
// ─────────────────────────────────────────────────────────────

// NOTE: You need to know the client's root locationId.
// This could come from the user's JWT claims or a config.
// For now using 1 as default — update when BE provides this.
const CLIENT_LOCATION_ID = Number(
  process.env.CLIENT_ROOT_LOCATION_ID ?? '1'
);

export const getLocationsTree = cache(async (): Promise<LocationNode[]> => {
  try {
    const response = await serverGqlFetch<GetLocationDescendentsResponse>(
      GET_LOCATION_DESCENDENTS,
      {
        type:       'client',
        locationId: CLIENT_LOCATION_ID,
      }
    );

    if (!response.getLocationDescendents.success) {
      console.error('[locations] API error:',
        response.getLocationDescendents.message);
      return [];
    }

    // Build nested tree from flat array
    return buildTreeFromFlat(response.getLocationDescendents.data);

  } catch (error) {
    console.error('[getLocationsTree]', error);
    return [];
  }
});

export async function resolveFirstLocation(): Promise<string | null> {
  const tree = await getLocationsTree();
  return getFirstLocation(tree)?.id ?? null;
}

'use client';

// ─────────────────────────────────────────────────────────────
// useLocations — CLIENT-SIDE location hooks
//
// PURPOSE: Reactive updates AFTER user mutations
//   (create location, delete location, etc.)
//
// PATTERN:
//   Server fetches initial tree → passes as initialData
//   Client hook uses that data instantly (zero loading state)
//   After mutation → invalidate → hook refetches silently
//   Sidebar updates without page reload
//
// IMPORTANT: Do NOT use these for initial page load.
//   Use server-locations.ts (getLocationsTree) for that.
// ─────────────────────────────────────────────────────────────

import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { useRouter }                              from 'next/navigation';
import { useAuthQuery, useAuthMutation }          from '@/modules/auth/hooks/useAuthQuery';
import { GET_LOCATION_DESCENDENTS }               from '../graphql/locations.graphql';
import {
  buildTreeFromFlat,
} from '../types/location.types';
import type {
  LocationNode,
  GetLocationDescendentsResponse,
} from '../types/location.types';

// ── Query key factory ─────────────────────────────────────────
export const locationKeys = {
  all:  ()         => ['locations']               as const,
  tree: ()         => ['locations', 'tree']        as const,
  node: (id: string) => ['locations', 'node', id] as const,
};

// ── CREATE / DELETE mutations (will be added when BE ready) ──

const CREATE_LOCATION_MUTATION = `
  mutation CreateLocation($label: String!, $parentId: String!) {
    createLocation(label: $label, parentId: $parentId) {
      Location_Id
      Location_Name
      Parent_Location_Id
      Level
    }
  }
`;

const DELETE_LOCATION_MUTATION = `
  mutation DeleteLocation($locationId: String!) {
    deleteLocation(locationId: $locationId) {
      success
    }
  }
`;

// ── useLocationsTree (CLIENT) ─────────────────────────────────
// Used to keep the sidebar REACTIVE after mutations.
// Receives initialData from server — renders immediately.
//
// Usage: <LocationSidebar initialTree={serverTree} />
// The client hook takes over and keeps it fresh.
export function useLocationsTree(initialData?: LocationNode[]) {
  const clientLocationId = Number(
    process.env.NEXT_PUBLIC_CLIENT_LOCATION_ID ?? '1'
  );

  return useAuthQuery<GetLocationDescendentsResponse>(
    locationKeys.tree(),
    GET_LOCATION_DESCENDENTS,
    { type: 'client', locationId: clientLocationId },
    {
      // Pre-populate with server-fetched data
      // This means ZERO loading state on first render
      initialData: initialData
        ? {
            getLocationDescendents: {
              success: true,
              message: 'hydrated from server',
              data:    flattenTree(initialData),
            },
          }
        : undefined,
      // Keep fresh: refetch every 5 minutes
      staleTime: 1000 * 60 * 5,
      // After mutation: auto-refetch to show new locations
      select: (data: GetLocationDescendentsResponse) =>
        buildTreeFromFlat(data.getLocationDescendents.data),
    } as any
  );
}

// ── useCreateLocation ─────────────────────────────────────────
export function useCreateLocation() {
  const queryClient = useQueryClient();
  const router      = useRouter();

  return useAuthMutation<
    { createLocation: { Location_Id: number } },
    { label: string; parentId: string }
  >(
    CREATE_LOCATION_MUTATION,
    {
      onSuccess: (data) => {
        // Invalidate tree → sidebar refreshes with new location
        queryClient.invalidateQueries({ queryKey: locationKeys.tree() });
        // Navigate to new location
        router.push(`/locations/${data.createLocation.Location_Id}/config`);
      },
    }
  );
}

// ── useDeleteLocation ─────────────────────────────────────────
export function useDeleteLocation() {
  const queryClient = useQueryClient();
  const router      = useRouter();

  return useAuthMutation<
    { deleteLocation: { success: boolean } },
    { locationId: string; parentId?: string | null }
  >(
    DELETE_LOCATION_MUTATION,
    {
      onSuccess: (_, vars) => {
        queryClient.invalidateQueries({ queryKey: locationKeys.tree() });
        // Navigate to parent or dashboard
        if (vars.parentId) {
          router.push(`/locations/${vars.parentId}/config`);
        } else {
          router.push('/dashboard');
        }
      },
    }
  );
}

// ── Helper: flatten nested tree back to raw items ─────────────
// Needed to convert server LocationNode[] back to raw format
// for TanStack Query initialData
function flattenTree(
  nodes: LocationNode[],
  result: Array<{
    Location_Id: number;
    Location_Name: string;
    Parent_Location_Id: number | null;
    Level: number;
  }> = []
) {
  for (const node of nodes) {
    result.push({
      Location_Id:        Number(node.id),
      Location_Name:      node.label,
      Parent_Location_Id: node.parentId ? Number(node.parentId) : null,
      Level:              node.type === 'client' ? 0 : 1,
    });
    if (node.children.length) flattenTree(node.children, result);
  }
  return result;
}

}

// Filter tree nodes by search term
function filterTree(nodes: LocationNode[], term: string): LocationNode[] {
  return nodes.reduce<LocationNode[]>((acc, node) => {
    const matches     = node.label.toLowerCase().includes(term);
    const filteredKids = node.children.length
      ? filterTree(node.children, term)
      : [];
    if (matches || filteredKids.length) {
      acc.push({ ...node, children: filteredKids });
    }
    return acc;
  }, []);
}
