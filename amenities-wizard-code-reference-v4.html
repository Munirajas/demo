Amenity wizard — RHF + Zod on both steps (v4)

Extends last message's React Hook Form + Zod pattern (previously only on the scheduling step) to step 1 as well, fixes a real bug in the recurring toggle, and wires in the date/time defaults you asked for. Both steps now expose the same validate() / getValues() / reset() handle shape via forwardRef, so the parent forms treat them identically regardless of which step they're driving.

  • Fixed the recurring toggle bug: turning it on now sets frequency: 'WEEKLY' directly instead of tracking a separate enableRecurring boolean that could desync from the real frequency value (the bug: the old version left frequency unset/invalid when toggled on).
  • Step 1 is now RHF + Zod — same pattern as the scheduler, exposed through a GeneralInfoFieldsHandle.
  • Sensible defaults wired in — start date/time = now, end date = today, end time = 23:59, recurrence end date defaults to one year out. All are just defaults
  • Simplified data flow: since each step now owns its own RHF instance, the shell no longer carries a GeneralInfoValues object — it only tracks the plain amenityTypeId needed to route to the right form component.
  • All validation messages render under their field via Zod issues surfaced through RHF's formState.errors.
"End time is tonight 12am" was implemented as 23:59 the same day. If you meant the literal stroke of midnight starting the next calendar day, that's a one-line change in recurrence-defaults.ts (roll endDate forward one day, set endTime to '00:00') — flagging it since it changes both fields together.

Step 2 — Scheduling
lib/recurrence-defaults.ts new

Builds the default form state: today's date, current time, end time 23:59, and a recurrence end date one year out -- used only if the user never explicitly picks an end condition themselves.

import type { RecurrenceRuleValues } from './recurrence-types';

// Local-date formatting on purpose -- Date.toISOString() is UTC-based and
// can silently roll to the wrong calendar day near midnight in timezones
// ahead of/behind UTC. This keeps "today" meaning the user's actual today.
function toDateInputValue(d: Date): string {
  const y = d.getFullYear();
  const m = String(d.getMonth() + 1).padStart(2, '0');
  const day = String(d.getDate()).padStart(2, '0');
  return `${y}-${m}-${day}`;
}

function toTimeInputValue(d: Date): string {
  const h = String(d.getHours()).padStart(2, '0');
  const m = String(d.getMinutes()).padStart(2, '0');
  return `${h}:${m}`;
}

export function buildDefaultRecurrenceRule(): RecurrenceRuleValues {
  const now = new Date();
  const today = toDateInputValue(now);

  const oneYearOut = new Date(now);
  oneYearOut.setFullYear(oneYearOut.getFullYear() + 1);

  return {
    startDate: today,
    startTime: toTimeInputValue(now),
    endDate: today,
    endTime: '23:59', // "tonight" -- see the flag note above if you meant next-day midnight instead
    frequency: 'NONE',
    interval: 1,
    daysOfWeek: ['MON', 'TUE', 'WED', 'THU', 'FRI'],
    monthlyMode: 'DAY_OF_MONTH',
    monthlyDayOfMonth: now.getDate(),
    monthlySetPosition: 1,
    monthlyWeekday: 'MON',
    yearlyMode: 'ON_DATE',
    yearlyMonth: now.getMonth() + 1,
    yearlyDayOfMonth: now.getDate(),
    yearlySetPosition: 1,
    yearlyWeekday: 'MON',
    recurrenceEndMode: 'ON_DATE',
    recurrenceEndDate: toDateInputValue(oneYearOut),
    occurrenceCount: undefined,
  };
}
lib/recurrence-schema.ts new

Zod schema for the full recurrence rule. Cross-field checks (end-after-start, required day selection for weekly, required end condition once recurring is on) live in superRefine so they attach to the specific field that's actually wrong.

import { z } from 'zod';

const weekDaySchema = z.enum(['SUN', 'MON', 'TUE', 'WED', 'THU', 'FRI', 'SAT']);
const setPositionSchema = z.union([
  z.literal(1), z.literal(2), z.literal(3), z.literal(4), z.literal(-1),
]);

export const recurrenceRuleSchema = z
  .object({
    startDate: z.string().min(1, 'Start date is required'),
    startTime: z.string().min(1, 'Start time is required'),
    endDate: z.string().min(1, 'End date is required'),
    endTime: z.string().min(1, 'End time is required'),
    frequency: z.enum(['NONE', 'DAILY', 'WEEKLY', 'MONTHLY', 'YEARLY']),
    interval: z.number({ invalid_type_error: 'Enter a number' }).min(1, 'Must be at least 1'),
    daysOfWeek: z.array(weekDaySchema),
    monthlyMode: z.enum(['DAY_OF_MONTH', 'WEEKDAY_POSITION']),
    monthlyDayOfMonth: z.number().min(1, 'Must be 1-31').max(31, 'Must be 1-31'),
    monthlySetPosition: setPositionSchema,
    monthlyWeekday: weekDaySchema,
    yearlyMode: z.enum(['ON_DATE', 'WEEKDAY_POSITION']),
    yearlyMonth: z.number().min(1).max(12),
    yearlyDayOfMonth: z.number().min(1, 'Must be 1-31').max(31, 'Must be 1-31'),
    yearlySetPosition: setPositionSchema,
    yearlyWeekday: weekDaySchema,
    recurrenceEndMode: z.enum(['ON_DATE', 'AFTER_OCCURRENCES', 'NEVER']),
    recurrenceEndDate: z.string().optional(),
    occurrenceCount: z.number().optional(),
  })
  .superRefine((val, ctx) => {
    if (val.endDate < val.startDate) {
      ctx.addIssue({ code: z.ZodIssueCode.custom, message: 'Cannot be before the start date', path: ['endDate'] });
    }
    if (val.endDate === val.startDate && val.endTime <= val.startTime) {
      ctx.addIssue({ code: z.ZodIssueCode.custom, message: 'Must be after the start time', path: ['endTime'] });
    }

    if (val.frequency === 'WEEKLY' && val.daysOfWeek.length === 0) {
      ctx.addIssue({ code: z.ZodIssueCode.custom, message: 'Select at least one day', path: ['daysOfWeek'] });
    }

    if (val.frequency !== 'NONE') {
      if (val.recurrenceEndMode === 'ON_DATE') {
        if (!val.recurrenceEndDate) {
          ctx.addIssue({ code: z.ZodIssueCode.custom, message: 'Choose an end date', path: ['recurrenceEndDate'] });
        } else if (val.recurrenceEndDate < val.startDate) {
          ctx.addIssue({ code: z.ZodIssueCode.custom, message: 'Must be on or after the start date', path: ['recurrenceEndDate'] });
        }
      }
      if (val.recurrenceEndMode === 'AFTER_OCCURRENCES' && (!val.occurrenceCount || val.occurrenceCount < 1)) {
        ctx.addIssue({ code: z.ZodIssueCode.custom, message: 'Enter a number of occurrences', path: ['occurrenceCount'] });
      }
    }
  });
components/schedule/RecurrenceRuleBuilder.tsx changed — toggle bug fixed

Same RHF + Zod shell as before, with the recurring toggle rewritten: it now derives isRecurring straight from frequency instead of a parallel boolean, and turning it on sets frequency: 'WEEKLY' directly (a valid, immediately usable state) instead of leaving the dropdown unselected.

'use client';

import { forwardRef, useImperativeHandle } from 'react';
import { useForm, Controller } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { recurrenceRuleSchema } from '../../lib/recurrence-schema';
import { buildDefaultRecurrenceRule } from '../../lib/recurrence-defaults';
import type { RecurrenceRuleValues, WeekDay, SetPosition } from '../../lib/recurrence-types';

const ALL_DAYS: { key: WeekDay; label: string }[] = [
  { key: 'SUN', label: 'Sun' }, { key: 'MON', label: 'Mon' }, { key: 'TUE', label: 'Tue' },
  { key: 'WED', label: 'Wed' }, { key: 'THU', label: 'Thu' }, { key: 'FRI', label: 'Fri' },
  { key: 'SAT', label: 'Sat' },
];
const SET_POSITIONS: { value: SetPosition; label: string }[] = [
  { value: 1, label: 'First' }, { value: 2, label: 'Second' }, { value: 3, label: 'Third' },
  { value: 4, label: 'Fourth' }, { value: -1, label: 'Last' },
];
const MONTHS = [
  'January', 'February', 'March', 'April', 'May', 'June',
  'July', 'August', 'September', 'October', 'November', 'December',
];

export interface RecurrenceRuleBuilderHandle {
  validate: () => Promise<boolean>;
  getValues: () => RecurrenceRuleValues;
  reset: (values: RecurrenceRuleValues) => void;
}

interface Props {
  defaultValues?: Partial<RecurrenceRuleValues>;
}

function FieldError({ message }: { message?: string }) {
  if (!message) return null;
  return <p className="text-xs text-red-600 mt-1">{message}</p>;
}

export const RecurrenceRuleBuilder = forwardRef<RecurrenceRuleBuilderHandle, Props>(
  function RecurrenceRuleBuilder({ defaultValues }, ref) {
    const {
      register, control, watch, setValue, trigger, getValues, reset,
      formState: { errors },
    } = useForm<RecurrenceRuleValues>({
      resolver: zodResolver(recurrenceRuleSchema),
      defaultValues: { ...buildDefaultRecurrenceRule(), ...defaultValues },
      mode: 'onBlur',
    });

    useImperativeHandle(ref, () => ({ validate: () => trigger(), getValues, reset }));

    const frequency = watch('frequency');
    const isRecurring = frequency !== 'NONE';
    const daysOfWeek = watch('daysOfWeek');
    const monthlyMode = watch('monthlyMode');
    const yearlyMode = watch('yearlyMode');
    const recurrenceEndMode = watch('recurrenceEndMode');

    function toggleDay(day: WeekDay) {
      const has = daysOfWeek.includes(day);
      setValue('daysOfWeek', has ? daysOfWeek.filter((d) => d !== day) : [...daysOfWeek, day], { shouldValidate: true });
    }

    // Fix: previously this only flipped a separate `enableRecurring` boolean,
    // leaving `frequency` unset/invalid when turned on. Now it directly
    // drives the field the rest of the form (and the engine) actually reads.
    function handleRecurringToggle(checked: boolean) {
      setValue('frequency', checked ? 'WEEKLY' : 'NONE', { shouldValidate: true });
    }

    return (
      <div className="flex flex-col gap-4">
        <div className="grid grid-cols-2 gap-3">
          <div className="flex flex-col gap-1">
            <label className="text-sm font-medium text-gray-700">Start</label>
            <div className="flex gap-2">
              <input type="date" className="w-full rounded-md border border-gray-300 px-3 py-2 text-sm" {...register('startDate')} />
              <input type="time" className="w-32 rounded-md border border-gray-300 px-3 py-2 text-sm" {...register('startTime')} />
            </div>
            <FieldError message={errors.startDate?.message ?? errors.startTime?.message} />
          </div>
          <div className="flex flex-col gap-1">
            <label className="text-sm font-medium text-gray-700">End</label>
            <div className="flex gap-2">
              <input type="date" className="w-full rounded-md border border-gray-300 px-3 py-2 text-sm" {...register('endDate')} />
              <input type="time" className="w-32 rounded-md border border-gray-300 px-3 py-2 text-sm" {...register('endTime')} />
            </div>
            <FieldError message={errors.endDate?.message ?? errors.endTime?.message} />
          </div>
        </div>
        <p className="text-xs text-gray-500 -mt-2">
          The gap between start and end is reused as the duration for every generated occurrence.
        </p>

        <div className="flex items-center justify-between rounded-lg border border-gray-200 bg-gray-50 px-4 py-3">
          <span className="text-sm font-medium text-gray-700">Recurring</span>
          <Controller
            name="frequency"
            control={control}
            render={({ field }) => (
              <button
                type="button"
                role="switch"
                aria-checked={field.value !== 'NONE'}
                onClick={() => handleRecurringToggle(field.value === 'NONE')}
                className={`relative h-6 w-11 rounded-full transition-colors ${
                  field.value !== 'NONE' ? 'bg-blue-600' : 'bg-gray-300'
                }`}
              >
                <span
                  className={`absolute top-0.5 h-5 w-5 rounded-full bg-white transition-transform ${
                    field.value !== 'NONE' ? 'translate-x-5' : 'translate-x-0.5'
                  }`}
                />
              </button>
            )}
          />
        </div>

        {isRecurring && (
          <>
            <div className="flex flex-col gap-1">
              <label className="text-sm font-medium text-gray-700">Frequency</label>
              <select className="w-full rounded-md border border-gray-300 px-3 py-2 text-sm" {...register('frequency')}>
                <option value="DAILY">Daily</option>
                <option value="WEEKLY">Weekly</option>
                <option value="MONTHLY">Monthly</option>
                <option value="YEARLY">Yearly</option>
              </select>
            </div>

            {frequency === 'DAILY' && (
              <div className="flex items-center gap-2">
                <span className="text-sm text-gray-700">Recur every</span>
                <input type="number" min={1} className="w-20 rounded-md border border-gray-300 px-3 py-2 text-sm" {...register('interval', { valueAsNumber: true })} />
                <span className="text-sm text-gray-600">day(s)</span>
                <FieldError message={errors.interval?.message} />
              </div>
            )}

            {frequency === 'WEEKLY' && (
              <>
                <div className="flex items-center gap-2">
                  <span className="text-sm text-gray-700">Recur every</span>
                  <input type="number" min={1} className="w-20 rounded-md border border-gray-300 px-3 py-2 text-sm" {...register('interval', { valueAsNumber: true })} />
                  <span className="text-sm text-gray-600">week(s)</span>

'use client';

import { useEffect, useRef } from 'react';
import { useInsertAmenity } from '../../hooks/useInsertAmenity';
import { useUpdateAmenity } from '../../hooks/useUpdateAmenity';
import { useAmenityGeneralInfo } from '../../hooks/useAmenityGeneralInfo';
import { GeneralInfoFields, type GeneralInfoFieldsHandle } from '../GeneralInfoFields';
import type { AmenityTypeFormProps } from './types';

export function GeneralAdmissionForm({
  amenityTypeId,
  isEditing,
  amenityId,
  locationFunctionId,
  locationSlug,
  onClose,
}: AmenityTypeFormProps) {
  const formRef = useRef<GeneralInfoFieldsHandle>(null);
  const hasHydrated = useRef(false);

  const {
    data: fetchedGeneralInfo,
    isLoading,
    isError,
    error: fetchError,
  } = useAmenityGeneralInfo(isEditing ? amenityId ?? null : null);

  useEffect(() => {
    if (fetchedGeneralInfo && !hasHydrated.current) {
      formRef.current?.reset({
        amenityName: fetchedGeneralInfo.amenityName,
        amenityDescription: fetchedGeneralInfo.amenityDescription,
      });
      hasHydrated.current = true;
    }
  }, [fetchedGeneralInfo]);

  const insert = useInsertAmenity(locationFunctionId, locationSlug);
  const update = useUpdateAmenity(locationFunctionId, locationSlug);
  const isSaving = insert.isPending || update.isPending;
  const isLoadingInitial = isEditing && isLoading && !hasHydrated.current;

    if (!isValid) return;

    const payload = {
      amenityTypeId,
      amenityName: values.amenityName,
      amenityDescription: values.amenityDescription ?? '',
    };

    if (isEditing && amenityId) {
      update.mutate({ amenityId, locationFunctionId, ...payload }, { onSuccess: onClose });
    } else {
      insert.mutate({ locationFunctionId, ...payload }, { onSuccess: onClose });
    }
  }

  if (isLoadingInitial) {
    return (
      <div className="flex flex-col gap-3 animate-pulse">
        <div className="h-9 bg-gray-100 rounded-md" />
        <div className="h-24 bg-gray-100 rounded-md" />
      </div>
    );
  }

  if (isEditing && isError) {
    return <p className="text-sm text-red-600">Couldn't load this amenity. {fetchError?.message}</p>;
  }

  return (
    <div className="flex flex-col gap-4">
      <GeneralInfoFields ref={formRef} />
      <div className="flex justify-end gap-2 pt-3 border-t border-gray-100 mt-1">
        <button type="button" onClick={onClose} className="px-4 py-2 text-sm rounded-md border border-gray-300 text-gray-700">Cancel</button>

        </button>

    </div>
  );

components/forms/ScheduledAdmissionForm.tsx changed

Two refs now -- one per step, each independently validated and hydrated. Next validates step 1's ref before advancing; Save validates step 2's ref, then reads both refs' values to assemble the final payload.

'use client';

import { useEffect, useRef, useState } from 'react';
import { useInsertAmenity } from '../../hooks/useInsertAmenity';
import { useUpdateAmenity } from '../../hooks/useUpdateAmenity';
import { useAmenityGeneralInfo } from '../../hooks/useAmenityGeneralInfo';
import { useAmenitySchedule } from '../../hooks/useAmenitySchedule';
import { GeneralInfoFields, type GeneralInfoFieldsHandle } from '../GeneralInfoFields';
import { RecurrenceRuleBuilder, type RecurrenceRuleBuilderHandle } from '../schedule/RecurrenceRuleBuilder';
import { StepIndicator, type StepIndicatorStep } from '../StepIndicator';
import { buildScheduleSavePayload } from '../../lib/recurrence-engine';
import type { AmenityTypeFormProps } from './types';

type Step = 'general' | 'scheduler';

export function ScheduledAdmissionForm({
  amenityTypeId,
  isEditing,
  amenityId,
  locationFunctionId,
  locationSlug,
  onClose,
}: AmenityTypeFormProps) {
  const [step, setStep] = useState<Step>('general');

  const generalFormRef = useRef<GeneralInfoFieldsHandle>(null);
  const scheduleFormRef = useRef<RecurrenceRuleBuilderHandle>(null);
  const hasHydratedGeneral = useRef(false);
  const hasHydratedSchedule = useRef(false);

  const {
    data: fetchedGeneralInfo,
    isLoading: isLoadingGeneral,
    isError: isGeneralError,
    error: generalError,
  } = useAmenityGeneralInfo(isEditing ? amenityId ?? null : null);

  useEffect(() => {
    if (fetchedGeneralInfo && !hasHydratedGeneral.current) {
      generalFormRef.current?.reset({
        amenityName: fetchedGeneralInfo.amenityName,
        amenityDescription: fetchedGeneralInfo.amenityDescription,
      });
      hasHydratedGeneral.current = true;
    }
  }, [fetchedGeneralInfo]);

  // Only fires once the user actually reaches step 2.
  const {
    data: fetchedSchedule,
    isLoading: isLoadingSchedule,
    isError: isScheduleError,
    error: scheduleError,
  } = useAmenitySchedule(isEditing ? amenityId ?? null : null, step === 'scheduler');

  useEffect(() => {
    if (fetchedSchedule?.rule && !hasHydratedSchedule.current) {
      scheduleFormRef.current?.reset(fetchedSchedule.rule);
      hasHydratedSchedule.current = true;
    }
  }, [fetchedSchedule]);

  const insert = useInsertAmenity(locationFunctionId, locationSlug);
  const update = useUpdateAmenity(locationFunctionId, locationSlug);
  const isSaving = insert.isPending || update.isPending;

  const steps: StepIndicatorStep[] = [
    { id: 'general', label: 'General info', status: step === 'general' ? 'active' : 'done' },
    { id: 'scheduler', label: 'Scheduling', status: step === 'scheduler' ? 'active' : 'pending' },
  ];

  async function handleNext() {
    const isValid = await generalFormRef.current?.validate();
    if (!isValid) return;
    setStep('scheduler');
  }

  async function handleSave() {
    const isValid = await scheduleFormRef.current?.validate();
    if (!isValid) return;

    const generalValues = generalFormRef.current!.getValues();
    const recurrenceRule = scheduleFormRef.current!.getValues();

    const payload = {
      amenityTypeId,
      amenityName: generalValues.amenityName,
      amenityDescription: generalValues.amenityDescription ?? '',
      schedule: buildScheduleSavePayload(recurrenceRule),
    };

    if (isEditing && amenityId) {
      update.mutate({ amenityId, locationFunctionId, ...payload }, { onSuccess: onClose });
    } else {
      insert.mutate({ locationFunctionId, ...payload }, { onSuccess: onClose });
    }
  }

  const isLoadingInitialGeneral = isEditing && isLoadingGeneral && !hasHydratedGeneral.current;
  const isLoadingInitialSchedule = isEditing && isLoadingSchedule && !hasHydratedSchedule.current;

  return (
    <div className="flex flex-col gap-4">
      <StepIndicator steps={steps} onStepClick={(i) => setStep(i === 0 ? 'general' : 'scheduler')} />

      {step === 'general' ? (
        isLoadingInitialGeneral ? (
          <div className="flex flex-col gap-3 animate-pulse">
            <div className="h-9 bg-gray-100 rounded-md" />
            <div className="h-24 bg-gray-100 rounded-md" />
          </div>
        ) : isGeneralError ? (
          <p className="text-sm text-red-600">Couldn't load general info. {generalError?.message}</p>
        ) : (
          <>
            <GeneralInfoFields ref={generalFormRef} />
            <div className="flex justify-end gap-2 pt-3 border-t border-gray-100 mt-1">
              <button type="button" onClick={onClose} className="px-4 py-2 text-sm rounded-md border border-gray-300 text-gray-700">Cancel</button>
              <button type="button" onClick={handleNext} className="px-4 py-2 text-sm rounded-md bg-blue-600 text-white">Next</button>
            </div>
          </>
        )
      ) : isLoadingInitialSchedule ? (
        <div className="flex flex-col gap-3 animate-pulse">
          <div className="h-9 bg-gray-100 rounded-md" />
          <div className="h-9 bg-gray-100 rounded-md" />
          <div className="h-32 bg-gray-100 rounded-md" />
        </div>
      ) : isScheduleError ? (
        <p className="text-sm text-red-600">Couldn't load the schedule. {scheduleError?.message}</p>
      ) : (
        <>
          <RecurrenceRuleBuilder ref={scheduleFormRef} />
          <div className="flex justify-between gap-2 pt-3 border-t border-gray-100 mt-1">
            <button type="button" onClick={() => setStep('general')} className="px-4 py-2 text-sm rounded-md border border-gray-300 text-gray-700">Back</button>
            <div className="flex gap-2">
              <button type="button" onClick={onClose} className="px-4 py-2 text-sm rounded-md border border-gray-300 text-gray-700">Cancel</button>
              <button type="button" onClick={handleSave} disabled={isSaving} className="px-4 py-2 text-sm rounded-md bg-blue-600 text-white disabled:opacity-60">
                {isSaving ? 'Saving...' : 'Save'}
              </button>
            </div>
          </div>
        </>
      )}
    </div>
  );
}

import { AmenityType } from '@/types/amenities';
import { AMENITY_FORM_REGISTRY } from './forms/amenity-form-registry';
import {
  GENERAL_ADMISSION_KEY,
  SCHEDULED_ADMISSION_KEY,
  resolveWizardKind,
} from '../types/wizard-types';

interface EditingAmenitySummary {
  amenityId: number;
  amenityTypeId: number; // already known from the table row -- no extra fetch needed to route
}

interface Props {
  isOpen: boolean;
  onClose: () => void;
  locationFunctionId: number;
  locationSlug: string;
  amenityTypes: AmenityType[];
  editingAmenity?: EditingAmenitySummary | null;
}

export function AmenityFormDrawer({
  isOpen, onClose, locationFunctionId, locationSlug, amenityTypes, editingAmenity = null,
}: Props) {
  const isEditing = editingAmenity !== null;
  const [amenityTypeId, setAmenityTypeId] = useState<number | ''>('');

  const originalWizardKind = editingAmenity
    ? resolveWizardKind(editingAmenity.amenityTypeId, amenityTypes)
    : null;

  useEffect(() => {
    if (!isOpen) return;
    setAmenityTypeId(editingAmenity ? editingAmenity.amenityTypeId : '');
  }, [isOpen, editingAmenity]);

  const wizardKind = resolveWizardKind(amenityTypeId, amenityTypes);
  const ActiveForm = AMENITY_FORM_REGISTRY[wizardKind];

  const showDowngradeWarning =
    isEditing && originalWizardKind === SCHEDULED_ADMISSION_KEY && wizardKind === GENERAL_ADMISSION_KEY;

  return (
    <Drawer title={isEditing ? 'Edit amenity' : 'Add amenity'} width="sm" isOpen={isOpen} onClose={onClose}>

          <label className="text-sm font-medium text-gray-700">Amenity type</label>
          <select
            className="w-full rounded-md border border-gray-300 bg-white px-3 py-2 text-sm text-gray-900 focus:outline-none focus:ring-2 focus:ring-blue-500"
            value={amenityTypeId}
            onChange={(e) => setAmenityTypeId(e.target.value ? Number(e.target.value) : '')}
          >
            <option value="">Select...</option>
            {amenityTypes.map((t) => (
              <option key={t.amenityTypeId} value={t.amenityTypeId}>{t.amenityTypeName}</option>
            ))}
          </select>
        </div>

        {showDowngradeWarning && (
          <p className="text-sm text-amber-700 bg-amber-50 border border-amber-200 rounded-md px-3 py-2">
            Switching back to General Admission will remove the existing schedule for this amenity.
          </p>
        )}

        {amenityTypeId !== '' && ActiveForm && (
          <ActiveForm
            key={wizardKind}
            amenityTypeId={Number(amenityTypeId)}
            isEditing={isEditing}
            amenityId={editingAmenity?.amenityId ?? null}
            locationFunctionId={locationFunctionId}
            locationSlug={locationSlug}
            onClose={onClose}
          />
        )}
      </div>
    </Drawer>
  );
}
