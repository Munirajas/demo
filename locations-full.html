// ── Tree utilities ────────────────────────────────────────────

// Find any node by id (any depth)
export function findLocationById(
  nodes: LocationNode[],
  id: string
): LocationNode | null {
  for (const node of nodes) {
    if (node.id === id) return node;
    if (node.children?.length) {
      const found = findLocationById(node.children, id);
      if (found) return found;
    }
  }
  return null;
}

// Get the first location node (skip client nodes)
export function getFirstLocation(
  nodes: LocationNode[]
): LocationNode | null {
  for (const node of nodes) {
    if (node.type === 'location') return node;
    if (node.children?.length) {
      const found = getFirstLocation(node.children);
      if (found) return found;
    }
  }
  return null;
}

// Get breadcrumb path for a location
export function getLocationBreadcrumb(
  nodes: LocationNode[],
  targetId: string,
  path: LocationNode[] = []
): LocationNode[] | null {
  for (const node of nodes) {
    const newPath = [...path, node];
    if (node.id === targetId) return newPath;
    if (node.children?.length) {
      const found = getLocationBreadcrumb(node.children, targetId, newPath);
      if (found) return found;
    }
  }
  return null;
}

// ── useUpdateLocationConfig ───────────────────────────────────
export function useUpdateLocationConfig(locationId: string) {
  const queryClient = useQueryClient();

  return useMutation<
    LocationConfig,
    Error,
    Omit<LocationConfig, 'locationId'>
  >({
    mutationFn: async (config) => {
      // ── MOCK ─────────────────────────────────────────────
      await new Promise(r => setTimeout(r, 300));
      return { locationId, ...config };
      // ── END MOCK ─────────────────────────────────────────
    },
    // Optimistic update — UI updates before server confirms
    onMutate: async (newConfig) => {
      await queryClient.cancelQueries({ queryKey: locationKeys.config(locationId) });
      const previous = queryClient.getQueryData<LocationConfig>(locationKeys.config(locationId));
      queryClient.setQueryData(locationKeys.config(locationId), {
        locationId,
        ...newConfig,
      });
      return { previous }; // rollback context
    },
    onError: (_, __, context) => {
      // Rollback on error
      if (context?.previous) {
        queryClient.setQueryData(locationKeys.config(locationId), context.previous);
      }
    },
    onSettled: () => {
      queryClient.invalidateQueries({ queryKey: locationKeys.config(locationId) });
    },
  });
}

'use client';

import { useState }      from 'react';
import { useRouter }     from 'next/navigation';
import type { LocationNode } from '../types/location.types';

interface TreeProps {
  nodes:             LocationNode[];
  activeLocationId?: string;
  depth?:            number;
}

export function LocationTree({ nodes, activeLocationId, depth = 0 }: TreeProps) {
  if (!nodes.length) return null;
  return (
    <ul className={depth > 0 ? 'ml-3 mt-0.5' : ''}>
      {nodes.map(node => (
        <LocationTreeNode
          key={node.id}
          node={node}
          activeLocationId={activeLocationId}
          depth={depth}
        />
      ))}
    </ul>
  );
}

function LocationTreeNode({
  node,
  activeLocationId,
  depth,
}: {
  node:              LocationNode;
  activeLocationId?: string;
  depth:             number;
}) {
  const router      = useRouter();
  const hasChildren = Boolean(node.children?.length);
  const isClient    = node.type === 'client';
  const isActive    = node.id === activeLocationId;

  // Auto-expand if active location is within this subtree
  const [expanded, setExpanded] = useState(() =>
    hasDescendantActive(node, activeLocationId)
  );

  function handleClick() {
    if (isClient) {
      // Client node — toggle expand only, not navigate
      if (hasChildren) setExpanded(e => !e);
      return;
    }
    // Location node — navigate to config
    router.push(`/locations/${node.id}/config`);
  }

  return (
    <li className="my-0.5">
      <button
        onClick={handleClick}
        className={`
          w-full flex items-center gap-1.5 px-2 py-1.5 rounded-lg
          text-sm text-left transition-colors
          ${isActive
            ? 'bg-blue-600 text-white font-medium'
            : isClient
              ? 'text-gray-800 font-semibold hover:bg-gray-100 cursor-default'
              : 'text-gray-600 hover:text-gray-900 hover:bg-gray-100'
          }
        `}
      >
        {/* Expand indicator */}
        {hasChildren ? (
          <svg
            width="11" height="11" viewBox="0 0 24 24" fill="none"
            className={`shrink-0 transition-transform opacity-50
              ${expanded ? 'rotate-90' : ''}`}
            onClick={e => { e.stopPropagation(); setExpanded(x => !x); }}
          >
            <path d="M9 18l6-6-6-6" stroke="currentColor" strokeWidth="2.5"
              strokeLinecap="round" strokeLinejoin="round"/>
          </svg>
        ) : (
          <span className="w-[11px] shrink-0" />
        )}

        <span className="truncate">{node.label}</span>
      </button>

      {hasChildren && expanded && (
        <LocationTree
          nodes={node.children!}
          activeLocationId={activeLocationId}
          depth={depth + 1}
        />
      )}
    </li>
  );
}

function hasDescendantActive(node: LocationNode, activeId?: string): boolean {
  if (!activeId) return false;
  if (node.id === activeId) return true;
  return node.children?.some(c => hasDescendantActive(c, activeId)) ?? false;
}

          </div>
        )}

          <button
            onClick={() => setShowCreate(true)}
            className="w-full flex items-center justify-center gap-2 h-10 bg-blue-600 hover:bg-blue-700 text-white text-xs font-bold rounded-lg transition-colors tracking-wide"

        />
      )}

// Filter tree nodes by search term
function filterTree(nodes: LocationNode[], term: string): LocationNode[] {
  return nodes.reduce<LocationNode[]>((acc, node) => {
    const matches = node.label.toLowerCase().includes(term);
    const filteredChildren = node.children
      ? filterTree(node.children, term)
      : [];

    if (matches || filteredChildren.length > 0) {
      acc.push({ ...node, children: filteredChildren });
    }
    return acc;
  }, []);
}

'use client';

// ─────────────────────────────────────────────────────────────
// PROVIDERS — CLIENT COMPONENT
//
// Wraps the client tree with:
//   1. TanStack QueryClient (data fetching + caching)
//   2. TokenRefreshProvider (proactive token refresh)
//
// Mounted in (dashboard)/layout.tsx (Server Component)
// Everything inside this is client-side
// ─────────────────────────────────────────────────────────────

import { useState }          from 'react';
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import { TokenRefreshProvider } from '@/modules/auth/components/TokenRefreshProvider';

export function Providers({ children }: { children: React.ReactNode }) {
  // useState ensures QueryClient is created once per browser session
  // NOT on every server render
  const [queryClient] = useState(() => new QueryClient({
    defaultOptions: {
      queries: {
        staleTime:          1000 * 60 * 5,  // 5 min
        refetchOnWindowFocus: false,         // don't refetch on tab focus
        retry:              1,              // retry once on failure
      },
    },
  }));

  return (
    <QueryClientProvider client={queryClient}>
      <TokenRefreshProvider>
        {children}
      </TokenRefreshProvider>
    </QueryClientProvider>
  );
}
