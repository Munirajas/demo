server-location-plans.ts ServerFixed

Your pattern kept. Added mapper normalization + imports from correct modules.

🚨
Move the mutation import. Change ../../locations/graphql/locations.mutations → ../graphql/plans.mutations. Plans mutations must live in the plans module.
src/modules/plans/actions/server-location-plans.ts 
'use server';

import { cache } from 'react';
import { serverGqlFetch } from '@/lib/graphql/server-gql-fetcher';

// ✅ Correct: mutation lives in plans module, not locations
import { GET_LOCATION_PLAN_MUTATION } from
  '../graphql/plans.mutations';

// ✅ Shared types — not inline interfaces
import { mapRawPlansToLocationPlans } from
  '../lib/plans.mappers';
import { PLAN_REQUEST_TYPE } from
  '../lib/plans.constants';
import type {
  RawGetLocationPlanResponse,
  GetLocationPlansResult,
} from '../types/plans.types';

// ─── Response shape from GraphQL ─────────────────────────
interface GetLocationPlanGQLResponse {
  getLocationPlan: RawGetLocationPlanResponse;
}

/**
 * Fetches all plans for a given location.
 *
 * React.cache() deduplicates calls within a single render pass —
 * if two server components call this with the same locationId,
 * only one network request is made. ✅ Keep this pattern.
 *
 * @param locationId - numeric ID from the location tree node
 */
export const getLocationPlans = cache(
  async (locationId: number): Promise<GetLocationPlansResult> => {
    try {
      const response = await serverGqlFetch<GetLocationPlanGQLResponse>(
        GET_LOCATION_PLAN_MUTATION,
        {
          type: PLAN_REQUEST_TYPE,
          locationId,
        },
      );

      const result = response.getLocationPlan;

      if (!result?.success) {
        return {
          plans: [],
          success: false,
          message: result?.message ?? 'Failed to fetch plans',
          error: result?.message,
        };
      }

      // ✅ Normalize here — snake_case never reaches components
      return {
        plans: mapRawPlansToLocationPlans(result.data),
        success: true,
        message: result.message,
      };
    } catch (error) {
      console.error('[getLocationPlans]', error);
      return {
        plans: [],
        success: false,
        message: 'Failed to fetch location plans',
        error: 'Failed to fetch location plans',
      };
    }
  },
);

PlanShell.tsx TSXClientNew

The top-level client shell. Receives server-fetched data as props. Owns client state like selected plan. Use this as the pattern for all future function shells (TicketsShell, EventsShell etc.)

💡
Shell pattern rule: Server page fetches → passes to Shell as props → Shell manages client state + renders child components. Never fetch in the shell itself. This is the pattern to reuse for every function page.
src/modules/plans/components/PlanShell.tsx 
'use client';

import { useState } from 'react';
import type { PlanShellProps } from '../types/plans.types';
import { ActivePlanBanner } from './ActivePlanBanner';
import { PlanGrid } from './PlanGrid';
import { PlanEmptyState } from './PlanEmptyState';

/**
 * PlanShell — Client shell for the plan-selection function page.
 *
 * PATTERN TO REUSE:
 *   Server page (async, fetches data)
 *     → passes data as props to Shell (client, manages state)
 *       → Shell renders child components
 *
 *   For the next function (e.g. tickets):
 *   Create TicketsShell with the same structure.
 */
export function PlanShell({ plans, locationId, message }: PlanShellProps) {
  // Client state — selected plan for highlight / action
  const [selectedPlanId, setSelectedPlanId] = useState<number | null>(null);

  // Find the active plan (first one for now — adjust when BE adds active flag)
  const activePlan = plans.length > 0 ? plans[0] : null;

  if (plans.length === 0) {
    return <PlanEmptyState locationId={locationId} />;
  }

  return (
    <div className="flex flex-col gap-6 p-6">
      {/* Page title */}
      <div>
        <h2 className="text-lg font-semibold text-gray-900">
          Plan Selection
        </h2>
        <p className="text-sm text-gray-500 mt-1">
          Manage the plan assigned to this location.
        </p>
      </div>

      {/* Active plan highlight */}
      {activePlan && (
        <ActivePlanBanner plan={activePlan} />
      )}

      {/* All available plans */}
      <PlanGrid
        plans={plans}
        selectedPlanId={selectedPlanId}
        onSelect={setSelectedPlanId}
        locationId={locationId}
      />
    </div>
  );
}

plan-selection/page.tsx ServerUpdated

Uses the barrel import. Node id fed directly as locationId — no string conversion needed if tree already provides number.

src/app/(protected)/location/[slug]/[fn]/plan-selection/page.tsx 
import { notFound } from 'next/navigation';
import { getLocationsTree } from
  '@/modules/locations/actions/server-locations';
import { getNodeAndPathBySlug } from
  '@/modules/locations/lib/slug';
import { getLocationPlans } from
  '@/modules/plans/actions/server-location-plans';

// ✅ Barrel import — clean, future-proof
import { PlanShell } from '@/modules/plans/components';

interface Props {
  params: Promise<{ slug: string; fn: string }>;
}

export default async function PlanSelectionPage({ params }: Props) {
  const { slug } = await params;

  // Resolve numeric locationId from slug
  const tree = await getLocationsTree();
  const found = getNodeAndPathBySlug(tree, slug);
  if (!found) notFound();

  const locationId: number = found.node.id;

  // React.cache() in server action deduplicates this call
  // if getLocationsTree already called it above ✅
  const result = await getLocationPlans(locationId);

  // Graceful: show empty state instead of crashing on error
  return (
    <PlanShell
      plans={result.plans}
      locationId={locationId}
      message={result.message}
    />
  );
}
