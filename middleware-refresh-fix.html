Where refreshAuth is used — full picture
TOKEN EXPIRY — TWO PATHS: PATH A: Server Component request expires serverGqlFetch() catches 401 error ↓ calls refreshAccessToken() directly refreshAuth(userId, token) → BE new tokens → setAuthCookies() retry original GQL request if still fails → throw SESSION_EXPIRED PATH B: Client Component query expires (Apollo) useQuery/useMutation gets UNAUTHENTICATED ↓ Apollo errorLink fires POST /api/auth/refresh ↓ calls refreshAccessToken() on server refreshAuth(userId, token) → BE new tokens → setAuthCookies() returns new accessToken to Apollo Apollo retries original failed request if refresh fails → window.href = '/login'

import { NextRequest, NextResponse } from 'next/server';

// ─────────────────────────────────────────────────────────────
// MIDDLEWARE — runs at the Edge BEFORE any page renders
//
// This is the FIRST line of defence for route protection.
// It reads the access_token httpOnly cookie directly from
// the request — fast, no async, no DB calls.
//
// Three jobs:
//   1. Let internal Next.js + auth API routes pass through
//   2. Unauthenticated user hits protected route → /login
//   3. Authenticated user hits /login or /register → /dashboard
// ─────────────────────────────────────────────────────────────

// Routes anyone can access without being logged in
const PUBLIC_ROUTES = ['/login', '/register'];

// Prefixes that must ALWAYS pass through — never intercepted
const BYPASS_PREFIXES = [
  '/api/auth',        // our own auth API routes (token, refresh, logout)
  '/_next/static',    // Next.js static assets
  '/_next/image',     // Next.js image optimisation
  '/favicon.ico',
  '/public',
];

export default function middleware(req: NextRequest) {
  const { pathname } = req.nextUrl;

  // ── 1. Always bypass internal routes ───────────────────────
  // Without this, middleware would intercept /api/auth/token
  // and cause an infinite loop
  if (BYPASS_PREFIXES.some((prefix) => pathname.startsWith(prefix))) {
    return NextResponse.next();
  }

  // ── 2. Read access_token from httpOnly cookie ───────────────
  // This is the cookie set by setAuthCookies() after login.
  // Middleware can read httpOnly cookies directly from request.
  const accessToken = req.cookies.get('access_token')?.value;
  const isAuthenticated = Boolean(accessToken);

  const isPublicRoute = PUBLIC_ROUTES.some(
    (route) => pathname === route || pathname.startsWith(route)
  );

  // ── 3. Protected route + not logged in → send to login ─────
  if (!isPublicRoute && !isAuthenticated) {
    const loginUrl = new URL('/login', req.url);
    // Remember where user was trying to go
    // After login they'll be redirected back here
    loginUrl.searchParams.set('callbackUrl', pathname);
    return NextResponse.redirect(loginUrl);
  }

  // ── 4. Already logged in + hitting public route → dashboard ─
  if (isPublicRoute && isAuthenticated) {
    return NextResponse.redirect(new URL('/dashboard', req.url));
  }

  // ── 5. All other cases — let the request through ────────────
  return NextResponse.next();
}

// Tell Next.js which paths this middleware runs on.
// Excludes static files and images for performance.
export const config = {
  matcher: [
    '/((?!_next/static|_next/image|favicon.ico|.*\\.png$|.*\\.jpg$|.*\\.svg$).*)',
  ],
};

src/modules/auth/lib/refresh.ts
SERVER ONLY
Calls refreshAuth(userId, token) — matches your exact schema. On success writes 4 new cookies. On failure clears all cookies so user gets clean login.
import 'server-only';

// ─────────────────────────────────────────────────────────────
// TOKEN REFRESH — SERVER ONLY
//
// Called when:
//   A) Server Component gets SESSION_EXPIRED from serverGqlFetch
//   B) Apollo errorLink gets UNAUTHENTICATED from a client query
//      (via POST /api/auth/refresh route)
//
// refreshAuth mutation requires:
//   userId: String!  → stored in user_id cookie after login
//   token:  String!  → current accessToken (access_token cookie)
//   (confirmed from your schema screenshot)
//
// On success: writes 4 new httpOnly cookies, returns new accessToken
// On failure: returns null — caller must redirect to /login
// ─────────────────────────────────────────────────────────────

import { GraphQLClient, ClientError } from 'graphql-request';
import { GQL_ENDPOINT } from '@/lib/constants';
import {
  getAccessToken,
  getUserId,
  setAuthCookies,
  clearAuthCookies,
} from '@/modules/auth/lib/cookies';
import type { RefreshAuthMutationResponse } from '@/types/auth.types';

// Must match your BE schema exactly
const REFRESH_AUTH_MUTATION = `
  mutation RefreshAuth($userId: String!, $token: String!) {
    refreshAuth(userId: $userId, token: $token) {
      token
      idToken
      refreshToken
    }
  }
`;

export async function refreshAccessToken(): Promise<string | null> {
  // We need both to call refreshAuth
  const [currentToken, userId] = await Promise.all([
    getAccessToken(),
    getUserId(),
  ]);

  // Can't refresh without both — session is broken
  if (!currentToken || !userId) {
    console.warn('[refreshAccessToken] Missing token or userId — cannot refresh');
    return null;
  }

  try {
    const client = new GraphQLClient(GQL_ENDPOINT, {
      headers: { 'Content-Type': 'application/json' },
    });

    const data = await client.request<RefreshAuthMutationResponse>(
      REFRESH_AUTH_MUTATION,
      {
        userId,
        token: currentToken,  // accessToken — confirmed from your schema
      }
    );

    // Validate the response
    if (!data?.refreshAuth?.token) {
      console.error('[refreshAccessToken] BE returned no token');
      return null;
    }

    // Write all 4 new cookies atomically
    await setAuthCookies(data.refreshAuth);

    return data.refreshAuth.token;

  } catch (error) {
    if (error instanceof ClientError) {
      const msg = error.response?.errors?.[0]?.message ?? '';
      console.error('[refreshAccessToken] BE error:', msg);

      // If refresh itself returns auth error — tokens fully expired
      // Clear all cookies so user gets clean login page
      if (
        msg.toLowerCase().includes('expired') ||
        msg.toLowerCase().includes('invalid') ||
        msg.toLowerCase().includes('unauthenticated')
      ) {
        await clearAuthCookies();
      }
    } else {
      console.error('[refreshAccessToken] Unexpected error:', error);
    }

    return null;
  }
}
server-gql-fetcher.ts
src/lib/apollo/server-gql-fetcher.ts
SERVER
Use in ALL Server Components. Auto-refreshes expired token before throwing SESSION_EXPIRED.
import 'server-only';

// ─────────────────────────────────────────────────────────────
// SERVER GQL FETCHER
//
// Use this in ALL Server Components for data fetching.
// Never use Apollo Client hooks in Server Components.
//
// Auto-refresh flow:
//   1. Read accessToken from cookie
//   2. If no token → try refresh once → if still nothing → SESSION_EXPIRED
//   3. Send request with Bearer token
//   4. If 401 → try refresh once → retry request → or SESSION_EXPIRED
//
// Callers (Server Component pages) handle SESSION_EXPIRED
// by catching the error and redirecting to /login.
// ─────────────────────────────────────────────────────────────

import { GraphQLClient, ClientError } from 'graphql-request';
import { GQL_ENDPOINT } from '@/lib/constants';
import { getAccessToken } from '@/modules/auth/lib/cookies';
import { refreshAccessToken } from '@/modules/auth/lib/refresh';

function buildClient(token: string): GraphQLClient {
  return new GraphQLClient(GQL_ENDPOINT, {
    headers: {
      Authorization: `Bearer ${token}`,
      'Content-Type': 'application/json',
    },
  });
}

function isAuthError(error: unknown): boolean {
  if (error instanceof ClientError) {
    const status = error.response?.status;
    const code   = error.response?.errors?.[0]?.extensions?.code;
    return status === 401 || code === 'UNAUTHENTICATED';
  }
  return false;
}

export async function serverGqlFetch<
  TData,
  TVars extends object = Record<string, never>
>(
  query:      string,
  variables?: TVars
): Promise<TData> {

  let token = await getAccessToken();

  // No token — attempt silent refresh before giving up
  if (!token) {
    token = await refreshAccessToken();
    if (!token) {
      throw new Error('SESSION_EXPIRED');
    }
  }

  try {
    return await buildClient(token).request<TData>(query, variables);

  } catch (error) {
    // Token expired mid-session — refresh once and retry
    if (isAuthError(error)) {
      const newToken = await refreshAccessToken();
      if (!newToken) {
        throw new Error('SESSION_EXPIRED');
      }
      // Retry original request with new token
      return buildClient(newToken).request<TData>(query, variables);
    }

    // Non-auth error — rethrow as-is
    throw error;
  }
}
token/route.ts
src/app/api/auth/token/route.ts
SERVER
Bridge: httpOnly cookie → Apollo authLink. Apollo runs in browser and needs the token to set Authorization header.
// ─────────────────────────────────────────────────────────────
// /api/auth/token  — GET
//
// WHY this exists:
// httpOnly cookies are INVISIBLE to JavaScript by design.
// Apollo Client's authLink runs in the browser and needs the
// accessToken to set the Authorization header.
// This route is the ONLY secure bridge:
//
//   Browser (Apollo) → GET /api/auth/token
//   → Next.js reads httpOnly cookie (server-side)
//   → Returns { accessToken }
//   → Apollo sets Authorization: Bearer <token>
//
// This is safe because:
//   - Same-origin only (no CORS issue)
//   - Cookie is still httpOnly (JS can't steal it via XSS)
//   - Only exposes the token value, not the cookie itself
// ─────────────────────────────────────────────────────────────

import { NextResponse } from 'next/server';
import { getAccessToken } from '@/modules/auth/lib/cookies';

export async function GET() {
  const accessToken = await getAccessToken();
  return NextResponse.json({
    accessToken: accessToken ?? null,
  });
}
refresh/route.ts
src/app/api/auth/refresh/route.ts
SERVER
Called by Apollo errorLink on UNAUTHENTICATED. Triggers server-side token refresh, returns new accessToken.
// ─────────────────────────────────────────────────────────────
// /api/auth/refresh  — POST
//
// Called by Apollo's errorLink when a CLIENT-SIDE query gets
// an UNAUTHENTICATED error.
//
// Flow:
//   Apollo errorLink → POST /api/auth/refresh
//   → refreshAccessToken() calls refreshAuth mutation on BE
//   → new tokens written to httpOnly cookies
//   → returns new accessToken so Apollo can retry failed request
//
// If refresh fails → 401 → Apollo errorLink redirects to /login
// ─────────────────────────────────────────────────────────────

import { NextResponse } from 'next/server';
import { refreshAccessToken } from '@/modules/auth/lib/refresh';

export async function POST() {
  const newToken = await refreshAccessToken();

  if (!newToken) {
    return NextResponse.json(
      { error: 'Session expired. Please sign in again.' },
      { status: 401 }
    );
  }

  return NextResponse.json({ accessToken: newToken });
}
