import type { TreeNode } from '@/types/locations';

export function slugify(label: string): string {
  return label
    .toLowerCase()
    .trim()
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/^-+|-+$/g, '');
}

export function getNodeAndPathBySlug(
  nodes: TreeNode[],
  slug:  string,
  path:  TreeNode[] = []
): { node: TreeNode; path: TreeNode[] } | null {
  for (const node of nodes) {
    const current = [...path, node];
    if (slugify(node.label) === slug) {
      return { node, path: current };
    }
    if (node.children?.length) {
      const found = getNodeAndPathBySlug(node.children, slug, current);
      if (found) return found;
    }
  }
  return null;
}

// fn omitted → /location/{slug}  (no-function page)
// fn given   → /location/{slug}/{fn}
export function locationUrl(node: TreeNode, fn?: string): string {
  const base = `/location/${slugify(node.label)}`;
  return fn ? `${base}/${fn}` : base;
}

// src/components/layout/LocationShell.tsx
// SERVER COMPONENT — shared by:
//   /location/{slug}          (activeFn = null)
//   /location/{slug}/{fn}     (activeFn = fn slug)
//
// ─────────────────────────────────────────────────────────────
// PERFORMANCE:
//   getLocationsTree() and getLocationFunctions(id) are both
//   wrapped in React cache(). Calling them here means EITHER
//   page can call this shell directly — no duplicated fetches
//   within the same request, and no shared-layout coupling
//   between the two/three route variants.
//
// BREADCRUMBS:
//   activeFn present:
//     [...path locations (each links to same fn)] + {fnLabel}
//   activeFn null:
//     [...path locations except last (link to /location/{slug})]
//       + {current location label}  ← non-clickable, current page
// ─────────────────────────────────────────────────────────────

import { notFound, redirect } from 'next/navigation';
import { getLocationsTree }     from '@/modules/locations/graphql/server-locations';
import { getLocationFunctions } from '@/modules/locations/graphql/server-location-functions';
import { getNodeAndPathBySlug, locationUrl } from '@/modules/locations/lib/slug';
import { FUNCTION_LABELS, type FunctionSlug } from '@/modules/locations/graphql/location-functions.types';
import { Breadcrumbs }   from '@/components/nav/Breadcrumbs';
import { FunctionsNav }  from '@/components/layout/FunctionsNav';
import { NoFunctionsAvailable }  from '@/modules/functions/noFunctions/NoFunctionsAvailable';
import type { BreadcrumbItem, TreeNode } from '@/types/locations';

interface Props {
  slug:     string;
  activeFn: string | null; // null = /location/{slug} (no function selected)
  children: React.ReactNode;
}

export async function LocationShell({ slug, activeFn, children }: Props) {
  const tree  = await getLocationsTree();          // cache() — one fetch/request
  const found = getNodeAndPathBySlug(tree, slug);
  if (!found) notFound();                          // location doesn't exist → real 404

  const { node, path } = found;

  const functions = await getLocationFunctions(node.id); // cache() — one fetch/request

  // ── No functions enabled for this location at all ──────────
  // Valid state regardless of activeFn — show message, not 404.
  if (functions.length === 0) {
    const breadcrumbs = buildBreadcrumbs(node, path, null);
    return (
      <Shell breadcrumbs={breadcrumbs}>
        <NoFunctionsAvailable locationLabel={node.label} />
      </Shell>
    );
  }

  // ── activeFn given but invalid/disabled → redirect to /location/{slug}
  // (the no-function page), NOT to config — config isn't guaranteed
  // to be the right "default" once functions become location-specific.
  if (activeFn && !functions.some(f => f.slug === activeFn)) {
    redirect(locationUrl(node));
  }

  const breadcrumbs = buildBreadcrumbs(node, path, activeFn);

  return (
    <Shell breadcrumbs={breadcrumbs}>
      <div className="flex flex-1 overflow-hidden">
        <aside className="w-48 border-r border-gray-200 flex flex-col shrink-0">
          <div className="p-3 border-b border-gray-100">
            <p className="text-xs font-semibold text-gray-500 uppercase tracking-wide truncate">
              {node.label}
            </p>
          </div>
          <div className="flex-1 overflow-y-auto">
            <FunctionsNav
              locationSlug={slug}
              activeFn={activeFn}
              functions={functions}
            />
          </div>
        </aside>

        <main className="flex-1 overflow-y-auto">
          {children}
        </main>
      </div>
    </Shell>
  );
}

// ── Breadcrumb bar wrapper ─────────────────────────────────────
function Shell({
  breadcrumbs, children,
}: { breadcrumbs: BreadcrumbItem[]; children: React.ReactNode }) {
  return (
    <div className="flex flex-col h-full">
      <div className="px-6 py-3 border-b border-gray-100 bg-white shrink-0">
        <Breadcrumbs items={breadcrumbs} />
      </div>
      {children}
    </div>
  );
}

// ── Build breadcrumb items ──────────────────────────────────────
function buildBreadcrumbs(
  node:     TreeNode,
  path:     TreeNode[],
  activeFn: string | null,
): BreadcrumbItem[] {

  if (activeFn) {
    // All locations link to the SAME fn (switching location keeps
    // user on the same tab, e.g. physical-address → physical-address)
    const fnLabel = FUNCTION_LABELS[activeFn as FunctionSlug] ?? activeFn;
    return [
      ...path.map(ancestor => ({
        label: ancestor.label,
        href:  ancestor.type === 'client' ? undefined : locationUrl(ancestor, activeFn),
      })),
      { label: fnLabel }, // current function — last, non-clickable
    ];
  }

  // No active function: ancestors link to their own no-fn page,
  // current location is the trailing non-clickable item.
  const ancestors = path.slice(0, -1);
  return [
    ...ancestors.map(ancestor => ({
      label: ancestor.label,
      href:  ancestor.type === 'client' ? undefined : locationUrl(ancestor),
    })),
    { label: node.label }, // current location — last, non-clickable
  ];
}
