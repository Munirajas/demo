'use client';
import { ApolloClient, InMemoryCache, from }
  from '@apollo/client';
import { createAuthLink, createErrorLink,
  createHttpLink } from './apollo-links';

let instance: ApolloClient<object> | null = null;

function createApolloClient() {
  return new ApolloClient({
    link: from([
      createErrorLink(), // 1 catch UNAUTHENTICATED
      createAuthLink(),  // 2 attach Bearer token
      createHttpLink(),  // 3 HTTP transport
    ]),
    cache: new InMemoryCache(),
    defaultOptions: {
      watchQuery: {
        fetchPolicy: 'cache-and-network',
        errorPolicy: 'all',
      },
      query: { errorPolicy: 'all' },
    },
  });
}

export function getApolloClient() {
  if (!instance) instance = createApolloClient();
  return instance;
}

// MUST call on logout to prevent data leak
export function resetApolloClient() {
  instance?.clearStore();
  instance = null;
}
apollo-links.ts
src/lib/apollo/apollo-links.ts
CLIENT▼
Error link queues concurrent requests. If 5 queries fire and all get 401, only ONE refresh happens. Others wait in queue.
'use client';
import { ApolloLink, HttpLink, Observable,
  FetchResult, fromPromise }
  from '@apollo/client';
import { onError }
  from '@apollo/client/link/error';
import { GQL_ENDPOINT }
  from '@/lib/constants';

export function createHttpLink() {
  return new HttpLink({ uri: GQL_ENDPOINT });
}

export function createAuthLink(): ApolloLink {
  return new ApolloLink((operation, forward) =>
    new Observable<FetchResult>(observer => {
      fetch('/api/auth/token')
        .then(r => r.json())
        .then(({ accessToken }) => {
          if (accessToken) operation.setContext(
            ({ headers = {} }: Record<string, unknown>) => ({
              headers: {
                ...(headers as object),
                Authorization: `Bearer ${accessToken}`,
              },
            })
          );
        })
        .catch(() => {})
        .finally(() => {
          forward(operation).subscribe({
            next:     observer.next.bind(observer),
            error:    observer.error.bind(observer),
            complete: observer.complete.bind(observer),
          });
        });
    })
  );
}

let isRefreshing = false;
let queue: Array<() => void> = [];

export function createErrorLink(): ApolloLink {
  return onError(
    ({ graphQLErrors, operation, forward }) => {
      const isAuth = graphQLErrors?.some(
        e => e.extensions?.code === 'UNAUTHENTICATED'
      );
      if (!isAuth) return;
      if (isRefreshing) {
        return fromPromise(
          new Promise<void>(res => queue.push(res))
        ).flatMap(() => forward(operation));
      }
      isRefreshing = true;
      return fromPromise(
        fetch('/api/auth/refresh', { method: 'POST' })
          .then(res => {
            if (!res.ok) throw 'fail';
            return res.json() as
              Promise<{ accessToken: string }>;
          })
          .then(({ accessToken }) => {
            operation.setContext(
              ({ headers = {} }: Record<string, unknown>) => ({
                headers: {
                  ...(headers as object),
                  Authorization: `Bearer ${accessToken}`,
                },
              })
            );
            queue.forEach(r => r()); queue = [];
          })
          .catch(() => {
            queue = [];
            window.location.href = '/login';
          })
          .finally(() => { isRefreshing = false; })
      ).flatMap(() => forward(operation));
    }
  );
}

import 'server-only';
import { GraphQLClient, ClientError }
  from 'graphql-request';
import { GQL_ENDPOINT }
  from '@/lib/constants';
import { getAccessToken }
  from '@/modules/auth/lib/cookies';
import { refreshAccessToken }
  from '@/modules/auth/lib/refresh';

function buildClient(token: string) {
  return new GraphQLClient(GQL_ENDPOINT, {
    headers: { Authorization: `Bearer ${token}` },
  });
}

function isAuthError(e: unknown): boolean {
  if (!(e instanceof ClientError)) return false;
  const code =
    e.response?.errors?.[0]?.extensions?.code;
  return e.response?.status === 401
    || code === 'UNAUTHENTICATED';
}

export async function serverGqlFetch<
  TData,
  TVars extends object = Record<string, never>
>(query: string, variables?: TVars):
  Promise<TData> {
  let token = await getAccessToken();
  if (!token) {
    token = await refreshAccessToken();
    if (!token) throw new Error('SESSION_EXPIRED');
  }
  try {
    return await buildClient(token)
      .request<TData>(query, variables);
  } catch (error) {
    if (isAuthError(error)) {
      const t = await refreshAccessToken();
      if (!t) throw new Error('SESSION_EXPIRED');
      return buildClient(t)
        .request<TData>(query, variables);
    }
    throw error;
  }
}

import 'server-only';
import { cookies } from 'next/headers';
import {
  COOKIE_KEYS, COOKIE_OPTIONS,
  REFRESH_COOKIE_OPTIONS,
} from '@/lib/constants';
import type { AuthToken, DecodedIdToken }
  from '@/types/auth.types';

export function decodeJwtPayload<T>(
  token: string
): T | null {
  try {
    const json = Buffer
      .from(token.split('.')[1], 'base64')
      .toString('utf-8');
    return JSON.parse(json) as T;
  } catch { return null; }
}

export async function setAuthCookies(
  tokens: AuthToken
): Promise<void> {
  const store = await cookies();
  const p =
    decodeJwtPayload<DecodedIdToken>(tokens.idToken);
  store.set(COOKIE_KEYS.ACCESS_TOKEN,
    tokens.token, COOKIE_OPTIONS);
  store.set(COOKIE_KEYS.ID_TOKEN,
    tokens.idToken, COOKIE_OPTIONS);
  store.set(COOKIE_KEYS.REFRESH_TOKEN,
    tokens.refreshToken, REFRESH_COOKIE_OPTIONS);
  store.set(COOKIE_KEYS.USER_ID,
    p?.sub ?? '', COOKIE_OPTIONS);
}

export async function getAccessToken() {
  return (await cookies())
    .get(COOKIE_KEYS.ACCESS_TOKEN)?.value;
}
export async function getRefreshToken() {
  return (await cookies())
    .get(COOKIE_KEYS.REFRESH_TOKEN)?.value;
}
export async function getIdToken() {
  return (await cookies())
    .get(COOKIE_KEYS.ID_TOKEN)?.value;
}
export async function getUserId() {
  return (await cookies())
    .get(COOKIE_KEYS.USER_ID)?.value;
}
export async function clearAuthCookies() {
  const store = await cookies();
  Object.values(COOKIE_KEYS)
    .forEach(k => store.delete(k));
}

import 'server-only';
import { GraphQLClient }
  from 'graphql-request';
import { GQL_ENDPOINT }
  from '@/lib/constants';
import {
  getAccessToken, getUserId, setAuthCookies,
} from '@/modules/auth/lib/cookies';
import { REFRESH_AUTH_MUTATION }
  from '@/modules/auth/graphql/auth.mutations';
import type { RefreshAuthMutationResponse }
  from '@/types/auth.types';

export async function refreshAccessToken():
  Promise<string | null> {
  const [token, userId] = await Promise.all([
    getAccessToken(), getUserId(),
  ]);
  if (!token || !userId) return null;
  try {
    const data =
      await new GraphQLClient(GQL_ENDPOINT)
        .request<RefreshAuthMutationResponse>(
          REFRESH_AUTH_MUTATION,
          { userId, token }
        );
    await setAuthCookies(data.refreshAuth);
    return data.refreshAuth.token;
  } catch (err) {
    console.error('[Refresh Failed]', err);
    return null;
  }
}

'use server';
import { GraphQLClient, ClientError }
  from 'graphql-request';
import { redirect }
  from 'next/navigation';
import { GQL_ENDPOINT, ROUTES }
  from '@/lib/constants';
import { setAuthCookies }
  from '@/modules/auth/lib/cookies';
import { LOGIN_MUTATION }
  from '@/modules/auth/graphql/auth.mutations';
import type {
  LoginMutationResponse, ActionResult,
} from '@/types/auth.types';

export async function loginAction(input: {
  email: string;
  password: string;
}): Promise<ActionResult> {
  const { email, password } = input;
  if (!email || !password) return {
    success: false,
    error: 'Email and password required.',
  };
  try {
    const data =
      await new GraphQLClient(GQL_ENDPOINT)
        .request<LoginMutationResponse>(
          LOGIN_MUTATION, {
            email: email.trim().toLowerCase(),
            password,
          }
        );
    await setAuthCookies(data.login);
  } catch (error) {
    if (error instanceof ClientError) {
      const msg =
        error.response?.errors?.[0]?.message ?? '';
      if (msg.toLowerCase().includes('invalid'))
        return {
          success: false,
          error: 'Invalid email or password.',
        };
      return {
        success: false,
        error: msg || 'Login failed.',
      };
    }
    return {
      success: false,
      error: 'Something went wrong.',
    };
  }
  // redirect() MUST be outside try/catch
  redirect(ROUTES.DASHBOARD);
}

    .min(1, 'Password is required')
    .min(6, 'Min 6 characters'),

export function LoginForm() {
  const [serverError, setServerError] =
    useState<string | null>(null);

    useTransition();

    register, handleSubmit,
    formState: { errors },

    resolver: zodResolver(schema),

      const result = await loginAction(values);
      if (!result.success)
        setServerError(result.error);
    });

  return (
    <div className="login-card">
      <h1 className="login-title">Admin Portal</h1>
      <p className="login-subtitle">Sign in</p>

          {serverError}
        </div>
      )}
      <div className="form-fields">
        <div className="field-group">
          <label htmlFor="email">Email</label>
          <input id="email" type="email"
            placeholder="you@company.com"
            disabled={isPending}
            {...register('email')} />
          {errors.email && (
            <span className="field-error">
              {errors.email.message}
            </span>
          )}
        </div>
        <div className="field-group">
          <label htmlFor="password">Password</label>
          <input id="password" type="password"
            placeholder="●●●●●●●●"

            {...register('password')} />

            </span>

        </div>

          disabled={isPending}
          className="submit-button">
          {isPending ? 'Signing in...' : 'Sign in'}
        </button>
      </div>
    </div>
  );
}

'use client';
import { useEffect, useState }
  from 'react';
import type { SessionUser }
  from '@/types/auth.types';

export function useSession() {
  const [user, setUser] =
    useState<SessionUser | null>(null);
  const [loading, setLoading] =
    useState(true);

  useEffect(() => {
    fetch('/api/auth/token')
      .then(r => r.json())
      .then(({ accessToken }) => {
        if (!accessToken) {
          setUser(null); setLoading(false);
          return;
        }
        try {
          const p = JSON.parse(
            atob(accessToken.split('.')[1])
          );
          setUser({ userId: p.sub, email: p.email });
        } catch { setUser(null); }
        setLoading(false);
      })
      .catch(() => {
        setUser(null); setLoading(false);
      });
  }, []);

  return { user, loading };
}

EDGE▼
Runs before every request. First wall. Edge runtime only — cannot use async cookie functions here.
import { NextResponse }
  from 'next/server';
import type { NextRequest }
  from 'next/server';
import { COOKIE_KEYS, ROUTES }
  from '@/lib/constants';

const PUBLIC = [ROUTES.LOGIN];
const BYPASS = [
  '/api/auth',
  '/_next',
  '/favicon.ico',
];

export function middleware(req: NextRequest) {
  const { pathname } = req.nextUrl;
  if (BYPASS.some(p => pathname.startsWith(p)))
    return NextResponse.next();

  const token = req.cookies
    .get(COOKIE_KEYS.ACCESS_TOKEN)?.value;
  const isPublic =
    PUBLIC.some(r => pathname.startsWith(r));

  if (!isPublic && !token) {
    const url = new URL(ROUTES.LOGIN, req.url);
    url.searchParams.set('callbackUrl', pathname);
    return NextResponse.redirect(url);
  }
  if (isPublic && token)
    return NextResponse.redirect(
      new URL(ROUTES.DASHBOARD, req.url)
    );
  return NextResponse.next();
}

export const config = {
  matcher: [
    '/((?!_next/static|_next/image|favicon.ico).*)',
  ],
};

  return <UserTable users={users} />;
}

'use client';
import { useMutation, gql }
  from '@apollo/client';

const CREATE_USER = gql`
  mutation CreateUser($email: String!) {
    createUser(email: $email) { id email }
  }
`;

export function CreateUserForm() {
  const [createUser, { loading }] =
    useMutation(CREATE_USER, {
      update(cache, { data }) {
        // No manual refetch needed
        cache.modify({
          fields: {
            users: (existing = []) => [
              ...existing, data.createUser
            ],
          },
        });
      },
    });
  return ( /* your form JSX */ );
}
