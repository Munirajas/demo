⚠️
BE named a data fetch as a Mutation. Don't rename it — just wrap it cleanly in your action layer so UI code treats it like a normal data fetch. Never call mutations directly from components.

plans.mappers.ts TypeScriptNEW

The normalization layer. Snake_case → camelCase happens here only. One place to fix if BE renames fields.

💡
Why a separate mapper file? When BE changes Client_Plan_Id to plan_id, you fix it in exactly one place. Without this, you'd be hunting snake_case across components, hooks, and pages.
src/modules/plans/lib/plans.mappers.ts 
import type {
  RawLocationPlan,
  LocationPlan,
} from '../types/plans.types';

/**
 * Normalizes a single raw BE plan to UI-safe camelCase shape.
 * Only place in the codebase that knows about Client_Plan_Id etc.
 */
export function mapRawPlanToLocationPlan(
  raw: RawLocationPlan
): LocationPlan {
  return {
    id:   raw.Client_Plan_Id,
    name: raw.Client_Plan_Name,
  };
}

/**
 * Maps an array of raw BE plans.
 * Filters out any null/undefined items defensively.
 */
export function mapRawPlansToLocationPlans(
  rawPlans: RawLocationPlan[] | null | undefined
): LocationPlan[] {
  if (!rawPlans) return [];
  return rawPlans
    .filter(Boolean)
    .map(mapRawPlanToLocationPlan);
}

server-plans.ts ServerUpdated

Calls the mutation, normalizes the response, returns clean types. This is the only file that deals with raw BE data.

src/modules/plans/actions/server-plans.ts 
'use server';

import { getClient } from '@/lib/apollo-server-client';
import { GET_LOCATION_PLAN } from '../graphql/plans.mutations';
import { mapRawPlansToLocationPlans } from '../lib/plans.mappers';
import { PLAN_REQUEST_TYPE } from '../lib/plans.constants';
import type {
  GetLocationPlansResult,
  RawGetLocationPlanResponse,
} from '../types/plans.types';

/**
 * Fetches all plans for a location.
 *
 * NOTE: BE exposes this as a mutation (getLocationPlan).
 * We use Apollo mutate() here to match that. The naming
 * in this file stays "get" to be clear about intent.
 *
 * @param locationId - numeric location ID from the tree node
 * @param clientPlanId - optional: filter to a specific plan
 */
export async function getLocationPlans(
  locationId: number,
  clientPlanId?: number,
): Promise<GetLocationPlansResult> {
  try {
    const { data } = await getClient().mutate<{
      getLocationPlan: RawGetLocationPlanResponse;
    }>({
      mutation: GET_LOCATION_PLAN,
      variables: {
        request: {
          type: PLAN_REQUEST_TYPE,
          locationId,
          ...(clientPlanId !== undefined && { clientPlanId }),
        },
      },
    });

    const result = data?.getLocationPlan;

    if (!result?.success) {
      return {
        plans: [],
        success: false,
        message: result?.message ?? 'Unknown error',
        error: result?.message,
      };
    }

    // Normalize snake_case → camelCase here, never downstream
    const plans = mapRawPlansToLocationPlans(result.data);

    return {
      plans,
      success: true,
      message: result.message,
    };
  } catch (error) {
    console.error('[getLocationPlans]', error);
    return {
      plans: [],
      success: false,
      message: 'Failed to fetch location plans',
      error: 'Failed to fetch location plans',
    };
  }
}

useLocationPlan.ts ClientUpdated

Client hook for when you need to re-fetch plans on user action (e.g. filter change). For initial load, the server action is sufficient.

💡
When to use this hook vs server action: Server action = initial page load (no spinner). This hook = user triggers a re-fetch (e.g. changes a filter, refreshes after action). Both call the same BE mutation internally.
src/modules/plans/hooks/useLocationPlan.ts 
'use client';

import { useMutation, useQueryClient } from '@tanstack/react-query';
import { useGraphQLClient } from '@/hooks/useGraphQLClient';
import { GET_LOCATION_PLAN } from '../graphql/plans.mutations';
import { mapRawPlansToLocationPlans } from '../lib/plans.mappers';
import { PLAN_QUERY_KEYS, PLAN_REQUEST_TYPE } from '../lib/plans.constants';
import type {
  LocationPlan,
  RawGetLocationPlanResponse,
} from '../types/plans.types';

interface UseLocationPlanOptions {
  locationId: number;
  onSuccess?: (plans: LocationPlan[]) => void;
  onError?: (message: string) => void;
}

export function useLocationPlan({
  locationId,
  onSuccess,
  onError,
}: UseLocationPlanOptions) {
  const queryClient = useQueryClient();
  const graphqlClient = useGraphQLClient();

  const mutation = useMutation<
    LocationPlan[],
    Error,
    { clientPlanId?: number }
  >({
    mutationFn: async ({ clientPlanId }) => {
      const response = await graphqlClient.request<{
        getLocationPlan: RawGetLocationPlanResponse;
      }>(GET_LOCATION_PLAN, {
        request: {
          type: PLAN_REQUEST_TYPE,
          locationId,
          ...(clientPlanId !== undefined && { clientPlanId }),
        },
      });

      const result = response.getLocationPlan;
      if (!result.success) throw new Error(result.message);

      // Normalize here too — same mapper, same safety
      return mapRawPlansToLocationPlans(result.data);
    },
    onSuccess: (plans) => {
      // Update cache so other components see fresh data
      queryClient.setQueryData(
        PLAN_QUERY_KEYS.locationPlans(locationId),
        plans
      );
      onSuccess?.(plans);
    },
    onError: (error) => {
      onError?.(error.message);
    },
  });

  return {
    fetchPlans: mutation.mutate,
    fetchPlansAsync: mutation.mutateAsync,
    plans: mutation.data ?? [],
    isLoading: mutation.isPending,
    isError: mutation.isError,
    error: mutation.error,
  };
}

plan-selection/page.tsx Server ComponentUpdated

Uses the updated server action. locationId converted to number here since slug is always a string from the URL.

src/app/(protected)/location/[slug]/[fn]/plan-selection/page.tsx 
import { notFound } from 'next/navigation';
import { getLocationsTree } from
  '@/modules/locations/actions/server-locations';
import { getNodeAndPathBySlug } from
  '@/modules/locations/lib/slug';
import { getLocationPlans } from
  '@/modules/plans/actions/server-plans';
import { PlanShell } from
  '@/modules/plans/components/PlanShell';

interface Props {
  params: Promise<{ slug: string; fn: string }>;
}

export default async function PlanSelectionPage({ params }: Props) {
  const { slug } = await params;

  // Get the numeric locationId from the tree node
  const tree = await getLocationsTree();
  const found = getNodeAndPathBySlug(tree, slug);
  if (!found) notFound();

  const locationId = found.node.id; // number from tree

  // Fetch plans using the real BE mutation
  const result = await getLocationPlans(locationId);

  if (!result.success && result.error) {
    // You can render an error UI instead of notFound if preferred
    notFound();
  }

  return (
    <PlanShell
      plans={result.plans}
      locationId={locationId}
      message={result.message}
    />
  );
}
