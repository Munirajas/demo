Types TypeScript

Always define types first — they are the contract everything else is built on.

src/modules/plans/types/plans.types.ts 
// Plan interval billing cycle
export type PlanInterval = 'MONTHLY' | 'ANNUAL';

// Lifecycle status for both plans and subscriptions
export type PlanStatus = 'ACTIVE' | 'INACTIVE' | 'DEPRECATED';

export interface PlanFeature {
  id: string;
  name: string;
  description: string;
  included: boolean;
}

export interface Plan {
  id: string;
  name: string;
  description: string;
  price: number;
  interval: PlanInterval;
  status: PlanStatus;
  features: PlanFeature[];
  isPopular?: boolean;
}

export interface ActivePlan {
  id: string;
  plan: Plan;
  locationId: string;
  startDate: string;
  renewalDate: string;
  status: PlanStatus;
}

// Server action return shapes
export interface GetActivePlanResult {
  activePlan: ActivePlan | null;
  error?: string;
}

export interface GetMasterPlansResult {
  plans: Plan[];
  error?: string;
}

// Props contract for the page-level shell component
export interface PlanShellProps {
  activePlan: ActivePlan | null;
  masterPlans: Plan[];
  locationId: string;
}

Server Actions Server

These run on the server. They call GraphQL and return typed results to Server Components.

⚠️ Make sure your getClient() from @/lib/apollo-server-client is already set up to forward Cognito session tokens to GraphQL. If not, do that before wiring these actions.
src/modules/plans/actions/server-plans.ts 
'use server';

import { getClient } from '@/lib/apollo-server-client';
import { GET_ACTIVE_PLAN, GET_MASTER_PLANS } from '../graphql/plans.queries';
import type {
  GetActivePlanResult,
  GetMasterPlansResult,
} from '../types/plans.types';

export async function getActivePlan(
  locationId: string
): Promise<GetActivePlanResult> {
  try {
    const { data } = await getClient().query({
      query: GET_ACTIVE_PLAN,
      variables: { locationId },
    });
    return { activePlan: data.activePlan ?? null };
  } catch (error) {
    console.error('[getActivePlan]', error);
    return { activePlan: null, error: 'Failed to fetch active plan' };
  }
}

export async function getMasterPlans(): Promise<GetMasterPlansResult> {
  try {
    const { data } = await getClient().query({
      query: GET_MASTER_PLANS,
    });
    return { plans: data.plans ?? [] };
  } catch (error) {
    console.error('[getMasterPlans]', error);
    return { plans: [], error: 'Failed to fetch master plans' };
  }
}

Centralised query keys prevent cache invalidation bugs across hooks.

💡 Never hardcode query key strings inside hooks. One rename here fixes cache everywhere.

// Centralised TanStack Query keys for the plans module.
// Use these everywhere — hooks, invalidations, prefetching.
export const PLAN_QUERY_KEYS = {
  all: ['plans'] as const,
  activePlan: (locationId: string) =>
    ['plans', 'active', locationId] as const,
  masterPlans: () =>
    ['plans', 'master'] as const,
} as const;

Client Mutation Hook Client

Client-side only — handles the plan change mutation and cache invalidation.

src/modules/plans/hooks/useChangePlan.ts 
'use client';

import { useMutation, useQueryClient } from '@tanstack/react-query';
import { CHANGE_PLAN } from '../graphql/plans.mutations';
import { useGraphQLClient } from '@/hooks/useGraphQLClient';
import { PLAN_QUERY_KEYS } from '../lib/plans.constants';
import type { ActivePlan } from '../types/plans.types';

interface ChangePlanVariables {
  planId: string;
}

export function useChangePlan(locationId: string) {
  const queryClient = useQueryClient();
  const graphqlClient = useGraphQLClient();

  return useMutation<ActivePlan, Error, ChangePlanVariables>({
    mutationFn: async ({ planId }) => {
      const data = await graphqlClient.request(CHANGE_PLAN, {
        locationId,
        planId,
      });
      return data.changePlan;
    },
    onSuccess: () => {
      // Invalidate so active plan refetches from server
      queryClient.invalidateQueries({
        queryKey: PLAN_QUERY_KEYS.activePlan(locationId),
      });
    },
  });
}

Plan Selection Page Server Component

Fetches all data server-side in parallel, passes down to the Shell client component.

💡 Always use Promise.all for parallel server fetches — never await them sequentially. Sequential awaits add latency equal to the sum of all requests. Parallel keeps it to the slowest one.
src/app/(protected)/location/[slug]/[fn]/plan-selection/page.tsx 
import { notFound } from 'next/navigation';
import { getActivePlan, getMasterPlans } from
  '@/modules/plans/actions/server-plans';
import { PlanShell } from
  '@/modules/plans/components/PlanShell';

interface Props {
  params: Promise<{ slug: string; fn: string }>;
}

export default async function PlanSelectionPage({ params }: Props) {
  const { slug } = await params;

  // ✅ Parallel fetch — never sequential awaits
  const [activePlanResult, masterPlansResult] = await Promise.all([
    getActivePlan(slug),
    getMasterPlans(),
  ]);

  // Both failed — nothing useful to render
  if (activePlanResult.error && masterPlansResult.error) {
    notFound();
  }

  return (
    <PlanShell
      activePlan={activePlanResult.activePlan}
      masterPlans={masterPlansResult.plans}
      locationId={slug}
    />
  );
}

[fn] Router — Replace Switch Server Component

Scalable dynamic map pattern. Adding a new function = one line. Zero logic changes.

⚠️ This replaces your existing switch statement in src/app/(protected)/location/[slug]/[fn]/page.tsx. The map pattern scales to 20+ functions without touching routing logic.
src/app/(protected)/location/[slug]/[fn]/page.tsx 
import { notFound } from 'next/navigation';
import { getNodeAndPathBySlug } from
  '@/modules/locations/lib/slug';
import { getLocationsTree } from
  '@/modules/locations/actions/server-locations';

// ─── Import each function page ───────────────────────
import { Config } from './config/page';
import PlanSelectionPage from './plan-selection/page';
// import TicketsPage   from './tickets/page';
// import EventsPage    from './events/page';
// Add one line per new function — nothing else changes

// ─── Map: url segment → page component ───────────────
const FUNCTION_PAGE_MAP = {
  'config':           Config,
  'plan-selection':   PlanSelectionPage,
  // 'tickets':        TicketsPage,
  // 'events':         EventsPage,
} as const;

type FunctionKey = keyof typeof FUNCTION_PAGE_MAP;

interface Props {
  params: Promise<{ slug: string; fn: string }>;
}

export default async function LocationFunctionPage({ params }: Props) {
  const { slug, fn } = await params;

  // Validate the location exists in the tree
  const tree = await getLocationsTree();
  const found = getNodeAndPathBySlug(tree, slug);
  if (!found) notFound();

  // Resolve the function page from the map
  const PageComponent = FUNCTION_PAGE_MAP[fn as FunctionKey];
  if (!PageComponent) notFound();

  return <PageComponent params={Promise.resolve({ slug, fn })} />;
}
