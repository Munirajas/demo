Amenities creation/edit wizard — component reference (v2)

Step 1: the drawer's amenity-type dropdown decides what's being created/edited. Step 2: the matching type component mounts — General Admission (single step: name + description) or Scheduled Admission (two internal steps: shared General Info, then Scheduling). Each type component owns its own Save/Cancel/Next/Back controls and fires its own mutation, so adding a third amenity type later is additive, not a rewrite.

  • Fixed: type dropdown now lives only in the shell — switching type reliably swaps the whole child component instead of getting trapped in stale child state.
  • Fixed: footer Save/Cancel/Next/Back logic moved out of the shell entirely and into each type component, which now also owns its own insert.mutate / update.mutate call.
  • Added: a form registry (AMENITY_FORM_REGISTRY) mapping type → component — the single switch point for future amenity types.
  • Added: a shared AmenityTypeFormProps contract every type form implements, so the shell never needs type-specific logic.
  • Added: a downgrade warning when switching an existing Scheduled Admission back to General, since that discards the schedule.
Still open with BE: confirm whether the original recurrence rule (not just expanded timestamps) is persisted and returned, so an existing schedule can be prefilled on edit rather than starting blank. Marked with TODO in ScheduledAdmissionForm.tsx.

export function resolveWizardKind(
  amenityTypeId: number | '',
  amenityTypes: AmenityType[],
): AmenityTypeKey {
  const match = amenityTypes.find((t) => t.amenityTypeId === amenityTypeId);
  const name = match?.amenityTypeName?.toLowerCase() ?? '';
  if (name.includes('schedul') || name.includes('slot')) {
    return SCHEDULED_ADMISSION_KEY;
  }
  return GENERAL_ADMISSION_KEY;
}
lib/schedule-timestamp-generator.ts unchanged

Expands a recurrence rule into concrete ISO start/end timestamps using moment.js, per requirement #4. Includes a hard safety cap so a bad end-date input can't generate an unbounded list. Consumed by ScheduledAdmissionForm.tsx on save.

import moment from 'moment';
import type { ScheduleRuleValues, GeneratedTimestamp, WeekDay } from '../types/wizard-types';

const DAY_INDEX: Record<WeekDay, number> = {
  SUN: 0, MON: 1, TUE: 2, WED: 3, THU: 4, FRI: 5, SAT: 6,
};

// Hard safety cap so a mistyped end date (e.g. year 2099) can't hang the tab
// generating tens of thousands of rows client-side.
const MAX_OCCURRENCES_SAFEGUARD = 366;

function dayKeyFromIndex(index: number): WeekDay {
  return (Object.keys(DAY_INDEX) as WeekDay[]).find((k) => DAY_INDEX[k] === index) ?? 'SUN';
}

/**
 * Expands a recurrence rule into concrete start/end ISO timestamps.
 * This is a first-pass implementation -- the "every N weeks" interval jump
 * logic especially needs unit tests against real calendar edge cases
 * (month/year boundaries, DST) before shipping.
 */
export function generateScheduleTimestamps(rule: ScheduleRuleValues): GeneratedTimestamp[] {
  if (!rule.startDate) return [];

  const [startHour, startMinute] = rule.startTime.split(':').map(Number);
  const [endHour, endMinute] = rule.endTime.split(':').map(Number);

  const results: GeneratedTimestamp[] = [];
  const rangeStart = moment(rule.startDate, 'YYYY-MM-DD');
  const rangeEnd = rule.endMode === 'ON_DATE' && rule.endDate
    ? moment(rule.endDate, 'YYYY-MM-DD')
    : null;

  function pushOccurrence(day: moment.Moment) {
    const start = day.clone().hour(startHour).minute(startMinute).second(0).millisecond(0);
    const end = day.clone().hour(endHour).minute(endMinute).second(0).millisecond(0);
    results.push({ startDateTime: start.toISOString(), endDateTime: end.toISOString() });
  }

  if (rule.frequency === 'NONE') {
    pushOccurrence(rangeStart);
    return results;
  }

  let cursor = rangeStart.clone();
  let count = 0;
  const anchorWeekday = rangeStart.day();

  while (count < MAX_OCCURRENCES_SAFEGUARD) {
    if (rangeEnd && cursor.isAfter(rangeEnd, 'day')) break;
    if (rule.endMode === 'AFTER_OCCURRENCES' && rule.occurrences && count >= rule.occurrences) break;

    if (rule.frequency === 'DAILY') {
      pushOccurrence(cursor);
      count += 1;
      cursor = cursor.clone().add(rule.interval, 'days');
      continue;
    }

    // WEEKLY
    if (rule.daysOfWeek.includes(dayKeyFromIndex(cursor.day()))) {
      pushOccurrence(cursor);
      count += 1;
      if (rule.endMode === 'AFTER_OCCURRENCES' && rule.occurrences && count >= rule.occurrences) break;
    }

    cursor = cursor.clone().add(1, 'day');

    // Once we've scanned a full week, skip ahead (interval - 1) more weeks
    // to honor "recur every N weeks".
    if (cursor.day() === anchorWeekday && rule.interval > 1) {
      cursor = cursor.clone().add((rule.interval - 1) * 7, 'days');
    }
  }

  return results;
}

components/ScheduleBuilder.tsx unchanged

The recurrence UI: start/end date, start/end time, daily/weekly frequency, interval, day-of-week chips, and end condition (on date vs after N occurrences). Purely a controlled fields component — no buttons, no mutation logic, so it stays reusable regardless of who wraps it.

'use client';

import type { ScheduleRuleValues, WeekDay } from '../types/wizard-types';

const ALL_DAYS: { key: WeekDay; label: string }[] = [
  { key: 'SUN', label: 'Sun' }, { key: 'MON', label: 'Mon' },
  { key: 'TUE', label: 'Tue' }, { key: 'WED', label: 'Wed' },
  { key: 'THU', label: 'Thu' }, { key: 'FRI', label: 'Fri' },
  { key: 'SAT', label: 'Sat' },
];

interface Props {
  values: ScheduleRuleValues;
  onChange: (values: ScheduleRuleValues) => void;
  error?: string | null;
}

export function ScheduleBuilder({ values, onChange, error }: Props) {
  function toggleDay(day: WeekDay) {
    const has = values.daysOfWeek.includes(day);
    onChange({
      ...values,
      daysOfWeek: has ? values.daysOfWeek.filter((d) => d !== day) : [...values.daysOfWeek, day],
    });
  }

  return (
    <div className="flex flex-col gap-4">
      {error && <p className="text-sm text-red-600">{error}</p>}

      <div className="grid grid-cols-2 gap-3">
        <div className="flex flex-col gap-1">
          <label className="text-sm font-medium text-gray-700">Start date</label>
          <input
            type="date"
            className="w-full rounded-md border border-gray-300 px-3 py-2 text-sm"
            value={values.startDate}
            onChange={(e) => onChange({ ...values, startDate: e.target.value })}
          />
        </div>
        <div className="flex flex-col gap-1">
          <label className="text-sm font-medium text-gray-700">Recurrence</label>
          <select
            className="w-full rounded-md border border-gray-300 px-3 py-2 text-sm"
            value={values.frequency}
            onChange={(e) => onChange({ ...values, frequency: e.target.value as ScheduleRuleValues['frequency'] })}
          >
            <option value="NONE">Does not repeat</option>
            <option value="DAILY">Daily</option>
            <option value="WEEKLY">Weekly</option>
          </select>
        </div>
      </div>

      <div className="grid grid-cols-2 gap-3">
        <div className="flex flex-col gap-1">
          <label className="text-sm font-medium text-gray-700">Start time</label>
          <input
            type="time"
            className="w-full rounded-md border border-gray-300 px-3 py-2 text-sm"
            value={values.startTime}
            onChange={(e) => onChange({ ...values, startTime: e.target.value })}
          />
        </div>
        <div className="flex flex-col gap-1">
          <label className="text-sm font-medium text-gray-700">End time</label>
          <input
            type="time"
            className="w-full rounded-md border border-gray-300 px-3 py-2 text-sm"
            value={values.endTime}
            onChange={(e) => onChange({ ...values, endTime: e.target.value })}
          />
        </div>
      </div>

      {values.frequency !== 'NONE' && (
        <div className="flex flex-col gap-1">
          <label className="text-sm font-medium text-gray-700">Recur every</label>
          <div className="flex items-center gap-2">
            <input
              type="number"
              min={1}
              className="w-20 rounded-md border border-gray-300 px-3 py-2 text-sm"
              value={values.interval}
              onChange={(e) => onChange({ ...values, interval: Number(e.target.value) || 1 })}
            />
            <span className="text-sm text-gray-600">{values.frequency === 'DAILY' ? 'day(s)' : 'week(s)'}</span>
          </div>
        </div>
      )}

      {values.frequency === 'WEEKLY' && (
        <div className="flex flex-col gap-1">
          <label className="text-sm font-medium text-gray-700">Repeat on</label>
          <div className="flex flex-wrap gap-2">
            {ALL_DAYS.map((day) => {
              const active = values.daysOfWeek.includes(day.key);
              return (
                <button
                  key={day.key}
                  type="button"
                  onClick={() => toggleDay(day.key)}
                  className={`text-xs px-3 py-1.5 rounded-full border ${
                    active ? 'bg-blue-600 text-white border-blue-600' : 'bg-white text-gray-600 border-gray-300'
                  }`}
                >
                  {day.label}
                </button>
              );
            })}
          </div>
        </div>
      )}

      {values.frequency !== 'NONE' && (
        <div className="flex flex-col gap-2">
          <label className="text-sm font-medium text-gray-700">Ends</label>
          <div className="flex items-center gap-2">
            <input
              type="radio"
              id="endOnDate"
              checked={values.endMode === 'ON_DATE'}
              onChange={() => onChange({ ...values, endMode: 'ON_DATE' })}
            />
            <label htmlFor="endOnDate" className="text-sm text-gray-700 w-16">On date</label>
            <input
              type="date"
              disabled={values.endMode !== 'ON_DATE'}

components/StepIndicator.tsx renamed / simplified

Renamed from StepperHeader. Now purely presentational — "Step X of Y", progress bar, and clickable chips for already-visited steps. No footer, no button logic; the owning type component decides what "Back"/"Next" actually do.

'use client';

export interface StepIndicatorStep {
  id: string;
  label: string;
  status: 'done' | 'active' | 'pending';
}

interface Props {
  steps: StepIndicatorStep[];
  onStepClick?: (index: number) => void;
}

export function StepIndicator({ steps, onStepClick }: Props) {
  if (steps.length <= 1) return null;
  const activeIndex = steps.findIndex((s) => s.status === 'active');

  return (
    <div className="mb-2">
      <div className="flex items-center justify-between mb-1">
        <span className="text-sm font-medium text-gray-900">
          Step {activeIndex + 1} of {steps.length}
        </span>
        <span className="text-sm text-gray-500">{steps[activeIndex]?.label}</span>
      </div>
      <div className="h-1 w-full rounded bg-gray-100 overflow-hidden mb-2">
        <div
          className="h-full bg-blue-600 transition-all"
          style={{ width: `${((activeIndex + 1) / steps.length) * 100}%` }}
        />
      </div>
      <div className="flex items-center gap-2">
        {steps.map((step, index) => {
          const clickable = step.status !== 'pending' && !!onStepClick;
          return (
            <button
              key={step.id}
              type="button"
              disabled={!clickable}
              onClick={() => clickable && onStepClick!(index)}
              className={`text-xs px-2 py-1 rounded ${
                step.status === 'active'
                  ? 'bg-blue-50 text-blue-700 font-medium'
                  : clickable
                  ? 'text-gray-600 hover:bg-gray-50'
                  : 'text-gray-300 cursor-not-allowed'
              }`}
            >
              {step.status === 'done' ? '\u2713 ' : ''}
              {step.label}
            </button>
          );
        })}
      </div>
    </div>
  );
}

components/forms/GeneralAdmissionForm.tsx changed

Single-step form. Owns its own Save/Cancel buttons and fires insert.mutate / update.mutate directly based on isEditing.

'use client';

import { useState } from 'react';
import { useInsertAmenity } from '../../hooks/useInsertAmenity';
import { useUpdateAmenity } from '../../hooks/useUpdateAmenity';
import { GeneralInfoFields } from '../GeneralInfoFields';
import type { AmenityTypeFormProps } from './types';

export function GeneralAdmissionForm({
  generalInfo,
  onGeneralInfoChange,
  isEditing,
  amenity,
  locationFunctionId,
  locationSlug,
  onClose,
}: AmenityTypeFormProps) {
  const [error, setError] = useState<string | null>(null);

  const insert = useInsertAmenity(locationFunctionId, locationSlug);
  const update = useUpdateAmenity(locationFunctionId, locationSlug);
  const isSaving = insert.isPending || update.isPending;

  function validate(): boolean {
    if (!generalInfo.amenityTypeId) {
      setError('Please select an amenity type.');
      return false;
    }
    if (!generalInfo.amenityName.trim()) {
      setError('Name is required.');
      return false;
    }
    setError(null);
    return true;
  }

  function handleSave() {
    if (!validate()) return;

    const payload = {
      amenityTypeId: Number(generalInfo.amenityTypeId),
      amenityName: generalInfo.amenityName.trim(),
      amenityDescription: generalInfo.amenityDescription.trim(),
    };

    // NOTE: confirm exact field names against InsertAmenityInput /
    // UpdateAmenityInput -- placeholders until BE schema is finalized.
    if (isEditing && amenity) {
      update.mutate(
        { amenityId: amenity.amenityId, locationFunctionId, ...payload },
        { onSuccess: onClose },
      );
    } else {
      insert.mutate(
        { locationFunctionId, ...payload },
        { onSuccess: onClose },
      );
    }
  }

  return (
    <div className="flex flex-col gap-4">
      <GeneralInfoFields values={generalInfo} onChange={onGeneralInfoChange} error={error} />

      <div className="flex justify-end gap-2 pt-3 border-t border-gray-100 mt-1">
        <button
          type="button"
          onClick={onClose}
          className="px-4 py-2 text-sm rounded-md border border-gray-300 text-gray-700"
        >
          Cancel
        </button>
        <button
          type="button"
          onClick={handleSave}
          disabled={isSaving}
          className="px-4 py-2 text-sm rounded-md bg-blue-600 text-white disabled:opacity-60"
        >
          {isSaving ? 'Saving...' : 'Save'}
        </button>
      </div>
    </div>
  );
}

              className="px-4 py-2 text-sm rounded-md border border-gray-300 text-gray-700"
            >
              Cancel
            </button>
            <button
              type="button"
              onClick={handleNext}
              className="px-4 py-2 text-sm rounded-md bg-blue-600 text-white"
            >
              Next
            </button>
          </div>
        </>
      ) : (
        <>
          {error && <p className="text-sm text-red-600">{error}</p>}
          <ScheduleBuilder values={scheduleRule} onChange={setScheduleRule} />
          <div className="flex justify-between gap-2 pt-3 border-t border-gray-100 mt-1">
            <button
              type="button"
              onClick={() => setStep('general')}
              className="px-4 py-2 text-sm rounded-md border border-gray-300 text-gray-700"
            >
              Back
            </button>
            <div className="flex gap-2">
              <button
                type="button"
                onClick={onClose}
                className="px-4 py-2 text-sm rounded-md border border-gray-300 text-gray-700"
              >
                Cancel
              </button>
              <button
                type="button"
                onClick={handleSave}
                disabled={isSaving}
                className="px-4 py-2 text-sm rounded-md bg-blue-600 text-white disabled:opacity-60"
              >
                {isSaving ? 'Saving...' : 'Save'}
              </button>
            </div>
          </div>
        </>
      )}
    </div>
  );
}

components/AmenityFormDrawer.tsx changed

The single component to import for both create and edit. Owns only the amenity type dropdown and looks up the matching form from the registry — no footer, no mutation logic, no step logic. key={wizardKind} on the mounted form ensures switching type cleanly resets type-specific state.

Assumes Drawer renders plain children when no tabs/footerActions props are passed. If your @hlt/react-component-library version requires one of those, this needs a small wrapper adjustment. 
'use client';

import { useEffect, useMemo, useState } from 'react';
import { Drawer } from '@hlt/react-component-library';
import { Amenity, AmenityType } from '@/types/amenities';
import { AMENITY_FORM_REGISTRY } from './forms/amenity-form-registry';
import {
  EMPTY_GENERAL_INFO,
  GENERAL_ADMISSION_KEY,
  SCHEDULED_ADMISSION_KEY,
  resolveWizardKind,
  type GeneralInfoValues,
} from '../types/wizard-types';

interface Props {
  isOpen: boolean;
  onClose: () => void;
  locationFunctionId: number;
  locationSlug: string;
  amenityTypes: AmenityType[];
  amenity?: Amenity | null;
}

export function AmenityFormDrawer({
  isOpen,
  onClose,
  locationFunctionId,
  locationSlug,
  amenityTypes,
  amenity = null,
}: Props) {
  const isEditing = amenity !== null;

  const [generalInfo, setGeneralInfo] = useState<GeneralInfoValues>(EMPTY_GENERAL_INFO);

  // Capture the type this amenity started as, so we can warn on downgrade.
  const originalWizardKind = useMemo(
    () => (amenity ? resolveWizardKind(amenity.amenityTypeId, amenityTypes) : null),
    [amenity, amenityTypes],
  );

  useEffect(() => {
    if (!isOpen) return;
    setGeneralInfo(
      amenity
        ? {
            amenityTypeId: amenity.amenityTypeId,
            amenityName: amenity.amenityName,
            amenityDescription: amenity.amenityDescription ?? '',
          }
        : EMPTY_GENERAL_INFO,
    );
  }, [isOpen, amenity]);

  const wizardKind = resolveWizardKind(generalInfo.amenityTypeId, amenityTypes);
  const ActiveForm = AMENITY_FORM_REGISTRY[wizardKind];

  const showDowngradeWarning =
    isEditing &&
    originalWizardKind === SCHEDULED_ADMISSION_KEY &&
    wizardKind === GENERAL_ADMISSION_KEY;

  return (
    <Drawer
      title={isEditing ? 'Edit amenity' : 'Add amenity'}
      width="sm"
      isOpen={isOpen}
      onClose={onClose}
    >
      <div className="flex flex-col gap-4">
        <div className="flex flex-col gap-1">
          <label className="text-sm font-medium text-gray-700">Amenity type</label>
          <select
            className="w-full rounded-md border border-gray-300 bg-white px-3 py-2 text-sm text-gray-900 focus:outline-none focus:ring-2 focus:ring-blue-500"
            value={generalInfo.amenityTypeId}
            onChange={(e) =>
              setGeneralInfo((prev) => ({
                ...prev,
                amenityTypeId: e.target.value ? Number(e.target.value) : '',
              }))
            }
          >
            <option value="">Select...</option>
            {amenityTypes.map((t) => (
              <option key={t.amenityTypeId} value={t.amenityTypeId}>
                {t.amenityTypeName}
              </option>
            ))}
          </select>
          <p className="text-xs text-gray-500">
            Changing this switches the form below between General and Scheduled admission.
          </p>
        </div>

        {showDowngradeWarning && (
          <p className="text-sm text-amber-700 bg-amber-50 border border-amber-200 rounded-md px-3 py-2">
            Switching back to General Admission will remove the existing schedule for this amenity.
          </p>
        )}

        {generalInfo.amenityTypeId !== '' && ActiveForm && (
          <ActiveForm
            key={wizardKind}
            generalInfo={generalInfo}
            onGeneralInfoChange={setGeneralInfo}
            isEditing={isEditing}
            amenity={amenity}
            locationFunctionId={locationFunctionId}
            locationSlug={locationSlug}
            onClose={onClose}
          />
        )}
      </div>
    </Drawer>
  );
}
