Server vs Client decision
SERVER (no 'use client'):
  (dashboard)/layout.tsx    auth check, providers
  (dashboard)/page.tsx      auto-redirect to first location
  [fn]/layout.tsx           PARALLEL fetch tree+functions
  any/page.tsx              fetch function-specific data

CLIENT ('use client'):
  DashboardShell.tsx        three-panel layout, search state
  LocationTree.tsx          expand/collapse state, router.push
  FunctionNav.tsx           useParams for active highlight
  Breadcrumbs.tsx           Link components

Why this split:
  Server fetches data = instant, no loading state
  Client handles interaction = expand, navigate, highlight

// ── Helpers ───────────────────────────────────────────────────
export function findLocationById(
  nodes: LocationNode[],
  id: string
): LocationNode | null {
  for (const node of nodes) {
    if (node.id === id) return node;
    if (node.children?.length) {
      const found = findLocationById(node.children, id);
      if (found) return found;
    }
  }
  return null;
}

// Get first selectable location (skip client node)
export function getFirstLocation(
  nodes: LocationNode[]
): LocationNode | null {
  for (const node of nodes) {
    if (node.type === 'location') return node;
    if (node.children?.length) {
      const found = getFirstLocation(node.children);
      if (found) return found;
    }
  }
  return null;
}

// Build breadcrumb path for a location
export function getLocationPath(
  nodes: LocationNode[],
  targetId: string,
  path: LocationNode[] = []
): LocationNode[] | null {
  for (const node of nodes) {
    const current = [...path, node];
    if (node.id === targetId) return current;
    if (node.children?.length) {
      const found = getLocationPath(node.children, targetId, current);
      if (found) return found;
    }
  }
  return null;
}

CLIENT LocationTree.tsx
Expand/collapse state. Navigates keeping current fn slug. Client nodes toggle only.
'use client';

import { useState }                from 'react';
import { useRouter, useParams }    from 'next/navigation';
import type { LocationNode }       from '../types/location.types';

interface Props {
  nodes:      LocationNode[];
  activeId?:  string;
  depth?:     number;
}

export function LocationTree({ nodes, activeId, depth = 0 }: Props) {
  if (!nodes.length) return null;
  return (
    <ul className={depth > 0 ? 'ml-4' : ''}>
      {nodes.map(node => (
        <LocationTreeNode
          key={node.id}
          node={node}
          activeId={activeId}
          depth={depth}
        />
      ))}
    </ul>
  );
}

function LocationTreeNode({
  node, activeId, depth,
}: {
  node: LocationNode; activeId?: string; depth: number;
}) {
  const router      = useRouter();
  const params      = useParams<{ locationId?: string; fn?: string }>();
  const hasChildren = Boolean(node.children?.length);
  const isClient    = node.type === 'client';
  const isActive    = node.id === activeId;

  // Auto-expand if active is a descendant
  const [expanded, setExpanded] = useState(() =>
    isClient || hasDescendant(node, activeId)
  );

  function handleClick() {
    if (isClient) {
      setExpanded(x => !x);
      return;
    }
    // Navigate to location — keep current fn or default to config
    const fn = params?.fn ?? 'config';
    router.push(`/locations/${node.id}/${fn}`);
  }

  return (
    <li className="my-0.5">
      <button
        onClick={handleClick}
        className={`
          w-full flex items-center gap-1.5 px-2 py-1.5 rounded-md
          text-sm text-left transition-colors
          ${isActive
            ? 'bg-blue-600 text-white font-medium'
            : isClient
              ? 'text-gray-700 font-semibold cursor-default'
              : 'text-gray-600 hover:bg-gray-100 hover:text-gray-900'
          }
        `}
      >
        {/* Expand/collapse indicator */}
        {hasChildren ? (
          <svg
            onClick={e => { e.stopPropagation(); setExpanded(x => !x); }}
            width="10" height="10" viewBox="0 0 24 24" fill="none"
            className={`shrink-0 opacity-50 transition-transform
              ${expanded ? 'rotate-90' : ''}`}
          >
            <path d="M9 18l6-6-6-6" stroke="currentColor"
              strokeWidth="2.5" strokeLinecap="round"/>
          </svg>
        ) : (
          <span className="w-2.5 shrink-0" />
        )}
        <span className="truncate">{node.label}</span>

        {/* Type badge for client node */}
        {isClient && (
          <span className="ml-auto text-[9px] font-bold tracking-wide
            text-gray-400 uppercase shrink-0">
            CLIENT
          </span>
        )}
      </button>

      {hasChildren && expanded && (
        <LocationTree
          nodes={node.children!}
          activeId={activeId}
          depth={depth + 1}
        />
      )}
    </li>
  );
}

function hasDescendant(node: LocationNode, id?: string): boolean {
  if (!id) return false;
  if (node.id === id) return true;
  return node.children?.some(c => hasDescendant(c, id)) ?? false;
}

import Link            from 'next/link';
import { useParams }   from 'next/navigation';
import type { LocationFunction } from '../types/location.types';

interface Props {
  locationId: string;
  functions:  LocationFunction[];
}

export function FunctionNav({ locationId, functions }: Props) {
  const params    = useParams<{ fn?: string }>();
  const activeFn  = params?.fn ?? 'config';
  const enabled   = functions.filter(f => f.enabled);

  if (!enabled.length) return (
    <div className="p-4 text-sm text-gray-400">
      No functions available.
    </div>
  );

  return (
    <nav className="py-2">
      {enabled.map(fn => {
        const isActive = fn.slug === activeFn;
        return (
          <Link
            key={fn.slug}
            href={`/locations/${locationId}/${fn.slug}`}
            className={`
              block px-4 py-2.5 text-sm transition-colors
              ${isActive
                ? 'text-blue-600 font-medium border-l-2 border-blue-600 bg-blue-50'
                : 'text-gray-700 hover:bg-gray-50 hover:text-gray-900 border-l-2 border-transparent'
              }
            `}
          >
            {fn.label}
          </Link>
        );
      })}
    </nav>
  );
}

import Link from 'next/link';
import type { BreadcrumbItem } from '@/modules/locations/types/location.types';

interface Props {
  items: BreadcrumbItem[];
}

export function Breadcrumbs({ items }: Props) {
  if (!items.length) return null;

  return (
    <nav aria-label="breadcrumb" className="flex items-center gap-1 flex-wrap">
      {items.map((item, i) => {
        const isLast = i === items.length - 1;
        return (
          <span key={i} className="flex items-center gap-1">
            {i > 0 && (
              <svg width="12" height="12" viewBox="0 0 24 24" fill="none"
                className="text-gray-300 shrink-0">
                <path d="M9 18l6-6-6-6" stroke="currentColor"
                  strokeWidth="2" strokeLinecap="round"/>
              </svg>
            )}
            {item.href && !isLast ? (
              <Link
                href={item.href}
                className="text-sm text-gray-500 hover:text-gray-700
                  transition-colors"
              >
                {item.label}
              </Link>
            ) : (
              <span className={`text-sm ${isLast
                ? 'text-gray-900 font-medium'
                : 'text-gray-500'
              }`}>
                {item.label}
              </span>
            )}
          </span>
        );
      })}
    </nav>
  );
}

// ── Tree search filter ────────────────────────────────────────
function filterTree(nodes: LocationNode[], term: string): LocationNode[] {
  return nodes.reduce<LocationNode[]>((acc, node) => {
    const matches          = node.label.toLowerCase().includes(term);
    const filteredChildren = node.children
      ? filterTree(node.children, term)
      : [];
    if (matches || filteredChildren.length) {
      acc.push({ ...node, children: filteredChildren });
    }
    return acc;
  }, []);
}

//   Passes everything to DashboardShell (client)
//   DashboardShell renders 3-panel layout
//
// PERFORMANCE:
//   React cache() = tree only fetched once even if
//   called from multiple server components
//   Data arrives WITH html = zero loading flash
// ─────────────────────────────────────────────────────────────

import { notFound }               from 'next/navigation';
import { DashboardShell }         from '@/components/layout/DashboardShell';
import {
  getLocationsTree,
  getLocationFunctions,
} from '@/modules/locations/graphql/server-locations';
import {
  findLocationById,
  getLocationPath,
} from '@/modules/locations/graphql/mock-data';
import { FUNCTION_LABELS }        from '@/modules/locations/types/location.types';
import type { BreadcrumbItem }    from '@/modules/locations/types/location.types';

interface Props {
  children:    React.ReactNode;
  params:      Promise<{ locationId: string; fn: string }>;
}

export default async function LocationFnLayout({ children, params }: Props) {
  const { locationId, fn } = await params;

  // ── Parallel fetch — both fire simultaneously ─────────────
  const [tree, functions] = await Promise.all([
    getLocationsTree(),
    getLocationFunctions(locationId),
  ]);

  // Validate location exists
  const location = findLocationById(tree, locationId);
  if (!location) notFound();

  // ── Build breadcrumbs from tree path ──────────────────────
  // Derive the full path: Client > Venue > Location > Function
  const locationPath = getLocationPath(tree, locationId) ?? [];
  const fnLabel      = FUNCTION_LABELS[fn as keyof typeof FUNCTION_LABELS]
    ?? fn;

  const breadcrumbs: BreadcrumbItem[] = [
    // Location path items (all clickable → go to that location/config)
    ...locationPath.map((node, i) => ({
      label: node.label,
      href:  node.type === 'client'
        ? undefined  // client node not navigable
        : `/locations/${node.id}/config`,
    })),
    // Current function (not a link — current page)
    { label: fnLabel },
  ];

  return (
    <DashboardShell
      tree={tree}
      locationId={locationId}
      functions={functions}
      breadcrumbs={breadcrumbs}
      fnLabel={fnLabel}
    >
      {children}
    </DashboardShell>
  );
}
