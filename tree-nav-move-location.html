'use client';

import { useState, useMemo, useCallback } from 'react';
import { useRouter, usePathname }          from 'next/navigation';
import { TreeView }                        from '@hlt-web-components/tree-view'; // adjust to your actual import
import { DeleteLocationModal }             from './DeleteLocationModal';
import { MoveLocationModal }               from '../moveLocation/MoveLocationModal';
import {
  slugify,
  getNodeAndPathBySlug,
} from '@/modules/locations/lib/slug';
import type { TreeNode } from '@/types/locations';

// ── Active node detection from URL ────────────────────────────
// URL shape: /location/{location-name-slug}/{fn}
// Returns the matching node's id, or null if no location in path
// (e.g. on `/` or any non-location route)
function getActiveNodeId(data: TreeNode[], pathname: string | null): string | null {
  if (!pathname) return null;

  const match = pathname.match(/^\/location\/([^/]+)/);
  if (!match) return null;

  const slug  = match[1];
  const found = getNodeAndPathBySlug(data, slug);
  return found?.node.id ?? null;
}

type Props = {
  data: TreeNode[];
};

export function LocationTreeView({ data }: Props) {
  const router   = useRouter();
  const pathname = usePathname();

  const [deleteTarget, setDeleteTarget] = useState<TreeNode | null>(null);
  const [moveTarget,   setMoveTarget]   = useState<TreeNode | null>(null);

  // Resolve which node (if any) corresponds to the current URL
  const activeNodeId = useMemo(
    () => getActiveNodeId(data, pathname),
    [data, pathname]
  );

  // Default-expand the branch containing the active node
  const defaultExpandedIds = useMemo(() => {
    if (!activeNodeId) return ['1'];
    const found = getNodeAndPathBySlug(
      data,
      // re-derive slug for active node so we get its ancestor path
      data.length ? slugAtId(data, activeNodeId) ?? '' : ''
    );
    return found
      ? found.path.map(n => n.id)
      : ['1'];
  }, [activeNodeId, data]);

  // Navigate to /location/{slugified-name}/config
  // Only 'location' type nodes are navigable —
  // client/root node click is handled by TreeView's own
  // expand/collapse (no page for the root).
  const handleSelect = useCallback((node: TreeNode) => {
    if (node.type !== 'location') return;
    router.push(`/location/${slugify(node.label)}/config`);
  }, [router]);

  return (
    <nav aria-label="Clients and locations" className="min-h-0 flex-1">
      <TreeView
        key={pathname}
        className="max-h-[calc(100vh-140px)] overflow-auto"
        data={data}
        activeId={activeNodeId ?? undefined}
        defaultExpandedIds={defaultExpandedIds}
        onActiveChange={handleSelect}
        showMoreButton
        showSearchbar
        onEdit={(node) => {
          // existing edit wiring — unchanged
        }}
        onDelete={(node) => setDeleteTarget(node)}
        onMove={(node) => setMoveTarget(node)}   // ← NEW
      />

      {deleteTarget && (
        <DeleteLocationModal
          locationId={deleteTarget.id}
          locationName={deleteTarget.label}
          onClose={() => setDeleteTarget(null)}
        />
      )}

      {moveTarget && (
        <MoveLocationModal
          node={moveTarget}
          onClose={() => setMoveTarget(null)}
        />
      )}
    </nav>
  );
}

// Helper: find the slug for a given node id (for default-expand path)
function slugAtId(nodes: TreeNode[], id: string): string | null {
  for (const n of nodes) {
    if (n.id === id) return slugify(n.label);
    if (n.children?.length) {
      const found = slugAtId(n.children, id);
      if (found) return found;
    }
  }
  return null;
}

export const MOVE_LOCATION_MUTATION = `
  mutation MoveLocation(
    $locationId:       Int!
    $parentLocationId: Int!
  ) {
    moveLocation(request: {
      type:             "client"
      locationId:       $locationId
      parentLocationId: $parentLocationId
    }) {
      success
      message
      data
    }
  }
`;

export interface MoveLocationInput {
  locationId:       number;
  parentLocationId: number;
}

export interface MoveLocationResponse {
  moveLocation: {
    success: boolean;
    message: string;
    data:    unknown;
  };
}
CLIENT src/modules/locations/hooks/useMoveLocation.ts
Mutation hook. success → toast + invalidate tree + close modal. error → toast + keep modal open.
'use client';

import { useQueryClient }  from '@tanstack/react-query';
import { toast }           from 'sonner';
import { useAuthMutation } from '@/modules/auth/hooks/useAuthQuery';
import { locationKeys }    from '@/modules/locations/hooks/useLocations';
import {
  MOVE_LOCATION_MUTATION,
  type MoveLocationInput,
  type MoveLocationResponse,
} from '../graphql/move-location.mutation';

export function useMoveLocation(onSuccess?: () => void) {
  const queryClient = useQueryClient();

  return useAuthMutation<MoveLocationResponse, MoveLocationInput>(
    MOVE_LOCATION_MUTATION,
    {
      onSuccess: (data) => {
        const result = data.moveLocation;

        if (!result.success) {
          toast.error(result.message || 'Failed to move location.');
          return; // keep modal open so user can retry
        }

        // Refresh sidebar tree — node now appears under new parent
        queryClient.invalidateQueries({
          queryKey: locationKeys.tree(),
        });

        toast.success('Location moved successfully.');
        onSuccess?.(); // close modal
      },

      onError: (error) => {
        toast.error(error.message || 'Something went wrong.');
      },
    }
  );
}

'use client';

// ─────────────────────────────────────────────────────────────
// MOVE LOCATION MODAL
// - Current location: read-only field
// - Parent location: dropdown (same source as CreateLocationModal —
//   flattenTreeForDropdown over existing tree cache)
// - Excludes the node itself + its own descendants from the
//   dropdown (can't move a location into itself/its children)
// - Move button → moveLocation mutation
// - success: toast + close modal | error: toast + keep modal open
// ─────────────────────────────────────────────────────────────

import { useForm }              from 'react-hook-form';
import { zodResolver }          from '@hookform/resolvers/zod';
import { z }                    from 'zod';
import { useLocationsTree }     from '@/modules/locations/hooks/useLocations';
import { useMoveLocation }      from '@/modules/locations/hooks/useMoveLocation';
import { flattenTreeForDropdown } from '@/modules/locations/lib/utils';
import type { TreeNode }        from '@/types/locations';

const schema = z.object({
  parentLocationId: z.string().min(1, 'Please select a parent location'),
});
type FormValues = z.infer<typeof schema>;

interface Props {
  node:    TreeNode;   // location being moved
  onClose: () => void;
}

// Collect ids of node + all its descendants — these can't be
// the new parent (would create a cycle)
function collectDescendantIds(node: TreeNode, ids = new Set<string>()): Set<string> {
  ids.add(node.id);
  for (const child of node.children ?? []) {
    collectDescendantIds(child, ids);
  }
  return ids;
}

export function MoveLocationModal({ node, onClose }: Props) {
  const { data: tree = [], isLoading: treeLoading } = useLocationsTree();

  const excludeIds = collectDescendantIds(node);
  const parentOptions = flattenTreeForDropdown(tree)
    .filter(opt => !excludeIds.has(opt.id));

  const moveLocation = useMoveLocation(onClose);
  const isLoading    = moveLocation.isPending;

  const {
    register,
    handleSubmit,
    formState: { errors },
  } = useForm<FormValues>({
    resolver: zodResolver(schema),
    defaultValues: { parentLocationId: '' },
  });

  function onSubmit(values: FormValues) {
    moveLocation.mutate({
      locationId:       Number(node.id),
      parentLocationId: Number(values.parentLocationId),
    });
  }

  return (
    <div
      className="fixed inset-0 z-50 flex items-center justify-center bg-black/40"
      onClick={(e) => e.target === e.currentTarget && !isLoading && onClose()}
    >
      <div className="w-full max-w-md rounded-2xl bg-white shadow-2xl">

        {/* Header */}
        <div className="flex items-center justify-between border-b border-gray-100 px-6 py-4">
          <h2 className="text-base font-semibold text-gray-900">
            Move Location
          </h2>
          <button
            onClick={onClose}
            disabled={isLoading}
            className="text-gray-400 hover:text-gray-600 transition-colors disabled:opacity-40"
          >
            <svg width="16" height="16" viewBox="0 0 24 24" fill="none">
              <path d="M18 6L6 18M6 6l12 12"
                stroke="currentColor" strokeWidth="2" strokeLinecap="round"/>
            </svg>
          </button>
        </div>

        {/* Fields */}
        <div className="px-6 py-5 space-y-4">

          {/* Current location — read only */}
          <div>
            <label className="block text-xs font-medium text-gray-700 mb-1.5">
              Current Location
            </label>
            <input
              type="text"
              value={node.label}
              readOnly
              disabled
              className="w-full h-10 rounded-lg border border-gray-200
                bg-gray-50 px-3 text-sm text-gray-500 cursor-not-allowed"
            />
          </div>

          {/* New parent dropdown */}
          <div>
            <label className="block text-xs font-medium text-gray-700 mb-1.5">
              Move To (New Parent) <span className="text-red-500">*</span>
            </label>
            <select
              disabled={treeLoading || isLoading}
              className={`w-full h-10 rounded-lg border px-3 text-sm bg-white
                outline-none transition-colors
                focus:border-blue-400 focus:ring-1 focus:ring-blue-400/30
                disabled:bg-gray-50 disabled:cursor-not-allowed
                ${errors.parentLocationId ? 'border-red-300' : 'border-gray-200'}`}
              {...register('parentLocationId')}
            >
              <option value="">
                {treeLoading ? 'Loading...' : 'Select new parent location'}
              </option>
              {parentOptions.map((option) => (
                <option key={option.id} value={option.id}>
                  {option.indent}{option.label}
                </option>
              ))}
            </select>
            {errors.parentLocationId && (
              <p className="mt-1 text-xs text-red-500">
                {errors.parentLocationId.message}
              </p>
            )}
          </div>

        </div>

        {/* Footer */}
        <div className="flex gap-3 border-t border-gray-100 px-6 py-4">
          <button
            type="button"
            onClick={onClose}
            disabled={isLoading}
            className="flex-1 h-10 rounded-lg border border-gray-200
              text-sm text-gray-600 hover:bg-gray-50
              transition-colors disabled:opacity-40"
          >
            Cancel
          </button>
          <button
            type="button"
            onClick={handleSubmit(onSubmit)}
            disabled={isLoading || treeLoading}
            className="flex-1 h-10 rounded-lg bg-blue-600 hover:bg-blue-700
              text-white text-sm font-medium transition-colors
              disabled:opacity-60 flex items-center justify-center gap-2"
          >
            {isLoading ? (
              <>
                <svg className="w-4 h-4 animate-spin" viewBox="0 0 24 24" fill="none">
                  <circle className="opacity-25" cx="12" cy="12" r="10"
                    stroke="currentColor" strokeWidth="4"/>
                  <path className="opacity-75" fill="currentColor"
                    d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4z"/>
                </svg>
                Moving...
              </>
            ) : (
              'Move Location'
            )}
          </button>
        </div>

      </div>
    </div>
  );
}
