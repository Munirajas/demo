/**
 * Typed error classes for GraphQL fetch failures.
 * Import these in server actions to handle specific error types.
 */

// Thrown when BE returns HTTP 429
export class RateLimitError extends Error {
  readonly statusCode = 429;
  constructor(message = 'Too many requests') {
    super(message);
    this.name = 'RateLimitError';
  }
}

// Thrown when session is expired or token is invalid
export class SessionExpiredError extends Error {
  readonly statusCode = 401;
  constructor(message = 'Session expired') {
    super(message);
    this.name = 'SessionExpiredError';
  }
}

// Thrown for all other GraphQL/network failures
export class GqlFetchError extends Error {
  constructor(
    message: string,
    public readonly statusCode?: number,
  ) {
    super(message);
    this.name = 'GqlFetchError';
  }
}

// Type guard helpers — use in catch blocks
export function isRateLimitError(e: unknown): e is RateLimitError {
  return e instanceof RateLimitError;
}

export function isSessionExpiredError(e: unknown): e is SessionExpiredError {
  return e instanceof SessionExpiredError;
}

server-gql-fetcher.ts TypeScriptUpdated

Keeps your existing GraphQLClient pattern. Adds rate limit detection + retry with exponential backoff.

⚠️
Your existing pattern is kept intact. Only additions: rate limit check before throwing, retry logic, and typed error throwing. Your isAuthError, buildClient, token refresh flow — all unchanged.
src/lib/graphql/server-gql-fetcher.ts 
'use server';

import { GraphQLClient, ClientError } from 'graphql-request';
import { GQL_ENDPOINT } from '@/lib/constants';
import { getAccessToken } from '@/modules/auth/lib/cookies';
import { refreshAccessToken } from '@/modules/auth/lib/refresh';
import {
  RateLimitError,
  SessionExpiredError,
  GqlFetchError,
} from './gql-errors';

// ─── Config ────────────────────────────────────────────
const MAX_RETRIES = 2;
const BASE_RETRY_DELAY_MS = 400;

function wait(ms: number) {
  return new Promise((r) => setTimeout(r, ms));
}

// ─── YOUR existing buildClient — unchanged ─────────────
function buildClient(token: string): GraphQLClient {
  return new GraphQLClient(GQL_ENDPOINT, {
    headers: { Authorization: `Bearer ${token}` },
  });
}

// ─── YOUR existing isAuthError — unchanged ─────────────
function isAuthError(error: unknown): boolean {
  if (error instanceof ClientError) {
    return error.response.status === 401;
  }
  return false;
}

// ─── NEW: rate limit detection ─────────────────────────
function isRateLimitResponse(error: unknown): boolean {
  if (error instanceof ClientError) {
    return error.response.status === 429;
  }
  // Also catch plain text "Too many r..." parse errors
  if (error instanceof SyntaxError) {
    return error.message.includes('Too many');
  }
  return false;
}

// ─── UPDATED serverGqlFetch ────────────────────────────
export async function serverGqlFetch<
  TData,
  TVars extends Record<string, unknown> = Record<string, unknown>
>(
  query: string,
  variables?: TVars,
  retryCount = 0,
): Promise<TData> {
  let token: string | undefined = await getAccessToken();
  let alreadyRefreshed = false;

  if (!token) {
    token = (await refreshAccessToken()) ?? undefined;
    alreadyRefreshed = true;
    if (!token) throw new SessionExpiredError();
  }

  try {
    const result = await buildClient(token).request<TData>(
      query,
      variables,
    );
    return result;

  } catch (error) {

    // ── Rate limit: retry with exponential backoff ──────
    if (isRateLimitResponse(error)) {
      if (retryCount < MAX_RETRIES) {
        const delay = BASE_RETRY_DELAY_MS * Math.pow(2, retryCount);
        console.warn(
          `[serverGqlFetch] Rate limited. Retry ${retryCount + 1}/${MAX_RETRIES}`
          + ` in ${delay}ms`
        );
        await wait(delay);
        // Recursive retry — passes retryCount up
        return serverGqlFetch<TData, TVars>(
          query, variables, retryCount + 1
        );
      }
      // All retries exhausted — throw typed error
      console.error('[serverGqlFetch] Rate limit exceeded after retries');
      throw new RateLimitError();
    }

    // ── Auth error: try token refresh once ──────────────
    if (isAuthError(error)) {
      if (alreadyRefreshed) throw new SessionExpiredError();
      const newToken = (await refreshAccessToken()) ?? undefined;
      if (!newToken) throw new SessionExpiredError();
      return buildClient(newToken).request<TData>(query, variables);
    }

    // ── Everything else ─────────────────────────────────
    console.error('[serverGqlFetch] Unexpected error:', error);
    throw new GqlFetchError(
      error instanceof Error
        ? error.message
        : 'GraphQL request failed'
    );
  }
}

src/modules/locations/actions/server-location-functions.ts 
'use server';

import { cache } from 'react';
import { serverGqlFetch } from '@/lib/graphql/server-gql-fetcher';
import { LocationFunction } from '@/types/locations';
import { FUNCTION_MAPPING } from '../lib/constants';
import { GET_LOCATION_FUNCTIONS } from '../graphql/locations.mutations';
import {
  isRateLimitError,
  isSessionExpiredError,
} from '@/lib/graphql/gql-errors';

interface GetLocationConfigResponse {
  getLocationFunctions: {
    success: boolean;
    message: string;
    data: LocationFunction[];
  };
}

export const getLocationFunctions = cache(
  async (locationId: string): Promise<LocationFunction[]> => {
    try {
      const response = await serverGqlFetch<GetLocationConfigResponse>(
        GET_LOCATION_FUNCTIONS,
        { type: 'client', locationId: Number(locationId) },
      );

      if (!response.getLocationFunctions?.success) {
        return [];
      }

      const functionsData =
        response.getLocationFunctions.data;
      const apiIds = new Set(
        functionsData.map((i: LocationFunction) =>
          i.Function_Type_Id
        ),
      );
      return FUNCTION_MAPPING.filter((fn) =>
        apiIds.has(fn.Function_Type_Id),
      );

    } catch (error) {
      // ── Rate limit: return empty, don't crash ──────────
      if (isRateLimitError(error)) {
        console.warn('[getLocationFunctions] Rate limited — returning []');
        return [];
      }
      // ── Session expired: re-throw so middleware handles ─
      if (isSessionExpiredError(error)) throw error;

      // ── Anything else: log and degrade gracefully ───────
      console.error('[getLocationFunctions]', error);
      return [];
    }
  },
);

NavLink.tsx — Shared TSXClientNew

Reusable nav link with built-in pending state via useTransition. Use this in FunctionNav AND LocationTreeView.

💡
Why useTransition here? loading.tsx only fires on full segment navigation. When switching between functions (same layout, different fn), there's no loading.tsx trigger — useTransition fills this gap and shows a spinner on the clicked item only.
src/components/nav/NavLink.tsx 
'use client';

import { useTransition } from 'react';
import { useRouter, usePathname } from 'next/navigation';

interface NavLinkProps {
  href: string;
  label: string;
  className?: string;
  activeClassName?: string;
  inactiveClassName?: string;
}

/**
 * NavLink — nav button with useTransition pending state.
 *
 * Shows a spinner on the clicked item while the next
 * server page loads. Works for ALL nav types:
 *   - function switching (FunctionNav)
 *   - location switching (LocationTreeView)
 *
 * PATTERN: use this instead of plain  in any nav
 * component that needs a loading indicator.
 */
export function NavLink({
  href,
  label,
  activeClassName = 'border-l-2 border-blue-500 bg-blue-50 font-medium text-blue-700',
  inactiveClassName = 'text-gray-600 hover:bg-gray-50 hover:text-gray-900',
}: NavLinkProps) {
  const router = useRouter();
  const pathname = usePathname();
  const [isPending, startTransition] = useTransition();

  const isActive = pathname === href;

  function handleClick() {
    if (isActive) return;
    startTransition(() => {
      router.push(href);
    });
  }

  return (
    <button
      type="button"
      onClick={handleClick}
      disabled={isPending}
      className={[
        'flex w-full items-center justify-between',
        'px-4 py-2.5 text-sm transition-colors',
        isActive ? activeClassName : inactiveClassName,
        isPending ? 'opacity-60' : '',
      ].join(' ')}
      aria-current={isActive ? 'page' : undefined}
    >
      <span>{label}</span>

      {/* Spinner — only on the clicked item */}
      {isPending && (
        <svg
          className="h-3.5 w-3.5 animate-spin text-blue-500"
          viewBox="0 0 24 24"
          fill="none"
        >
          <circle
            className="opacity-25"
            cx="12" cy="12" r="10"
            stroke="currentColor" strokeWidth="4"
          />
          <path
            className="opacity-75"
            fill="currentColor"
            d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4z"
          />
        </svg>
      )}
    </button>
  );
}

LocationTreeView.tsx TSXClientUpdated

Location items now use NavLink too — spinner appears on clicked location during tree/layout reload.

src/modules/locations/components/LocationTreeView.tsx 
'use client';

import { useState } from 'react';
import { NavLink } from '@/components/nav/NavLink';
import { generateSlug } from '@/modules/locations/lib/utils';
import type { LocationNode } from '@/modules/locations/lib/utils';

interface LocationTreeViewProps {
  tree: LocationNode[];
}

// Recursive node component
function TreeNode({
  node,
  depth = 0,
}: {
  node: LocationNode;
  depth?: number;
}) {
  const [expanded, setExpanded] = useState(true);
  const hasChildren = (node.children?..length ?? 0) > 0;
  const slug = generateSlug(node.name);

  return (
    <div>
      <div
        className="flex items-center"
        style={{ paddingLeft: `${depth * 12}px` }}
      >
        {/* Expand/collapse toggle */}
        {hasChildren ? (
          <button
            type="button"
            onClick={() => setExpanded((e) => !e)}
            className="mr-1 flex h-4 w-4 flex-shrink-0
              items-center justify-center text-gray-400
              hover:text-gray-600"
          >
            {expanded ? '▾' : '▸'}
          </button>
        ) : (
          <span className="mr-1 w-4 flex-shrink-0" />
        )}

        {/* NavLink — spinner on click ✅ */}
        <div className="flex-1">
          <NavLink
            href={`/location/${slug}`}
            label={node.name}
            activeClassName="bg-blue-50 font-semibold text-blue-700"
            inactiveClassName="text-gray-700 hover:bg-gray-100"
          />
        </div>
      </div>

      {/* Children */}
      {hasChildren && expanded && (
        <div>
          {node.children!.map((child) => (
            <TreeNode
              key={child.id}
              node={child}
              depth={depth + 1}
            />
          ))}
        </div>
      )}
    </div>
  );
}

export function LocationTreeView({ tree }: LocationTreeViewProps) {
  if (!tree.length) {
    return (
      <p className="px-4 py-3 text-xs text-gray-400">
        No locations found
      </p>
    );
  }

  return (
    <div className="py-2">
      {tree.map((node) => (
        <TreeNode key={node.id} node={node} />
      ))}
    </div>
  );
}
